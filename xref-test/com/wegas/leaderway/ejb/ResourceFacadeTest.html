45          vdf.create(gameModel.getId(), res);
46  
47          // Test history
48          ResourceInstance resI = res.getInstance(player);
49          for (int i = 0; i < ResourceInstance.HISTORYSIZE + 10; i++) {
50              resI.setConfidence(i);

238 
239         // Create a task
240         TaskDescriptor task1 = new TaskDescriptor();
241         task1.setLabel("My task");
242         task1.setDefaultInstance(new TaskInstance());
243         task1.setScope(new TeamScope());
244         vdf.create(gameModel.getId(), task1);
245 
246         TaskDescriptor task2 = new TaskDescriptor();
247         task2.setLabel("My task");
248         task2.setDefaultInstance(new TaskInstance());
249         task2.setScope(new TeamScope());
250         vdf.create(gameModel.getId(), task2);
251 
252         TaskDescriptor task3 = new TaskDescriptor();
253         task3.setLabel("My task");
254         task3.setDefaultInstance(new TaskInstance());
255         task3.setScope(new TeamScope());
256         vdf.create(gameModel.getId(), task3);
257 
258         // Assign resource to task
259         resourceFacade.assign(res.getInstance(player).getId(), task1.getInstance(player).getId());
260         resourceFacade.assign(res.getInstance(player).getId(), task2.getInstance(player).getId());
261         Assignment assignment = resourceFacade.assign(res.getInstance(player).getId(), task3.getInstance(player).getId());
262 
263         //Move last assignement (pos 2) at pos (0)
264         resourceFacade.moveAssignment(assignment.getId(), 0);
265

267                 ((ResourceInstance) vif.find(res.getId(), player)).getAssignments().get(0).getId(),

269
