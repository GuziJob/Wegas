 13 YUI.add('wegas-popuplistener', function(Y) {
 14     "use strict";
 15     var stringToObject = function(o) {
 16         return (Y.Lang.isString(o)) ? {content: o} : o;
 17     },
 18             PopupListener = Y.Base.create("wegas-popuplistener", Y.Plugin.Base, [], {
 19         DEFAULT_CONFIG: function() {
 20             return {
 21                 align: {
 22                     node: this.get("host").get(this.get("alignAttr")),
 23                     points: [Y.WidgetPositionAlign.TC, Y.WidgetPositionAlign.TC]
 24                 },
 25                 buttons: {
 26                     footer: [
 27                         {
 28                             name: 'proceed',
 29                             label: 'OK',
 30                             action: "exit"
 31                         }
 32                     ]
 33                 },
 34                 modal: false,
 35                 centered: false,
 36                 width: "80%"
 37             };
 38         },
 39         handlers: [],
 40         initializer: function() {
 41             var bb = this.get("host").get(this.get("targetAttr"));
 42 
 43             this.handlers = [
 44                 bb.on("dom-message:showPopup", this._show, this),
 45                 bb.on("dom-message:error", this._system, this, "error"),
 46                 bb.on("dom-message:success", this._system, this, "success"),
 47                 bb.on("dom-message:info", this._system, this, "info"),
 48                 bb.on("dom-message:warn", this._system, this, "warn")
 49             ];
 50 
 51             this.onHostEvent("*:showOverlay", this.onShowOverlay);
 52             this.onHostEvent("*:hideOverlay", this.onHideOverlay);
 53             this.onHostEvent("*:message", this.onShowMessage);
 54 
 55             if (this.get("showServerMessages")) {
 56                 this.handlers.push(Y.Wegas.Facade.VariableDescriptor.on("popupEvent", function(e) {
 57                     this.get("host").showMessage("info", e.content);
 58                 }, this));
 59             }
 60         },
 61         destructor: function() {
 62             var i;
 63             for (i = 0; i < this.handlers.length; i += 1) {
 64                 this.handlers[i].detach();
 65             }
 66         },
 67         _show: function(event) {
 68             var instance;
 69             event = stringToObject(event);
 70             event = Y.mix(this.DEFAULT_CONFIG(), event, true, null, 0, false);
 71             instance = new Y.Wegas.Panel(event).render(this.get("host").get(this.get("targetAttr")));
 72             if (event.timeout) {
 73                 setTimeout(function() {
 74                     instance && instance.exit();
 75                 }, event.timeout);
 76                 // Y.later(event.timeout, instance, instance.hide);
 77             }
 78             // this.instance.setAttrs(event);
 79             // this.instance.show();
 80         },
 81         _system: function(event, lvl) {
 82             event = stringToObject(event);
 83             this._show({
 84                 content: "<div class='icon icon-" + lvl + "'>" + ((event && event.content) ? event.content : "") + "</div>",
 85                 timeout: event.timeout ? event.timeout : false
 86             });
 87         },
 88         onShowMessage: function(e) {
 89             if (e.level) {
 90                 e.content = "<div class='icon icon-" + e.level + "'>" + (e.content) + "</div>";
 91             }
 92             this._show(e);
 93             e.halt(true);
 94         },
 95         onShowOverlay: function(e) {
 96             this.get("host").get(this.get("targetAttr"))
 97                     .addClass("wegas-loading")
 98                     .prepend("<div class='wegas-loading-overlay'></div>");
 99             e.halt(true);
100         },
101         onHideOverlay: function(e) {
102             this.get("host").get(this.get("targetAttr"))
103                     .removeClass("wegas-loading")
104                     .all("> .wegas-loading-overlay").remove(true);
105             e.halt(true);
106         }
107     }, {
108         NS: "popuplistener",
109         ATTRS: {
110             targetAttr: {
111                 value: "boundingBox"
112             },
113             alignAttr: {
114                 value: "contentBox"
115             },
116             showServerMessages: {
117                 value: false
118             }
119         }
120     });
121 
122     Y.namespace("Plugin").PopupListener = PopupListener;
123 });
