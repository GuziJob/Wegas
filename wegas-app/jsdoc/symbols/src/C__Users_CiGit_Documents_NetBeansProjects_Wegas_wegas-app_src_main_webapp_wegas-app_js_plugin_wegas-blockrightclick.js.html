 12 YUI.add('wegas-blockrightclick', function(Y) {
 13     "use strict";
 14 
 15     /**
 16      *  @class Block right click
 17      *  @name Y.Plugin.BlockRightclick
 18      *  @extends Y.Plugin.Base
 19      *  @constructor
 20      */
 21     var Wegas = Y.Wegas,
 22             BlockRightclick = Y.Base.create("wegas-blockrightclick", Y.Plugin.Base, [Wegas.Plugin, Wegas.Editable], {
 23         /** @lends Y.Plugin.BlockRightclick */
 24 
 25         /**
 26          * @function
 27          * @private
 28          */
 29         initializer: function() {
 30             this.get("host").get(this.get("targetNode")).on('contextmenu', function(e) {
 31                 e.preventDefault();
 32             });
 33         }
 34     }, {
 35         ATTRS: {
 36             targetNode: {
 37                 value: "boundingBox",
 38                 type: "string"
 39             }
 40         },
 41         NS: "BlockRightclick",
 42         NAME: "BlockRightclick"
 43     });
 44     Y.namespace("Plugin").BlockRightclick = BlockRightclick;
 45 
 46 });
