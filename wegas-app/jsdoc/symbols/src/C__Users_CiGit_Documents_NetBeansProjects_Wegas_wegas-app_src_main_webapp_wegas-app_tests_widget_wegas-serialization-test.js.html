183         assertUseAndRevive: function(cfg) {
184             Y.Wegas.Widget.use(cfg, Y.bind(function(cfg) {                      // Load the subwidget dependencies
185                 var widget = Y.Wegas.Widget.create(Y.clone(cfg)),
186                         // Render the subwidget
187                         serializedCfg = widget.toObject(), // Serialize
188                         escape = function(t) {
189                     return Y.JSON.stringify(this.sortObject(t), null, "\t")
190                 },
191                         a = escape.call(this, cfg),
192                         b = escape.call(this, serializedCfg);
193 
194                 this.logResult(a, b);
195                 Y.Assert.areSame(a, b, "Seralized version does not match original version");
196 
197                 //widget.destroy();                                            // Delete
198                 //Y.Assert.isTrue(widget.get("destroyed"));

204                         this.nextPage();
205                         //Y.Assert.areEqual(document.getElementById("testDiv").offsetWidth, 400, "Width of the DIV should be 400.");

207                 });

243         sortObject: function(obj) {
244 
245             if (Y.Lang.isArray(obj)) {
246                 var sorted_obj = [];
247                 Y.Array.each(obj, function(i) {
248                     sorted_obj.push(this.sortObject(i));
249                 }, this);
250                 return sorted_obj;
251             } else if (Y.Lang.isObject(obj)) {
252 
253                 var keys = Y.Object.keys(obj),
254                         sorted_obj = {};
255 
256                 keys.sort();                                                        // sort keys
257 
258                 Y.Array.each(keys, function(item) {                               // create new array based on Sorted Keys
259                     sorted_obj[item] = this.sortObject(obj[item]);
260                 }, this);
261                 return sorted_obj;
262             } else {
263                 return obj;
264             }
