 13 YUI.add("wegas-image", function(Y) {
 14     "use strict";
 15 
 16     var CONTENTBOX = "contentBox", WImage;
 17     /**
 18      * @name Y.Wegas.WWImage
 19      * @extends Y.Widget
 20      * @borrows Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable
 21      * @class class to display simple image
 22      * @constructor
 23      * @description  Display a string (given as ATTRS) in content box
 24      */
 25     WImage = Y.Base.create("wegas-image", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 26         /** @lends Y.Wegas.WImage# */
 27         image: null,
 28         CONTENT_TEMPLATE: "<img></img>",
 29         /**
 30          * Lifecycle method
 31          * @function
 32          * @private
 33          */
 34         initializer: function() {
 35             this.publish("error", {fireOnce: true, async: true});
 36             this.publish("load", {fireOnce: true, async: true});
 37             this.handlers = [];
 38         },
 39         /**
 40          * Lifecycle method
 41          * @function
 42          * @private
 43          */
 44         renderUI: function() {
 45             this.image = this.get(CONTENTBOX).getDOMNode();
 46             this.set("url", this.get("url"));
 47         },
 48         /**
 49          * Lifecycle method
 50          * @function
 51          * @private
 52          */
 53         bindUI: function() {
 54             this.handlers.push(this.get(CONTENTBOX).on("load", function(e) {
 55                 this.getEvent("load").fired = true;
 56                 this.getEvent("render").fired = true;
 57                 this.fire("load");
 58                 this.fire("render");
 59             }, this));
 60             this.handlers.push(this.get(CONTENTBOX).on("error", function(e) {
 61                 this.getEvent("error").fired = true;
 62                 this.getEvent("render").fired = true;
 63                 this.fire("error");
 64                 this.fire("render");
 65             }, this));
 66         },
 67         /**
 68          * Lifecycle method
 69          * @function
 70          * @private
 71          */
 72         destructor: function() {
 73             var i;
 74             for (i in this.handlers) {
 75                 this.handlers[i].detach();
 76             }
 77         }
 78 
 79     }, {
 80         /** @lends Y.Wegas.WImage */
 81         EDITORNAME: "Image",
 82         FILEENTRY: Y.Wegas.FileFacade.get("source") + "read",
 83         ATTRS: {
 84             url: {
 85                 type: "string",
 86                 setter: function(val) {
 87                     this.getEvent("load").fired = false;
 88                     this.getEvent("error").fired = false;
 89                     this.getEvent("render").fired = false;
 90                     this.get(CONTENTBOX).setAttribute("src", (val.indexOf("/") === 0) ?
 91                             this.constructor.FILEENTRY + val : //Wegas Filesystem
 92                             val);
 93                     return val;
 94                 },
 95                 _inputex: {
 96                     _type: "wegasurl"
 97                 }
 98             },
 99             complete: {
100                 readOnly: true,
101                 "transient": true,
102                 getter: function() {
103                     return this.image.complete;
104                 }
105             }
106         }
107     });
108 
109     Y.namespace("Wegas").Image = WImage;
110 });
