 12 YUI.add('wegas-pmg-occupationcolor', function(Y) {
 13     "use strict";
 14 
 15     /**
 16      *  @class color occupation in datatable
 17      *  @name Y.Plugin.OccupationColor
 18      *  @extends Y.Plugin.Base
 19      *  @constructor
 20      */
 21     var Wegas = Y.Wegas,
 22             OccupationColor = Y.Base.create("wegas-pmg-occupationcolor", Y.Plugin.Base, [Wegas.Plugin, Wegas.Editable], {
 23         /** @lends Y.Plugin.OccupationColor */
 24 
 25         /**
 26          * Lifecycle methods
 27          * @function
 28          * @private
 29          */
 30         initializer: function() {
 31             Y.log("initializer", "info", "Wegas.OccupationColor");
 32             this.onceAfterHostEvent("render", function() {
 33                 this.sync();
 34                 this.afterHostMethod("syncUI", this.sync);
 35             });
 36             this.get("host").datatable.after("sort", this.sync, this);
 37         },
 38         sync: function() {
 39             Y.log("sync()", "info", "Wegas.OccupationColor");
 40             var i, ii, time,
 41                     host = this.get("host"),
 42                     dt = host.datatable,
 43                     abstractAssignement;
 44 
 45             this.addEngagementDelay();
 46 
 47             for (i = 0; i < dt.data.size(); i++) {
 48                 abstractAssignement = dt.data.item(i).get("descriptor").getInstance().get("occupations");
 49                 for (ii = 0; ii < abstractAssignement.length; ii++) {
 50                     time = abstractAssignement[ii].get("time");
 51                     if (time >= host.schedule.currentPeriod()
 52                             || !abstractAssignement[ii].get("editable")) {      //Affiche les occupations
 53                         this.addColor(host.schedule.getCell(i, time), abstractAssignement[ii].get("editable"));
 54                     }
 55                 }
 56             }
 57         },
 58         addColor: function(cell, editable) {
 59             if (editable) {
 60                 cell.setContent("<span class='editable'></span>");
 61             } else {
 62                 cell.setContent("<span class='notEditable'></span>");
 63             }
 64         },
 65         addEngagementDelay: function() {
 66             var i, ii, cell, host = this.get("host"),
 67                     dt = host.datatable,
 68                     currentPeriod = host.schedule.currentPeriod();
 69             for (i = 0; i < dt.data.size(); i++) {
 70                 for (ii = 0; ii < dt.data.item(i).get("properties.engagementDelay"); ii++) {
 71                     cell = host.schedule.getCell(i, currentPeriod + ii);
 72                     cell.setContent("<span class='engagementDelay'></span>");
 73                     cell.getDOMNode().className = "yui3-datatable-col-2 schedulecolumn delay yui3-datatable-cell";
 74                 }
 75             }
 76         }
 77     }, {
 78         NS: "occupationcolor",
 79         NAME: "OccupationColor"
 80     });
 81     Y.namespace("Plugin").OccupationColor = OccupationColor;
 82 });
