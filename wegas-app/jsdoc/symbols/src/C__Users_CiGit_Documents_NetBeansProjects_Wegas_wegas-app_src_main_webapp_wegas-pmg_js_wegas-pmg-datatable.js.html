 55         },

 57             Y.log("destructor()", "log", "Wegas.Datatable");
 58             this.updateHandler.detach();
 59             this.datatable.destroy();
 60         },
 61         //*** Private Methods ***/
 62         getData: function() {
 63             var oneRowDatas,
 64                     variables = this.get('variable.evaluated'),
 65                     data = [];
 66             if (!variables) {
 67                 this.showMessage("error", "Could not find variable");
 68                 return [];
 69             } else if (!variables instanceof Y.Wegas.persistence.ListDescriptor) {
 70                 this.showMessage("error", "Variable is not a ListDescriptor");
 71                 return [];
 72             }
 73             Y.Array.each(variables.get("items"), function(item) {
 74                 if (item.getInstance().get("active")) {
 75                     oneRowDatas = item.toJSON();
 76                     oneRowDatas.descriptor = item;
 77                     oneRowDatas.instance = item.getInstance().toJSON();
 78                     data.push(oneRowDatas);
 79                 }
 80             });
 81             return data;
 82         }

122                         aa = aa.toLowerCase();
123                         bb = bb.toLowerCase();
124                     }
125                     return (dir) ? aa < bb : aa > bb;
126                 };

223     Y.namespace("Wegas").PMGBodyView = Y.Base.create("pmg-bodyview", Y.DataTable.BodyView, [], {
224         _createRowHTML: function(model, index, columns) {
225             var data = model.toJSON(),
226                     clientId = model.get('clientId'),
227                     values = {
228                 rowId: this._getRowId(clientId),
229                 clientId: clientId,
230                 rowClass: (index % 2) ? this.CLASS_ODD : this.CLASS_EVEN
231             },
232             host = this.host || this,
233                     i, len, col, token, value, formatterData;
234             for (i = 0, len = columns.length; i < len; ++i) {
235                 col = columns[i];
236                 value = (col.key) ? model.get(col.key) : null; // @modified
237                 //value = data[col.key];
238                 token = col._id || col.key;
239                 values[token + '-className'] = '';
240                 if (col._formatterFn) {
241                     formatterData = {
242                         value: value,
243                         data: data,
244                         column: col,
245                         record: model,
246                         className: '',
247                         rowClass: '',
248                         rowIndex: index
249                     };
250                     value = col._formatterFn.call(host, formatterData); // Formatters can either return a value
251                     if (value === undefined) {// or update the value property of the data obj passed
252                         value = formatterData.value;
253                     }
254                     values[token + '-className'] = formatterData.className;
255                     values.rowClass += ' ' + formatterData.rowClass;
256                 }
257                 if (!values.hasOwnProperty(token) || data.hasOwnProperty(col.key)) {// if the token missing OR is the value a legit value
258                     if (value === undefined || value === null || value === '') {
259                         value = col.emptyCellValue || '';
260                     }
261                     values[token] = col.allowHTML ? value : Y.Escape.html(value);
262                 }
263             }
264             values.rowClass = values.rowClass.replace(/\s+/g, ' '); // replace consecutive whitespace with a single space
265             return Y.Lang.sub(this._rowTemplate, values); // @modified
266         },
267         refreshCell: function(cell, model, col) {
268             var content,
269                     formatterFn,
270                     formatterData,
271                     data = model.toJSON();
272             cell = this.getCell(cell);
273             model || (model = this.getRecord(cell));
274             col || (col = this.getColumn(cell));
275             if (col.nodeFormatter) {
276                 formatterData = {
277                     cell: cell.one('.' + this.getClassName('liner')) || cell,
278                     column: col,
279                     data: data,
280                     record: model,
281                     rowIndex: this._getRowIndex(cell.ancestor('tr')),
282                     td: cell,
283                     value: model.get(col.key)                                       // @Modified
284                             //value: data[col.key]
285                 };
286                 keep = col.nodeFormatter.call("host", formatterData);
287                 if (keep === false) {
288                     cell.destroy(true);
289                 }
290             } else if (col.formatter) {
291                 if (!col._formatterFn) {
292                     col = this._setColumnsFormatterFn([col])[0];
293                 }
294                 formatterFn = col._formatterFn || null;
295                 if (formatterFn) {
296                     formatterData = {
297                         value: model.get(col.key), // @Modified
298                         //value: data[col.key],
299                         data: data,
300                         column: col,
301                         record: model,
302                         className: '',
303                         rowClass: '',
304                         rowIndex: this._getRowIndex(cell.ancestor('tr'))
305                     };
306                     content = formatterFn.call(this.get('host'), formatterData); // Formatters can either return a value ...
307                     if (content === undefined) {// ... or update the value property of the data obj passed
308                         content = formatterData.value;
309                     }
310                 }
311                 if (content === undefined || content === null || content === '') {
312                     content = col.emptyCellValue || '';
313                 }
314             } else {
315                 content = model.get(col.key) || col.emptyCellValue || '';
316             }
317 
318             cell.setHTML(col.allowHTML ? content : Y.Escape.html(content));
319             return this;
320         }
321     });
