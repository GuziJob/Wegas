 14 YUI.add("wegas-inputex-var-autocomplete", function(Y) {
 15     "use strict";
 16 
 17     var inputEx = Y.inputEx;
 18 
 19 
 20     Y.namespace("inputEx.Wegas").VarAutoComplete = function(options) {
 21         inputEx.Wegas.VarAutoComplete.superclass.constructor.call(this, options);
 22     };
 23 
 24     Y.extend(inputEx.Wegas.VarAutoComplete, inputEx.StringField, {
 25         /**
 26          *
 27          */
 28         variables: null,
 29         /**
 30          *
 31          * @param {type} options
 32          * @returns {undefined}
 33          */
 34         setOptions: function(options) {
 35             inputEx.StringField.superclass.setOptions.call(this, options);
 36             if (options.variableClass) {
 37                 this.findVariable(options.variableClass);
 38             }
 39         },
 40         onFocus: function(d) {
 41             inputEx.StringField.superclass.onFocus.call(this, d);
 42         },
 43         findVariable: function(varName) {
 44             //use flatten function (in Wegas.cache)
 45             this.variables = Y.Wegas.Facade.VariableDescriptor.cache.doFind("GET", "TaskDescriptor");
 46             // console.log(variables);
 47         }
 48     });
 49 
 50     inputEx.registerType("wegasvarautocomplete", inputEx.Wegas.VarAutoComplete);// Register this class as "wegasvarautocomplete" type
 51 });
