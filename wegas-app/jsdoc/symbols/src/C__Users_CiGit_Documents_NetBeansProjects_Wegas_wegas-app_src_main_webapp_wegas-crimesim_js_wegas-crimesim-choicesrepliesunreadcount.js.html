 13 YUI.add('wegas-crimesim-choicesrepliesunreadcount', function(Y) {
 14     "use strict";
 15 
 16     /**
 17      * Plugin which count replies from choices and adds an unread message
 18      * counter to a widget.
 19      *
 20      * @class Y.Wegas.ChoicesRepliesUnreadCount
 21      * @extends Y.Plugin.UnreadCount
 22      * @borrows Y.Wegas.Editable
 23      */
 24     var ChoicesRepliesUnreadCount = Y.Base.create("wegas-crimesim-choicesRepliesUnreadCount", Y.Plugin.UnreadCount, [Y.Wegas.Plugin], {
 25         getUnreadCount: function() {
 26             var i, j, count = 0, questionInstance, reply,
 27                     questions = Y.Wegas.Facade.VariableDescriptor.cache.find('name', "evidences");
 28 
 29             if (!questions) {
 30                 return;
 31             }
 32             questions = questions.flatten();
 33 
 34             for (i = 0; i < questions.length; i = i + 1) {
 35                 questionInstance = questions[i].getInstance();
 36                 if (questionInstance instanceof Y.Wegas.persistence.QuestionInstance) {
 37                     for (j = 0; j < questionInstance.get("replies").length; j = j + 1) {
 38                         reply = questionInstance.get("replies")[j];
 39                         if (reply.getAttrs() && reply.getAttrs().unread) {
 40                             count++;
 41                         }
 42                     }
 43                 }
 44             }
 45             return count;
 46         }
 47     }, {
 48         NS: "ChoicesRepliesUnreadCount",
 49         NAME: "ChoicesRepliesUnreadCount"
 50     });
 51     Y.namespace('Plugin').ChoicesRepliesUnreadCount = ChoicesRepliesUnreadCount;
 52 
 53 });
