 14 YUI.add('wegas-loginwidget', function (Y) {
 15     "use strict";
 16 
 17     var CONTENTBOX = 'contentBox',
 18             LoginWidget;
 19 
 20     /**
 21      * @name Y.Wegas.LoginWidget
 22      * @extends Y.Widget
 23      * @borrows Y.WidgetChild, Y.Wegas.Widget
 24      * @class class to manage the login of a user.
 25      * @constructor
 26      * @description Widget to display some forms relative to 'login' like
 27      *  login itself, forgot password and create a new user.  
 28      */
 29     LoginWidget = Y.Base.create("wegas-loginwidget", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
 30         /**
 31          * @lends Y.Wegas.LoginWidget#
 32          */
 33         // *** Private fields *** //
 34         /**
 35          * Default link to redirect user
 36          */
 37         defaultRedirect: "wegas-app/view/lobby.html",
 38         /**
 39          * form group to login
 40          */
 41         loginForm: null,
 42         /**
 43          * form group to create account
 44          */
 45         createAccountForm: null,
 46         /**
 47          * form group to send a new password
 48          */
 49         sendNewPasswordForm: null,
 50         /**
 51          * Button to submit form
 52          */
 53         submitButton: null,
 54         /**
 55          * Reference to each used functions
 56          */
 57         handlers: null,
 58         // *** Lifecycle Methods *** //
 59         /**
 60          * @function
 61          * @private
 62          * @description Set variable with initials values.
 63          * init the submit button.
 64          * init all the three forms groups
 65          */
 66         initializer: function () {
 67             var cb = this.get(CONTENTBOX);
 68             this.handlers = {};
 69             this.submitButton = new Y.Button();
 70             this.loginForm = new Y.inputEx.Group({
 71                 fields: [{
 72                         name: "email",
 73                         label: "Email",
 74                         required: true,
 75                         type: "email"
 76                     }, {
 77                         name: "password",
 78                         label: "Password",
 79                         required: true,
 80                         type: "password",
 81                         capsLockWarning: true
 82                     }, {
 83                         label: "",
 84                         type: "boolean",
 85                         name: "remember",
 86                         rightLabel: " Remember me"
 87                     }],
 88                 parentEl: cb
 89             });
 90 
 91             this.createAccountForm = new Y.inputEx.Group({
 92                 parentEl: cb,
 93                 fields: [{
 94                         name: "id",
 95                         type: "hidden"
 96                     }, {
 97                         name: "@class",
 98                         required: true,
 99                         type: "hidden",
100                         value: "JpaAccount"
101                     }, {
102                         label: "First name",
103                         name: "firstname",
104                         required: true,
105                         type: "string",
106                         showMsg: true
107                     }, {
108                         label: "Last name",
109                         name: "lastname",
110                         required: true,
111                         type: "string",
112                         showMsg: true
113                     }, {
114                         label: "Email",
115                         name: "email",
116                         required: true,
117                         type: "email",
118                         showMsg: true
119                     }, {
120                         label: "Password",
121                         name: "password",
122                         strengthIndicator: true,
123                         capsLockWarning: true,
124                         id: "password",
125                         required: true,
126                         type: "password",
127                         showMsg: true
128                     }, {
129                         label: "Password (confirm)",
130                         name: "passwordConfirm",
131                         showMsg: true,
132                         required: true,
133                         confirm: "password",
134                         type: "password"
135                     }]
136             });
137 
138             this.sendNewPasswordForm = new Y.inputEx.Group({
139                 fields: [{
140                         name: "email",
141                         label: "Email",
142                         required: true,
143                         type: "email"
144                     }],
145                 parentEl: cb
146             });
147         },
148         /**
149          * @function
150          * @private
151          * @description render du submit button.
152          * display link to display other form than default one.
153          * Call 'redirect' function if user is alread logged.
154          */
155         renderUI: function () {
156             var cb = this.get(CONTENTBOX),
157                     cUser = Y.Wegas.app.get("currentUser");
158 
159             if (cUser.accounts[0]["@class"] !== "GuestAccount") {
160                 this.showMessage("success", "You are already logged in.", 4000);
161                 this.redirect();
162             }
163 
164             this.submitButton.render(cb);
165             cb.append('<div class="links" style="margin-left: 136px;margin-top:10px;">'
166                     + '<p><a class="alt-link" href="#"></a></p>'
167                     + '<p><a class="send-new-password" href="#">Forgot your password?</a></p>'
168                     + '</div>');
169         },
170         /**
171          * @function
172          * @private
173          * @description bind function to events.
174          * When node alt-link is clicked, do toggleCreateAccount.
175          * When node send-new-password is clicked, do toggleSendNewPassword.
176          * When submit button is clicked, do onSubmit;
177          * When return key is pressed, click on submitButton by a fire event.
178          * When widget is render, set focus to 'input' node.
179          */
180         bindUI: function () {
181             var cb = this.get(CONTENTBOX),
182                     inputNode = cb.one("input");
183 
184             this.handlers.toggleCreateAccount = cb.one(".alt-link").on("click", this.toggleCreateAccount, this);
185             this.handlers.toggleSendNewPassword = cb.one(".send-new-password").on("click", this.toggleSendNewPassword, this);
186 
187             this.handlers.onSubmit = this.submitButton.on("click", this.onSubmit, this);
188             this.handlers.keypress = this.on("keypress", function (e) {
189                 if (e.domEvent.keyCode === 13) {
190                     this.submitButton.fire("click");
191                 }
192             });
193             this.handlers.render = this.after("render", inputNode.focus, inputNode);
194         },
195         /**
196          * @function
197          * @private
198          * @description set the displayed form with the current form
199          */
200         syncUI: function () {
201             this.set("mode", this.get("mode"));
202         },
203         /**
204          * @function
205          * @private
206          * @description Detach all functions created by this widget.
207          */
208         destructor: function () {
209             for (var i = 0; i < this.handlers.length; i += 1) {
210                 this.handlers[i].detach();
211             }
212         },
213         // *** Private methods *** //
214         /**
215          * @function
216          * @private
217          * @description toggle mode to createAccount or login one.
218          */
219         toggleCreateAccount: function () {
220             if (this.get("mode") === "login") {
221                 this.set("mode", "createaccount");
222             } else {
223                 this.set("mode", "login");
224             }
225         },
226         /**
227          * @function
228          * @private
229          * @description toggle mode to sendNewPassword or login one.
230          */
231         toggleSendNewPassword: function () {
232             if (this.get("mode") === "login") {
233                 this.set("mode", "sendNewPassword");
234             } else {
235                 this.set("mode", "login");
236             }
237         },
238         /**
239          * @function
240          * @private
241          * @param e
242          * @description On submit, call function depending on the current mode :
243          * Mode login, call function login.
244          * Mode createaccount, call function createaccount.
245          * Mode sendNewPassword, call function sendNewPassword.
246          */
247         onSubmit: function (e) {
248             var data;
249             switch (this.get("mode")) {
250                 case "login":
251                     if (this.loginForm.validate()) {
252                         data = this.loginForm.getValue();
253                         this.login(data.email, data.password, data.remember);
254                     }
255                     break;
256 
257                 case "createaccount":
258                     if (this.createAccountForm.validate()) {
259                         this.createAccount(this.createAccountForm.getValue());
260                     }
261                     break;
262 
263                 case "sendNewPassword":
264                     if (this.sendNewPasswordForm.validate()) {
265                         data = this.sendNewPasswordForm.getValue();
266                         this.showOverlay();
267                         this.sendNewPassword(data.email);
268                     }
269                     break;
270             }
271         },
272         /**
273          * @function
274          * @private
275          * @param email
276          * @param password
277          * @param remember
278          * @description Send REST request to login a user with form's values
279          *  informations.
280          */
281         login: function (email, password, remember) {
282             Y.Wegas.UserFacade.rest.sendRequest({
283                 request: "/Authenticate/?email=" + email
284                         + "&password=" + password
285                         + "&remember=" + remember,
286                 cfg: {
287                     method: "POST"
288                 },
289                 on: {
290                     success: Y.bind(function (e) {
291                         this.showMessage("success", "Login successful", 4000);
292                         this.redirect();
293                         return;
294                     }, this),
295                     failure: Y.bind(function (e) {
296                         this.showMessage("error", e.response.results.message || "Email/password combination not found", 4000);
297                     }, this)
298                 }
299             });
300         },
301         /**
302          * @function
303          * @private
304          * @param data
305          * @description Send REST request to create a new account based on the
306          *  form's values
307          */
308         createAccount: function (data) {
309             Y.Wegas.UserFacade.rest.sendRequest({
310                 request: "/Signup/",
311                 cfg: {
312                     method: "POST",
313                     data: data
314                 },
315                 on: {
316                     success: Y.bind(function (data, e) {
317                         this.showMessage("success", "User created, you can now use it to login", 4000);
318                         //this.set("mode", "login");
319                         this.login(data.email, data.password, false);
320                     }, this, data),
321                     failure: Y.bind(function (e) {
322                         this.showMessage("error", e.response.results.message || "Error creating user", 4000);
323                     }, this)
324                 }
325             });
326         },
327         /**
328          * @function
329          * @private
330          * @param email
331          * @description Send REST request to set and set a new password for
332          *  the user.
333          */
334         sendNewPassword: function (email) {
335             Y.Wegas.UserFacade.rest.sendRequest({
336                 request: "/SendNewPassword/?email=" + email,
337                 cfg: {
338                     method: "POST"
339                 },
340                 on: {
341                     success: Y.bind(function (e) {
342                         this.hideOverlay();
343                         this.showMessage("success", "Your new password had been sent", 4000);
344                         this.set("mode", "login");
345                     }, this),
346                     failure: Y.bind(function (e) {
347                         this.hideOverlay();
348                         this.showMessage("error", e.response.results.message || "Error sent new password", 4000);
349                     }, this)
350                 }
351             });
352         },
353         /**
354          * @function
355          * @private
356          * @description redirect the user to a page given by the function 'getRedirect'
357          */
358         redirect: function () {
359             window.location = this.getRedirect();
360         },
361         /**
362          * @function
363          * @private
364          * @return url to redirect 
365          * @description return redirection given by the function
366          *  'getQueryParameter' or by the default redirection.
367          */
368         getRedirect: function () {
369             return this.getQueryParameter("redirect") || (Y.Wegas.app.get("base") + this.defaultRedirect);
370         },
371         /**
372          * @function
373          * @private
374          * @param name
375          * @return null or url
376          * @description Returns a parameter from the GET parameters (from
377          *  url obviously).
378          */
379         getQueryParameter: function (name) {
380             var i, pair, query = window.location.search.substring(1),
381                     vars = query.split("&");
382 
383             for (i = 0; i < vars.length; i = i + 1) {
384                 pair = vars[i].split("=");
385                 if (pair[0] === name) {
386                     return decodeURIComponent(pair[1]);
387                 }
388             }
389             return null;
390         }
391 
392     }, {
393         /**
394          * @lends Y.Wegas.LoginWidget#
395          */
396         /**
397          * @field
398          * @static
399          * @description
400          * <p><strong>Method</strong></p>
401          * <ul>
402          *    <li>Mode: login, createaccount or sendNewPassword, setter change
403          *     the mode and thus, the displayed form</li>
404          * </ul>
405          */
406         ATTRS: {
407             mode: {
408                 value: "login",
409                 setter: function (val) {
410                     var cb = this.get(CONTENTBOX);
411 
412                     if (val === "login") {
413                         //this.loginForm.clear();
414                         this.loginForm.show();
415                         this.createAccountForm.hide();
416                         this.sendNewPasswordForm.hide();
417                         this.submitButton.set("label", "Login");
418                         cb.one(".alt-link").setContent("Create a new user");
419                         cb.one(".send-new-password").show();
420                     } else if (val === 'createaccount') {
421                         this.loginForm.hide();
422                         this.createAccountForm.show();
423                         this.sendNewPasswordForm.hide();
424                         //this.createAccountForm.clear();
425                         this.submitButton.set("label", "Submit");
426                         cb.one(".alt-link").setContent("Login with existing account");
427                         cb.one(".send-new-password").hide();
428                     } else {
429                         this.loginForm.hide();
430                         this.createAccountForm.hide();
431                         this.sendNewPasswordForm.show();
432                         this.submitButton.set("label", "Submit");
433                         cb.one(".alt-link").setContent("Return to the login page");
434                         cb.one(".send-new-password").hide();
435                     }
436                 }
437             }
438         }
439     });
440 
441     Y.namespace('Wegas').LoginWidget = LoginWidget;
442 });
