 12 YUI.add('wegas-pmg-activitycolor', function(Y) {
 13     "use strict";
 14 
 15     /**
 16      *  @class color occupation in datatable
 17      *  @name Y.Plugin.ActivityColor
 18      *  @extends Y.Plugin.OccupationColor
 19      *  @constructor
 20      */
 21     var Wegas = Y.Wegas,
 22             ActivityColor = Y.Base.create("wegas-pmg-activitycolor", Y.Plugin.OccupationColor, [Wegas.Plugin, Wegas.Editable], {
 23         /** @lends Y.Plugin.ActivityColor */
 24 
 25         sync: function() {
 26             var i, ii, time, abstractAssignement,
 27                     host = this.get("host"),
 28                     currentPeriod = host.schedule.currentPeriod(),
 29                     dt = host.datatable;
 30 
 31             for (i = 0; i < dt.data.size(); i++) {
 32                 abstractAssignement = dt.data.item(i).get("descriptor").getInstance().get("activities");
 33                 for (ii = 0; ii < abstractAssignement.length; ii++) {
 34                     time = parseInt(abstractAssignement[ii].get("time"));
 35                     if (time < currentPeriod) {
 36                         this.addColor(host.schedule.getCell(i, time),
 37                                 abstractAssignement[ii].get("editable") === null ? "true" : abstractAssignement[ii].get("editable"));
 38 
 39                     }
 40                 }
 41             }
 42         }
 43     }, {
 44         NS: "activitycolor",
 45         NAME: "ActivityColor"
 46     });
 47     Y.namespace("Plugin").ActivityColor = ActivityColor;
 48 });
