 15 YUI.add('wegas-shareuser', function(Y) {
 16     var CONTENTBOX = 'contentBox',
 17             ShareUser = Y.Base.create("wegas-shareuser", Y.Widget, [Y.WidgetChild, Y.Wegas.Editable, Y.Wegas.Widget], {
 18         /**
 19          *
 20          */
 21         CONTENT_TEMPLATE: "<div><div class=\"title\">Permissions</div><div class=\"wegas-userlist\"></div>"
 22                 + "<div class=\"wegas-adduser\"><div class=\"title\">Add user</div></div></div>",
 23         /**
 24          *
 25          */
 26         renderUI: function() {
 27             var el = this.get(CONTENTBOX),
 28                     e = this.get("entity"),
 29                     permissions = [{
 30                     name: "username",
 31                     type: 'markup',
 32                     readonly: true,
 33                     className: "username-field"
 34                 }, {
 35                     name: "userId",
 36                     type: "hidden"
 37                 }];
 38 
 39             permissions = permissions.concat(Y.Array.map(this.get("permsList"), function(item) {
 40                 item.type = "boolean";
 41                 return item;
 42             }));
 43 
 44             this.userList = new UserPermissionList({
 45                 elementType: {
 46                     type: 'permissiongroup',
 47                     fields: permissions,
 48                     className: "permission-group"
 49                 },
 50                 parentEl: el.one(".wegas-userlist"),
 51                 useButtons: true
 52             });
 53 
 54             if (e instanceof Y.Wegas.persistence.GameModel) {
 55                 this.userList.targetEntityId = "gm" + e.get("id");
 56             } else {
 57                 this.userList.targetEntityId = "g" + e.get("id");
 58             }
 59 
 60 
 61             this.autocompleteValue = [];
 62             this.field = new Y.inputEx.AutoComplete({
 63                 parentEl: el.one(".wegas-adduser"),
 64                 typeInvite: "e-mail, name or lastname",
 65                 // Format the hidden value (value returned by the form)
 66                 returnValue: Y.bind(function(oResultItem) {
 67                     if (!this.field.options.value) {
 68                         this.field.options.value = [];
 69                         // console.log("in instanciate");
 70                     }
 71                     this.field.options.value.push(oResultItem);
 72                     return oResultItem.value;
 73                 }, this),
 74                 autoComp: {
 75                     minQueryLength: 1,
 76                     maxResults: 30,
 77                     resultTextLocator: 'label',
 78                     resultHighlighter: 'phraseMatch',
 79                     queryDelimiter: ',',
 80                     source: Y.Wegas.app.get("base") + "rest/User/AutoComplete/{query}",
 81                     enableCache: false,
 82                     resultListLocator: Y.bind(function(responses) {
 83                         var i;
 84                         Y.Array.forEach(this.userList.subFields, function(user) {
 85                             for (i = 0; i < responses.length; i++) {
 86                                 if (user.getValue().userId === responses[i].value) {
 87                                     responses.splice(responses[i], 1);
 88                                     break;
 89                                 }
 90                             }
 91                         });
 92                         return responses;
 93                     }, this)
 94                 }
 95             });
 96 
 97             this.saveButton = new Y.Wegas.Button({
 98                 label: "Add",
 99                 render: el.one(".wegas-adduser")
100             });
101 
102             this.loadPermissions();
103         },
104         bindUI: function() {
105             this.saveButton.on("click", function() {
106                 var values = [],
107                         fieldValue = this.field.yEl.get("value"),
108                         userNames = fieldValue.split(",");
109 
110                 if (fieldValue === "") {                                        // Check the input element is not empty
111                     return;
112                 }
113                 //if (this.field.getValue() === "")
114                 //    return;
115 
116                 Y.Array.each(userNames, function(value) {                       // Remove items that have
117                     if ((!this.field.options.value || !this.field.options.value.indexOf[value.trim()])
118                             && value.trim() !== "") {
119                         values.push(value);
120                     }
121                 }, this);
122                 if (values.length > 0) {
123                     Y.Wegas.Facade.User.sendRequest({
124                         request: "/FindAccountsByValues/",
125                         cfg: {
126                             data: {
127                                 values: values
128                             }
129                         },
130                         on: {
131                             success: Y.bind(function(e) {
132                                 Y.Array.each(e.response.results.entities, function(account) {
133                                     this.userList.addElement({
134                                         username: account.get("val.label"),
135                                         userId: account.get("val.value"),
136                                     });
137                                 }, this);
138                             }, this),
139                             failure: Y.bind(this.defaultFailureHandler, this)
140                         }
141                     });
142                 }
143 
144                 Y.Array.each(this.field.options.value, function(account) {
145                     this.userList.addElement({
146                         username: account.label,
147                         userId: account.value
148                     });
149                 }, this);
150 
151                 this.field.options.value = [];
152                 this.field.setValue("");
153             }, this);
154         },
155         destructor: function() {
156             this.field.destroy();
157             this.saveButton.destroy();
158             this.userList.destroy();
159         },
160         loadPermissions: function() {
161             Y.Wegas.Facade.User.sendRequest({
162                 request: "/FindAccountPermissionByInstance/" + this.userList.targetEntityId,
163                 on: {
164                     success: Y.bind(function(e) {
165                         var data = e.response.results.entities,
166                                 i, permissions, splitedPerm, newField;
167 
168                         Y.Array.forEach(data, function(account) {
169                             permissions = account.get('permissions');
170                             newField = this.userList.addElement({
171                                 username: account.getPublicName(),
172                                 userId: account.get('id')
173                             });
174 
175                             Y.Array.forEach(newField.inputs, function(field) {
176                                 for (i = 0; i < permissions.length; i++) {
177                                     splitedPerm = permissions[i].get("val").value.split(":");
178                                     if (splitedPerm[2] === this.userList.targetEntityId) {
179                                         if (field.options.value === splitedPerm[0] + ":" + splitedPerm[1]) {
180                                             field.setValue(true, false);
181                                         }
182                                     }
183                                 }
184                             }, this);
185 
186                         }, this);
187                     }, this),
188                     failure: Y.bind(this.defaultFailureHandler, this)
189                 }
190             });
191         }
192     }, {
193         ATTRS: {
194             permsList: {
195                 value: []
196             },
197             entity: {}
198         }
199     });
200     Y.namespace('Wegas').ShareUser = ShareUser;
201 
202     /**
203      * @name Y.inputEx.Wegas.UserPermissionList
204      * @class
205      * @extends Y.inputEx.ListField
206      * @constructor
207      * @param {Object} options
208      */
209     var UserPermissionList = function(options) {
210         UserPermissionList.superclass.constructor.call(this, options);
211     };
212     Y.extend(UserPermissionList, Y.inputEx.ListField, {
213         /** @lends Y.inputEx.Wegas.UserPermissionList# */
214 
215         onDelete: function(e) {
216             var elementDiv = e.target._node.parentNode,
217                     subFieldEl = elementDiv.childNodes[this.options.useButtons ? 1 : 0];
218             for (var i = 0; i < this.subFields.length; i++) {
219                 if (this.subFields[i].getEl() === subFieldEl) {
220                     this.deletePermission(this.targetEntityId, this.subFields[i].getValue().userId); //param : entity Id, have userId
221                     break;
222                 }
223             }
224 
225             UserPermissionList.superclass.onDelete.apply(this, arguments);
226         },
227         deletePermission: function(entityId, userId) {
228             Y.Wegas.Facade.User.sendRequest({
229                 request: "/DeleteAccountPermissionByInstanceAndAccount/" + entityId + "/" + userId,
230                 cfg: {
231                     method: "DELETE"
232                 },
233                 on: {
234                     failure: Y.bind(this.defaultFailureHandler, this)
235                 }
236             });
237         }
238     });
239 
240     /**
241      * @name Y.inputEx.Wegas.PermissionGroup
242      * @class
243      * @extends Y.inputEx.Group
244      * @constructor
245      * @param {Object} options
246      */
247     var PermissionGroup = function(options) {
248         PermissionGroup.superclass.constructor.call(this, options);
249     };
250     Y.extend(PermissionGroup, Y.inputEx.Group, {
251         /** @lends Y.inputEx.Wegas.PermissionGroup# */
252         onChange: function(fieldValue, fieldInstance) {
253             if (fieldValue) {
254                 this.addPermission(fieldInstance.options.value + ":" + this.parentField.targetEntityId, this.getValue().userId);
255             } else {
256                 this.removePermission(fieldInstance.options.value + ":" + this.parentField.targetEntityId, this.getValue().userId);
257             }
258             PermissionGroup.superclass.onChange.apply(this, arguments);
259         },
260         removePermission: function(permission, userId) {
261             Y.Wegas.Facade.User.sendRequest({
262                 request: "/DeleteAccountPermissionByPermissionAndAccount/" + permission + "/" + userId,
263                 cfg: {
264                     method: "DELETE"
265                 },
266                 on: {
267                     failure: Y.bind(function(e) {
268                         this.showMessage("error", "Error removing permission");
269                     }, this)
270                 }
271             });
272         },
273         addPermission: function(permission, userId) {
274             Y.Wegas.Facade.User.sendRequest({
275                 request: "/addAccountPermission/" + permission + "/" + userId,
276                 cfg: {
277                     method: "POST"
278                 },
279                 on: {
280                     failure: Y.bind(function(e) {
281                         this.showMessage("error", "Error adding permission");
282                     }, this)
283                 }
284             });
285         }
286     });
287     Y.inputEx.registerType("permissiongroup", PermissionGroup);
288 });
