 12 YUI.add('wegas-editor-widgetaction', function(Y) {
 13     "use strict";
 14 
 15     var Plugin = Y.Plugin, Action = Y.Plugin.Action, Wegas = Y.Wegas,
 16             WidgetAction;
 17 
 18     /**
 19      * @class
 20      * @name Y.Plugin.WidgetAction
 21      * @extends Y.Plugin.Action
 22      * @constructor
 23      */
 24     WidgetAction = function() {
 25         WidgetAction.superclass.constructor.apply(this, arguments);
 26     };
 27     Y.extend(WidgetAction, Action, {}, {
 28         /** @lends Y.Wegas.EntityAction */
 29         NS: "WidgetAction",
 30         NAME: "WidgetAction",
 31         ATTRS: {
 32             widget: {},
 33             dataSource: {
 34                 getter: function(val) {
 35                     if (!val) {
 36                         return Wegas.Facade.Page;
 37                     }
 38                     return val;
 39                 }
 40             }
 41         }
 42     });
 43     //Plugin.WidgetAction = WidgetAction;
 44 
 45     /**
 46      * @name Y.Plugin.EditWidgetAction
 47      * @extends Y.Plugin.WidgetAction
 48      * @constructor
 49      */
 50     var EditWidgetAction = function() {
 51         EditWidgetAction.superclass.constructor.apply(this, arguments);
 52     };
 53     Y.extend(EditWidgetAction, WidgetAction, {
 54         /**
 55          * @function
 56          * @private
 57          */
 58         execute: function() {
 59             Plugin.EditEntityAction.showEditForm(this.get("widget"), Y.bind(function(val, e, f) {
 60                 Plugin.EditEntityAction.hideEditFormOverlay();
 61                 var i, targetWidget = this.get("widget"), plugins = {}, plugin;
 62                 targetWidget.setAttrs(val);
 63                 for (i = 0; i < val.plugins.length; i += 1) {
 64                     plugin = Y.Plugin[Y.Wegas.Plugin.getPluginFromName(val.plugins[i].fn)];
 65                     if (!Y.Lang.isUndefined(targetWidget._plugins[plugin.NS])) { //that plugin exists on target
 66                         targetWidget[plugin.NS].setAttrs(val.plugins[i].cfg);
 67                         plugins[plugin.NS] = true;                              //store namespace as treated
 68                     } else {
 69                         targetWidget.plug(plugin, val.plugins[i].cfg);
 70                         plugins[plugin.NS] = true;                              //store namespace as treated
 71                     }
 72                 }
 73                 for (i in targetWidget._plugins) {                                 // remove
 74                     if (Y.Lang.isUndefined(plugins[i])) {                           //An inexistant namespace
 75                         targetWidget.unplug(i);
 76                     }
 77                 }
 78                 targetWidget.syncUI();
 79                 this.get("dataSource").cache.patch(targetWidget.get("root").toObject());
 80             }, this), Y.bind(function() {
 81                 this.get("widget").highlight(false);
 82             }, this));
 83 
 84             this.get("widget").highlight(true);
 85         }
 86     }, {
 87         NS: "EditWidgetAction",
 88         NAME: "EditWidgetAction"
 89     });
 90     Plugin.EditWidgetAction = EditWidgetAction;
 91 
 92     /**
 93      * @class
 94      * @name Y.Plugin.AddChildWidgetAction
 95      * @extends Y.Plugin.WidgetAction
 96      * @constructor
 97      */
 98     var AddChildWidgetAction = function() {
 99         AddChildWidgetAction.superclass.constructor.apply(this, arguments);
100     };
101     Y.extend(AddChildWidgetAction, WidgetAction, {
102         execute: function() {
103             Y.use(Y.Wegas.Editable.getRawModulesFromDefinition({type: this.get("childType")}), Y.bind(function() {
104                 var newWidget = new Y.Wegas.Widget.create({
105                     "type": this.get("childType")
106                 });
107 
108                 Wegas.Editable.use(newWidget, Y.bind(function() {                  // Load target widget dependencies
109 
110                     Plugin.EditEntityAction.showEditForm(newWidget, Y.bind(function(val) {
111                         Plugin.EditEntityAction.hideEditFormOverlay();
112                         var targetWidget = this.get("widget"), widget = new Y.Wegas.Widget.create(val);
113                         targetWidget.add(widget);
114                         this.get("dataSource").cache.patch(targetWidget.get("root").toObject(), {success: Y.bind(function() {
115                                 var tw = new Y.Wegas.Text();
116                                 tw.plug(Plugin.EditWidgetAction, {"widget": this});
117                                 tw.EditWidgetAction.execute();
118                             }, widget)
119                         });
120                     }, this));
121 
122                 }, this));
123             }, this));
124 
125         }
126     }, {
127         NS: "AddChildWidgetAction",
128         NAME: "AddChildWidgetAction",
129         ATTRS: {
130             childType: {}
131         }
132     });
133     Plugin.AddChildWidgetAction = AddChildWidgetAction;
134 
135     /**
136      * @class
137      * @name Y.Plugin.DeleteWidgetAction
138      * @extends Y.Plugin.WidgetAction
139      * @constructor
140      */
141     var DeleteWidgetAction = function() {
142         DeleteWidgetAction.superclass.constructor.apply(this, arguments);
143     };
144     Y.extend(DeleteWidgetAction, WidgetAction, {
145         execute: function() {
146             if (confirm("Are your sure your want to delete this widget ?")) {
147                 var targetWidget = this.get("widget"),
148                         root = targetWidget.get("root");
149 
150                 if (root !== targetWidget) {
151                     targetWidget.destroy();
152                 } else if (targetWidget.item(0)) { // @TODO: Panic mode, to change
153                     targetWidget.removeAll();
154                 }
155 
156                 this.get("dataSource").cache.patch(root.toObject());
157             }
158         }
159     }, {
160         NS: "DeleteWidgetAction",
161         NAME: "DeleteWidgetAction"
162     });
163     Plugin.DeleteWidgetAction = DeleteWidgetAction;
164 
165 });
