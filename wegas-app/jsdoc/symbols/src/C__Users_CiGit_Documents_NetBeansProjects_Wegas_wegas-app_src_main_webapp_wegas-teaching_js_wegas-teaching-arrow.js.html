 13     Y.extend(Y.TeachingArrow, Y.Path, {

 62

 64

 66 
 67             if (src && Y.Lang.isFunction(src.removeWire)) {
 68                 src.removeWire(this);

 70             if (tgt && Y.Lang.isFunction(tgt.removeWire)) {
 71                 tgt.removeWire(this);

 73         },
 74         /**
 75          * Drawing method. Meant to be overriden by a plugin (by WireIt)
 76          * @method _draw
 77          * @private
 78          */
 79         _draw: function() {
 80             var type = this.get('val');
 81 
 82             var d = 7; // arrow width/2
 83             var redim = d + 3; //we have to make the canvas a little bigger because of arrows
 84             var margin = [4 + redim, 4 + redim];
 85 
 86             var src = this.get('src')/*.getXY()*/;
 87             var tgt = this.get('tgt')/*.getXY()*/;
 88 
 89             var distance = Math.sqrt(Math.pow(src[0] - tgt[0], 2) + Math.pow(src[1] - tgt[1], 2));
 90             this.moveTo((src[0] + 6), (src[1] + 6));
 91             this.lineTo((tgt[0] + 6), (tgt[1] + 6));
 92 
 93             // start drawing arrows
 94 
 95             var t1 = src;
 96             var t2 = tgt;
 97 
 98             var z = [0, 0]; //point on the wire with constant distance (dlug) from terminal2
 99             var dlug = 20; //arrow length
100             var t = (distance === 0) ? 0 : 1 - (dlug / distance);
101             z[0] = Math.abs(t1[0] + t * (t2[0] - t1[0]));
102             z[1] = Math.abs(t1[1] + t * (t2[1] - t1[1]));
103 
104             //line which connects the terminals: y=ax+b
105             var W = src[0] - tgt[0],
106                     Wa = src[1] - tgt[1],
107                     Wb = src[0] * tgt[1] - src[1] * tgt[0],
108                     a, b, aProst, bProst;
109             if (W !== 0) {
110                 a = Wa / W;
111                 b = Wb / W;
112             }
113             else {
114                 a = 0;
115             }
116             //line perpendicular to the main line: y = aProst*x + b
117             if (a === 0) {
118                 aProst = 0;
119             }
120             else {
121                 aProst = -1 / a;
122             }
123             bProst = z[1] - aProst * z[0]; //point z lays on this line
124 
125             //we have to calculate coordinates of 2 points, which lay on perpendicular line and have the same distance (d) from point z
126             var A = 1 + Math.pow(aProst, 2);
127             var B = 2 * aProst * bProst - 2 * z[0] - 2 * z[1] * aProst;
128             var C = -2 * z[1] * bProst + Math.pow(z[0], 2) + Math.pow(z[1], 2) - Math.pow(d, 2) + Math.pow(bProst, 2);
129             var delta = Math.pow(B, 2) - 4 * A * C;
130             if (delta < 0) {
131                 return;
132             }
133 
134             var x1 = (-B + Math.sqrt(delta)) / (2 * A);
135             var x2 = (-B - Math.sqrt(delta)) / (2 * A);
136             var y1 = aProst * x1 + bProst;
137             var y2 = aProst * x2 + bProst;
138 
139             if (t1[1] == t2[1]) {
140                 var o = (t1[0] > t2[0]) ? 1 : -1;
141                 x1 = t2[0] + o * dlug;
142                 x2 = x1;
143                 y1 -= d;
144                 y2 += d;
145             }
146 
147             if (type == 1 || type == 3) {
148                 this.moveTo(t2[0] + 6, t2[1] + 6);
149                 this.lineTo(x1 + 6, y1 + 6);
150                 this.moveTo(t2[0] + 6, t2[1] + 6);
151                 this.lineTo(x2 + 6, y2 + 6);
152             }
153 
154             t1 = tgt;
155             t2 = src;
156 
157             var z = [0, 0]; //point on the wire with constant distance (dlug) from terminal2
158             var dlug = 20; //arrow length
159             var t = (distance == 0) ? 0 : 1 - (dlug / distance);
160             z[0] = Math.abs(t1[0] + t * (t2[0] - t1[0]));
161             z[1] = Math.abs(t1[1] + t * (t2[1] - t1[1]));
162 
163             //line which connects the terminals: y=ax+b
164             var W = t1[0] - t2[0];
165             var Wa = t1[1] - t2[1];
166             var Wb = t1[0] * t2[1] - t1[1] * t2[0];
167             if (W !== 0) {
168                 a = Wa / W;
169                 b = Wb / W;
170             }
171             else {
172                 a = 0;
173             }
174             //line perpendicular to the main line: y = aProst*x + b
175             if (a == 0) {
176                 aProst = 0;
177             }
178             else {
179                 aProst = -1 / a;
180             }
181             bProst = z[1] - aProst * z[0]; //point z lays on this line
182 
183             //we have to calculate coordinates of 2 points, which lay on perpendicular line and have the same distance (d) from point z
184             var A = 1 + Math.pow(aProst, 2);
185             var B = 2 * aProst * bProst - 2 * z[0] - 2 * z[1] * aProst;
186             var C = -2 * z[1] * bProst + Math.pow(z[0], 2) + Math.pow(z[1], 2) - Math.pow(d, 2) + Math.pow(bProst, 2);
187             var delta = Math.pow(B, 2) - 4 * A * C;
188             if (delta < 0) {
189                 return;
190             }
191 
192             var x1 = (-B + Math.sqrt(delta)) / (2 * A);
193             var x2 = (-B - Math.sqrt(delta)) / (2 * A);
194             var y1 = aProst * x1 + bProst;
195             var y2 = aProst * x2 + bProst;
196 
197             if (t1[1] == t2[1]) {
198                 var o = (t1[0] > t2[0]) ? 1 : -1;
199                 x1 = t2[0] + o * dlug;
200                 x2 = x1;
201                 y1 -= d;
202                 y2 += d;
203             }
204 
205             if (type == 2 || type == 3) {
206                 this.moveTo(t2[0] + 6, t2[1] + 6);
207                 this.lineTo(x1 + 6, y1 + 6);
208                 this.moveTo(t2[0] + 6, t2[1] + 6);
209                 this.lineTo(x2 + 6, y2 + 6);
210             }
211 
212             this.end();
213         },

235         },
