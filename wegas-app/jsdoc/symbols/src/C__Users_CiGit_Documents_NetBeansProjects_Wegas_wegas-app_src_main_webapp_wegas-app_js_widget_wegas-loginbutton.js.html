 12 YUI.add("wegas-loginbutton", function (Y) {
 13     "use strict";
 14 
 15     var LoginButton;
 16 
 17     /**
 18      * @name Y.Wegas.LoginButton
 19      * @extends Y.Wegas.Button
 20      * @class  Button with a defined behavior.
 21      * @constructor
 22      * @description Button with special label and menu with two
 23      * options : set user preferences or logout
 24      */
 25     LoginButton = Y.Base.create("wegas-login", Y.Wegas.Button, [], {
 26         /** @lends Y.Wegas.LoginButton# */
 27 
 28         // *** Lifecycle Methods *** //
 29         /**
 30          * @function
 31          * @private
 32          * @description bind function to events.
 33          * Call widget parent to execute its proper bind function.
 34          * When UserFacade is updated, do syncUI
 35          * Add plugin menu with 2 options : open page "user preferences" and logout
 36          */
 37         bindUI: function () {
 38             Y.Wegas.LoginButton.superclass.bindUI.apply(this, arguments);
 39             Y.Wegas.UserFacade.after("update", this.syncUI, this);
 40 
 41             if (this.menu) {                                                    // Don't add the plugin if it already exist.
 42                 return;
 43             }
 44 
 45             this.plug(Y.Plugin.WidgetMenu, {
 46                 children: [{
 47                     type: "Button",
 48                     label: "Preferences",
 49                     plugins: [{
 50                         "fn": "OpenPageAction",
 51                         "cfg": {
 52                             "subpageId": this.get("preferencePageId"), // @fixme
 53                             "targetPageLoaderId": this.get("targetPageLoader")
 54                         }
 55                     }]
 56                 }, {
 57                     type: "Button",
 58                     label: "Logout",
 59                     plugins: [{
 60                         fn: "OpenUrlAction",
 61                         cfg: {
 62                             url: "wegas-app/logout",
 63                             target: "self"
 64                         }
 65                     }]
 66                 }]
 67             });
 68         },
 69 
 70         /**
 71          * @function
 72          * @private
 73          * @description Call widget parent to execute its proper sync function.
 74          * Set label of this button with team and/or player name.
 75          */
 76         syncUI: function () {
 77             Y.Wegas.LoginButton.superclass.syncUI.apply(this, arguments);
 78 
 79             var cUser = Y.Wegas.app.get("currentUser"),
 80             cPlayer = Y.Wegas.GameFacade.rest.getCurrentPlayer(),
 81             cTeam = Y.Wegas.GameFacade.rest.getCurrentTeam(),
 82             name = cUser.name || "undefined";
 83             if (!this.get('labelIsUser')) {
 84                 if (cPlayer) {
 85                     name = cPlayer.get("name");
 86                 }
 87                 if (cTeam) {
 88                     name = cTeam.get("name") + " : " + name;
 89                 }
 90             }
 91             this.set("label", name);
 92         }
 93     }, {
 94         /** @lends Y.Wegas.LoginButton */
 95 
 96         /**
 97          * @field
 98          * @static
 99          * @description
100          * <p><strong>Attributes</strong></p>
101          * <ul>
102          *    <li>labelIsUser: Select what kind of label you want (user/team  or team/player)</li>
103          *    <li>preferencePageId: Id of the the page which contains widget userPreferences</li>
104          *    <li>targetPageLoader: Zone to display the page which contains widget userPreferences</li>
105          * </ul>
106          */
107         ATTRS: {
108             /**
109              * Select what kind of label you want (user/team  or team/player)
110              */
111             labelIsUser: {
112                 value: false,
113                 validator: function (b) {
114                     return (b === 'true' || b === true);
115                 }
116             },
117             /**
118              * Id of the the page which contains widget userPreference
119              */
120             preferencePageId: {
121                 value: 1000                                                     //@fixme
122             },
123             /**
124              * targetPageLoader: Zone to display the page which contains widget userPreferences
125              */
126             targetPageLoader: {
127                 value: "maindisplayarea"
128             }
129         }
130     });
131     Y.namespace('Wegas').LoginButton = LoginButton;
132 
133 });
