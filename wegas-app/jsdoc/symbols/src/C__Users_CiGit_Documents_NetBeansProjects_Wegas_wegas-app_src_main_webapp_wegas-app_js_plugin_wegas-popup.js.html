 12 YUI.add('wegas-popup', function(Y) {
 13 
 14     /**
 15      *  @class Show a message when the host widget is rendered, useful for welcome
 16      *  messages
 17      *  @name Y.Plugin.Popup
 18      *  @extends Y.Plugin.Base
 19      *  @constructor
 20      */
 21     var Popup = Y.Base.create("wegas-popup", Y.Plugin.Base, [Y.Wegas.Plugin, Y.Wegas.Editable], {
 22         initializer: function() {
 23             if (this.get("content")) {
 24                 this.afterHostEvent("render", function() {
 25                     if (this.get("host").showMessage) {
 26 
 27                         this.get("host").showMessage("info", this.get("content"));
 28                     }
 29                 });
 30             }
 31 
 32             Y.Wegas.Facade.VariableDescriptor.on(this.get("event"), function(e) {
 33                 this.get("host").showMessage("info", e.content);
 34             }, this);
 35         }
 36     }, {
 37         NS: "Popup",
 38         NAME: "Popup",
 39         ATTRS: {
 40             content: {
 41                 type: "string",
 42                 format: "text"
 43             },
 44             event: {
 45                 value: "popupEvent",
 46                 type: "string"
 47             }
 48         }
 49     });
 50     Y.Plugin.Popup = Popup;
 51 
 52 });
