282         // *** Private methods *** //
283         /**
284          * @function
285          * @private
286          * @param showAskForm, a boolean
287          * @description if showAskForm is true, hide "signup" form and "forgot Password" <p/> and show "ask password" form
288          * do the opposite else.
289          */
290         changeRightForms: function(showAskForm) {
291             var cb = this.get(CONTENTBOX);
292             if (showAskForm) {
293                 cb.one(".ask-pass-zone").show();
294                 cb.one(".signup-zone").hide();

302         /**
303          * @function
304          * @private
305          * @param email
306          * @param password
307          * @param remember
308          * @description Send REST request to login a user with form's values
309          *  informations.
310          */
311         login: function(email, password, remember) {
312             Y.Wegas.Facade.User.sendRequest({
313                 request: "/Authenticate/?email=" + email
314                         + "&password=" + password
315                         + "&remember=" + remember,
316                 cfg: {
317                     method: "POST"
318                 },
319                 on: {
320                     success: Y.bind(function(e) {
321                         this.showMessage("success", "Login successful");
322                         this.redirect();
323                     }, this),
324                     failure: Y.bind(function(e) {
325                         this.showMessage("error", e.response.results.message || "Email/password combination not found", 6000);
326                     }, this)
327                 }
328             });
329         },
330         guestLogin: function() {
331             Y.Wegas.Facade.User.sendRequest({
332                 request: "/GuestLogin/",
333                 cfg: {
334                     method: "POST"
335                 },
336                 on: {
337                     success: Y.bind(function(e) {
338                         this.showMessage("success", "Login successful");
339                         this.redirect();
340                     }, this),
341                     failure: Y.bind(function(e) {
342                         this.showMessage("error", e.response.results.message || "Guest login failed", 6000);
343                     }, this)
344                 }
345             });

347         /**
348          * @function
349          * @private
350          * @param data
351          * @description Send REST request to create a new account based on the
352          *  form's values
353          */
354         createAccount: function(data) {
355             Y.Wegas.Facade.User.sendRequest({
356                 request: "/Signup/",
357                 cfg: {
358                     method: "POST",
359                     data: data
360                 },
361                 on: {
362                     success: Y.bind(function(data, e) {
363                         //this.showMessage("success", "User created", 6000);
364                         //this.set("mode", "login");
365                         this.login(data.email, data.password, false);
366                     }, this, data),
367                     failure: Y.bind(function(e) {
368                         this.showMessage("error", e.response.results.message || "Error creating user");
369                     }, this)
370                 }
371             });
372         },
373         /**
374          * @function
375          * @private
376          * @param email
377          * @description Send REST request to set and set a new password for
378          *  the user.
379          */
380         sendNewPassword: function(email) {
381             Y.Wegas.Facade.User.sendRequest({
382                 request: "/SendNewPassword/?email=" + email,
383                 cfg: {
384                     method: "POST"
385                 },
386                 on: {
387                     success: Y.bind(function(e) {
388                         this.hideOverlay();
389                         this.showMessage("success", "A new password has been sent", 4000);
390                         this.changeRightForms(false);
391                     }, this),
392                     failure: Y.bind(function(e) {
393                         this.hideOverlay();
394                         this.showMessage("error", e.response.results.message || "Error sending new password", 4000);
395                     }, this)
396                 }
397             });
398         },
399         /**
400          * @function
401          * @private
402          * @description redirect the user to a page given by the function 'getRedirect'
403          */
404         redirect: function() {
405             window.location = this.getRedirect();
406         },
407         /**
408          * @function
409          * @private
410          * @return url to redirect
411          * @description return redirection given by the function
412          *  'getQueryParameter' or by the default redirection.
413          */
414         getRedirect: function() {
415             return Y.Wegas.Helper.getURLParameter("redirect") || (Y.Wegas.app.get("base") + this.defaultRedirect);
416         }

419     /**
420      * Hack because "typeInvite" and password work bad (typeInvite is hid)
421      * Password field needs a "password" class.
422      * Change color property (in grey) when input is not focused (black else)
423      */
424     Y.inputEx.StringField.prototype.updateTypeInvite = function() {
425 
426         // field not focused
427         if (!Y.one(this.divEl).hasClass("inputEx-focused")) {
428 
429             // show type invite if field is empty
430             if (this.isEmpty()) {
431                 Y.one(this.divEl).addClass("inputEx-typeInvite");
432                 Y.one(this.divEl).one("input").setStyle("color", "#888");
433                 if (this.fieldContainer.className.indexOf("password") > -1) {
434                     this.el.setAttribute("type", "");
435                 }
436                 this.el.value = this.options.typeInvite;
437 
438                 // important for setValue to work with typeInvite
439             } else {
440                 if (this.fieldContainer.className.indexOf("password") > -1) {
441                     this.el.setAttribute("type", "password");
442                 }
443                 Y.one(this.divEl).one("input").setStyle("color", "#000");
444                 Y.one(this.divEl).removeClass("inputEx-typeInvite");
445             }
446 
447             // field focused : remove type invite
448         } else {
449             if (Y.one(this.divEl).hasClass("inputEx-typeInvite")) {
450                 // remove text
451                 this.el.value = "";
452                 // remove the "empty" state and class
453                 this.previousState = null;
454                 if (this.fieldContainer.className.indexOf("password") > -1) {
455                     this.el.setAttribute("type", "password");
456                 }
457                 Y.one(this.divEl).one("input").setStyle("color", "#000");
458                 Y.one(this.divEl).removeClass("inputEx-typeInvite");
459             }
460         }
461     };
