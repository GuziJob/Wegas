 12 YUI.add('wegas-panelwidget', function (Y) {
 13     "use strict";
 14 
 15     /**
 16     * @name Y.Wegas.PanelWidget
 17     * @extends Y.Panel
 18     * @class  class for creating panel from a JSON file with children
 19     * @constructor
 20     * @param Object Will be used to fill attributes field
 21     * @description Create a panel with all children
 22     */
 23     var PanelWidget = Y.Base.create("wegas-panelwidget", Y.Panel, [Y.WidgetChild, Y.Wegas.Widget], {
 24         /** @lends Y.Wegas.PanelWidget# */
 25 
 26         /**
 27          * @function
 28          * @private
 29          * @description Instantiates all children widget
 30          */
 31         syncUI: function () {
 32             PanelWidget.superclass.syncUI.apply(this);
 33             Y.Array.forEach(this.get("children"), function (child) {
 34                 var widget = Y.Wegas.Widget.create(child);
 35                 widget.render(this.get("contentBox"));
 36             }, this);
 37         }
 38 
 39     }, {
 40         /** @lends Y.Wegas.PanelWidget */
 41 
 42         /**
 43          * @field
 44          * @static
 45          * @description
 46          * <p><strong>Attributes</strong></p>
 47          * <ul>
 48          *    <li>chlidren: All children widget used</li>
 49          *    <li>zIndex : define a default zIndex</li>
 50          * </ul>
 51          */
 52         ATTRS: {
 53             children: {
 54                 value: []
 55             },
 56             zIndex: {
 57                 value: 6
 58             }
 59         }
 60 
 61     });
 62     Y.namespace('Wegas').PanelWidget = PanelWidget;
 63 
 64 });
