 21     /**
 22      * @name Y.Wegas.Button
 23      * @extends Y.Button
 24      * @borrows Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable
 25      * @class Custom Button implementation.
 26      * @constructor
 27      * @description Custom Button implementation. Adds Y.WidgetChild and
 28      * Y.Wegas.Widget extensions to the original Y.Button
 29      */
 30     Button = Y.Base.create("button", Y.Button, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 31         /** @lends Y.Wegas.Button# */
 32         // *** Private fields *** //
 33 
 34         // *** Lifecycle Methods *** //
 35         /**
 36          * @function
 37          * @private
 38          * @description Set variable with initials values.
 39          * Plug tooltip add given (by ATTRS) css class to contentbox
 40          */
 41         initializer: function () {
 42             Button.superclass.initializer.apply(this, arguments);
 43 
 44             //this.constructor.CSS_PREFIX = "yui3-button";                      // Revert changes done by Y.Wegas.Widget so styling will work
 45             this._cssPrefix = "yui3-button";
 46 
 47             if (this.get("cssClass")) {
 48                 this.get(CONTENTBOX).addClass(this.get("cssClass"));
 49             }
 50 
 51             if (this.get("tooltip")) {
 52                 this.plug(Y.Plugin.Tooltip, {
 53                     content: this.get("tooltip")
 54                 });
 55             }
 56         },
 57 
 58         /**
 59          * @function
 60          * @private
 61          * @description Call widget parent to execute its proper render function.
 62          * add "wegas-button" class to bounding box.
 63          */
 64         renderUI: function () {
 65             Button.superclass.renderUI.apply(this, arguments);
 66             this.get(BOUNDINGBOX).addClass("wegas-button");
 67         }
 68     }, {
 69         /**
 70          * @lends Y.Wegas.Button
 71          */
 72         /**
 73          * @field
 74          * @static
 75          * @description
 76          * <p><strong>Attributes</strong></p>
 77          * <ul>
 78          *    <li>label: the label of the button</li>
 79          *    <li>data: the data used by the button</li>
 80          *    <li>tooltip: the tooltip of the button</li>
 81          *    <li>disabled: boolean to choose state of the button</li>
 82          *    <li>cssClass: cssClass of the button</li>
 83          *    <li>plugins: impact to bind at the buttons</li>
 84          * </ul>
 85          */
 86         ATTRS: {
 87             label: {
 88                 type: "string"
 89             },
 90             data: {},
 91             tooltip: {
 92                 type: "string",
 93                 optional: true,
 94                 "transient": true
 95             },
 96             disabled: {
 97                 "transient": false,
 98                 type: "boolean"
 99             },
100             cssClass: {
101                 value: null
102             },
103             plugins: {
104                 "transient": false,
105                 _inputex: {
106                     _type: "editablelist",
107                     items: [{
108                             type: "Button",
109                             label: "Tooltip",
110                             data: "Tooltip"
111                         }, {
112                             type: "Button",
113                             label: "On click",
114                             data: "ExecuteScriptAction"
115                         }, {
116                             type: "Button",
117                             label: "Open page",
118                             data: "OpenPageAction"
119                         }]
120                 }
121             }
122         }
123     });

136     /**
137      * Plugin which adds an unread message counter to a widget.
138      *
139      * @class Y.Wegas.UnreadCount
140      * @extends Y.Plugin.Base
141      * @borrows Y.Wegas.Plugin, Y.Wegas.Editable
142      */
143     var UnreadCount = Y.Base.create("wegas-unreadCount", Y.Plugin.Base, [Y.Wegas.Plugin, Y.Wegas.Editable], {
144         /** @lends Y.Plugin.UnreadCount# */
145 
146         // *** Private fields *** //
147         /**
148          * Reference to each used functions
149          */
150         handlers: null,
151 
152         // *** Lifecycle methods *** //
153         /**
154          * @function
155          * @private
156          * @description Set variable with initials values.
157          */
158         initializer: function () {
159             this.handlers = {};
160             this.bindUI();
161         },
162 
163         /**
164          * @function
165          * @private
166          * @description bind function to events.
167          * When VariableDescriptorFacade is updated, do sync.
168          * When plugin's host is render, do sync.
169          */
170         bindUI: function () {
171             this.handlers.update = Y.Wegas.app.VariableDescriptorFacade.after("update", this.syncUI, this);
172             this.handlers.render = this.afterHostEvent("render", this.syncUI, this);
173         },
174 
175         /**
176          * @function
177          * @private
178          * @description call function 'getUnreadCount' to set the number of
179          * unread on the host.
180          */
181         syncUI: function () {
182             var cb = this.get('host').get(CONTENTBOX),
183                     target = cb.one(".unread-count"),
184                     unreadCount = this.getUnreadCount();
185 
186             if (!target) {                                                      // If the counter span has not been rendered, do it
187                 cb.append('<span class="unread-count"></span>');
188                 target = cb.one(".unread-count");
189             }
190 
191             if (unreadCount > 0) {                                              // Update the content
192                 target.setContent("<span class='value'>" + unreadCount + "</span>");
193             } else {
194                 target.setContent("");
195             }
196         },
197 
198         /**
199          * @function
200          * @private
201          * @description Detach all functions created by this widget
202          */
203         destructor: function () {
204             for (var k in this.handlers) {
205                 this.handlers[k].detach();
206             }
207         },
208 
209         // *** Private methods *** //
210        /**
211          * @function
212          * @private
213          * @return Number of unread.
214          * @description Count the number of unread reply in given variable.
215          */
216         getUnreadCount: function () {
217             var i, instance, messages, count = 0,
218                     descriptor = this.get('variable.evaluated');
219 
220             if (!descriptor) {
221                 return 0;
222             }
223 
224             if (descriptor.get("items")) {                                      // For ListDescriptors, we count the children instance's
225                 for (i = 0; i < descriptor.get("items").length; i = i + 1) {
226                     instance = descriptor.get("items")[i].getInstance();
227                     //count += instance.get("unread") ? 1 : 0;
228                     if (instance.get("replies")) {
229                         count += instance.get("replies").length === 0 && instance.get("active") ? 1 : 0; // only count if it is active
230                     }
231                 }
232             }
233 
234             messages = descriptor.getInstance().get("messages");                // For InboxVariableDescriptors, we count the replies
235             if (messages) {
236                 for (i = 0; i < messages.length; i = i + 1) {
237                     count += messages[i].get("unread") ? 1 : 0;
238                 }
239             }
240 
241             return count;
242         }
243     }, {
244         NS: "UnreadCount",
245         NAME: "UnreadCount",
246         /**
247          * @lends Y.Plugin.UnreadCount
248          */
249         /**
250          * @field
251          * @static
252          * @description
253          * <p><strong>Attributes</strong></p>
254          * <ul>
255          *    <li>variable: The target variable, returned either based on the name
256          *     attribute, and if absent by evaluating the expr attribute.</li>
257          * </ul>
258          */
259         ATTRS: {
260             /**
261              * The target variable, returned either based on the variableName attribute,
262              * and if absent by evaluating the expr attribute.
263              */
264             variable: {
265                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
266                 _inputex: {
267                     _type: "variableselect",
268                     label: "List to count unread"
269                 }
270             }
271         }
272     });
273     Y.namespace('Plugin').UnreadCount = UnreadCount;
