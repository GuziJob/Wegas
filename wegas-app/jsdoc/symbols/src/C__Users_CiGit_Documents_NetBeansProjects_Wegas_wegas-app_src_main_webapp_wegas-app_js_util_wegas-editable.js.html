 13 YUI.add('wegas-editable', function(Y) {
 14     "use strict";
 15 
 16     var Lang = Y.Lang;
 17 
 18     /**
 19      *  Add custom attributes to be used in ATTR param in static cfg.
 20      */
 21     Y.Base._ATTR_CFG.push("type", "properties", "_inputex", "optional", "format",
 22             "choices", "items", "enum", "default", "transient");
 23     Y.Base._ATTR_CFG_HASH = Y.Array.hash(Y.Base._ATTR_CFG);
 24 
 25     /**
 26      * @name Y.Wegas.Editable
 27      * @class Extension to be added to a Y.Widget, allowing edition on wegas entities.
 28      * @constructor
 29      */
 30     function Editable() {
 31     }
 32 
 33     Y.mix(Editable.prototype, {
 34         /** @lends Y.Wegas.Editable# */
 35 
 36         /**
 37          * Serialize to a json object. Method used <b>recursively</b> by JSON.stringify
 38          *
 39          * @function
 40          * @returns {Object}
 41          */
 42         toJSON: function() {
 43             var k, ret = this.getAttrs(),
 44                     attrCfgs = this.getAttrCfgs();
 45 
 46             for (k in ret) {
 47                 if (attrCfgs[k] && attrCfgs[k]["transient"]) {                  // Remove any transient attribute
 48                     delete ret[k];
 49                 }
 50             }
 51             return ret;                                                         // Return a copy of this's fields.
 52         },
 53         /**
 54          * Create a new JSON Object from this entity, filtered out by mask
 55          *
 56          * @fixme This method will parse the field and the clone it
 57          *
 58          * @function
 59          * @param {Array|String} mask or a list of string parameters
 60          * @return {Object} a filtered out clone
 61          */
 62         toObject: function(mask) {
 63             var masker;
 64             mask = Lang.isArray(mask) ? mask : Array.prototype.slice.call(arguments);
 65             masker = mask.length > 0 ? function(key, value) {
 66                 if (mask.indexOf(key) !== -1) {
 67                     return undefined;
 68                 } else {
 69                     return value;
 70                 }
 71             } : null;
 72             return Y.JSON.parse(Y.JSON.stringify(this), masker);
 73         },
 74         /**
 75          * Create a new Object from this entity
 76          * may be used by revive
 77          * @method clone
 78          * @return {Object} a clone
 79          */
 80         clone: function() {
 81             return this.toObject(["id", "variableInstances"]);
 82         },
 83         /**
 84          * Returns the form configuration associated to this object, to be used a an inputex object.
 85          * @param {Array} fieldsToIgnore (optional), don't create these inputs.
 86          */
 87         getFormCfg: function(fieldsToIgnore) {
 88             var i, form, schemaMap, attrCfgs, builder;
 89             fieldsToIgnore = (fieldsToIgnore || []);
 90             // forms = Y.Wegas.app.get('editorForms'),                          // Select first server defined forms, based on the @class or the type attribute
 91             // form = forms[this.get('@class')] || forms[this.get("type")]
 92 
 93             form = form || this.constructor.EDITFORM;                           // And if no form is defined we check if there is a default one defined in the entity
 94 
 95             if (!form) {                          // If no edit form could be found, we generate one based on the ATTRS parameter.
 96                 attrCfgs = this.getAttrCfgs();
 97 
 98                 for (i in attrCfgs) {
 99                     attrCfgs[i]["default"] = attrCfgs[i].value;// Use the value as default (useful form json object serialization)
100 
101                     if (attrCfgs[i]["transient"] || fieldsToIgnore.indexOf(i) > -1) {
102                         delete attrCfgs[i];
103                     }
104                 }
105 
106                 schemaMap = {
107                     Entity: {
108                         properties: attrCfgs
109                     }
110                 };
111 
112                 builder = new Y.inputEx.JsonSchema.Builder({
113                     'schemaIdentifierMap': schemaMap,
114                     'defaultOptions': {
115                         'showMsg': true
116                     }
117                 });
118                 form = builder.schemaToInputEx(schemaMap.Entity);
119             }
120             return form || [];
121         },
122         /**
123          * Returns the edition menu associated to this object, to be used a an inputex object.
124          *
125          * @function
126          */
127         getMenuCfg: function(data) {
128             var menu, menus = Y.Wegas.app.get('editorMenus');
129             //    staticMenus =
130 
131             if (menus) {
132                 menu = menus[this.get('@class')] || menus[this.get("type")];    // Select first server defined forms, based on the @class or the type attribute
133             }
134             menu = menu || this.getStatic("EDITMENU", true)[0] || [];                // And if no form is defined we return the default one defined in the entity
135 
136             function mixMenuCfg(elts, data) {
137                 var i, j;
138                 for (i = 0; i < elts.length; i += 1) {
139                     Y.mix(elts[i], data, true);                                 // Attach self and the provided datasource to the menu items, to allow them to know which entity to update
140 
141                     if (elts[i].plugins) {
142                         for (j = 0; j < elts[i].plugins.length; j = j + 1) {
143                             elts[i].plugins[j].cfg = elts[i].plugins[j].cfg || {};
144                             Y.mix(elts[i].plugins[j].cfg, data, true);
145                             if (elts[i].plugins[j].cfg.children) {
146                                 mixMenuCfg(elts[i].plugins[j].cfg.children, data); // push data in children arg
147                             }
148                             if (elts[i].plugins[j].cfg.wchildren) {
149                                 mixMenuCfg(elts[i].plugins[j].cfg.wchildren, data);// push data in wchildren
150                             }
151                         }
152                     }
153                 }
154 
155             }
156 
157             mixMenuCfg(menu, data);
158             return menu;
159         },
160         /**
161          * Returns the edition menu associated to this object, to be used a an wysiwyg editor.
162          * @function
163          */
164         getMethodCfgs: function(data) {
165             var menu = this.getStatic("METHODS")[0] || {};
166             return menu;
167         },
168         /**
169          *  Helper function that walks the class hierarchy and returns it's attributes
170          *  cfg (ATTRS), used in Y.Wegas.Entity.getFormCfg().
171          *  @function
172          *  @private
173          */
174         getAttrCfgs: function() {
175             return this._aggregateAttrs(this.getStatic("ATTRS"));
176         },
177         /**
178          *  @function
179          *  @private
180          */
181         getStatic: function(key, withExtensions) {
182             var c = this.constructor, ret = [], i;
183 
184             while (c) {
185                 if (c[key]) {                                                  // Add to attributes
186                     ret[ret.length] = c[key];
187                 }
188                 if (withExtensions && c._yuibuild && c._yuibuild.exts) {
189                     for (i = 0; i < c._yuibuild.exts.length; i += 1) {
190                         if (c._yuibuild.exts[i][key]) {
191                             ret.push(c._yuibuild.exts[i][key]);
192                         }
193                     }
194                 }
195                 c = c.superclass ? c.superclass.constructor : null;
196             }
197             return ret;
198         },
199         /**
200          * Retrives editable's editorname or name
201          * @function
202          * @public
203          * @returns {String} static EDITORNAME or NAME
204          */
205         getName: function() {
206             return this.constructor.EDITORNAME || this.constructor.NAME;
207         },
208         /**
209          * Check if this widget is augmented (extended) by a specifique Widget
210          * @param {Class} extension to check for
211          * @returns {Boolean}
212          */
213         isAugmentedBy: function(extension) {
214             var self = this.constructor, i;
215             while (self._yuibuild) {
216                 for (i = 0; i < self._yuibuild.exts.length; i += 1) {
217                     if (self._yuibuild.exts[i] === extension) {
218                         return true;
219                     }
220                 }
221                 self = self.superclass.constructor;
222             }
223             return false;
224 
225         }
226     });
227     Y.mix(Editable, {
228         /** @lends Y.Wegas.Editable */
229 
230         /**
231          * Load the modules from an Wegas widget definition
232          * @function
233          * @static
234          * @param {Object}
235          * @param {Function} cb callback to be called when modules are loaded
236          */
237         use: function(cfg, cb) {
238             var modules = Editable.getModulesFromDefinition(cfg);
239             modules.push(cb);
240             Y.use.apply(Y, modules);
241         },
242         /**
243          * Return recursively the inputex modules from their 'type' property using (modulesByType from loader.js)
244          * @function
245          * @static
246          * @private
247          */
248         getRawModulesFromDefinition: function(cfg) {
249             var i, props, type = cfg.type || cfg["@class"],
250                     module = YUI_config.groups.wegas.modulesByType[type],
251                     modules = [],
252                     pushFn = function(field) {
253                 if (field) {
254                     modules = modules.concat(Editable.getModulesFromDefinition(field));
255                 }
256             };
257 
258             if (module) {
259                 modules.push(module);
260             }
261 
262             props = ["children", "entities", "items"];                          // Revive array attributes
263             for (i = 0; i < props.length; i += 1) {
264                 if (cfg[props[i]]) {                                            // Get definitions from children (for Y.WidgetParent widgets)
265                     Y.Array.each(cfg[props[i]], pushFn);
266                 }
267             }
268             if (cfg.plugins) {                                                  // Plugins must be revived in the proper way
269                 Y.Array.each(cfg.plugins, function(field) {
270                     field.cfg = field.cfg || {};
271                     field.cfg.type = field.fn;
272                     modules = modules.concat(Editable.getModulesFromDefinition(field.cfg));
273                 });
274             }
275 
276             props = ["left", "right", "center", "top", "bottom"];               // Revive  objects attributes
277             for (i = 0; i < props.length; i = i + 1) {
278                 if (cfg[props[i]]) {
279                     modules = modules.concat(Editable.getModulesFromDefinition(cfg[props[i]]));
280                 }
281             }
282 
283             return modules;
284         },
285         /**
286          * Return unique modules definitions
287          * @function
288          * @static
289          * @private
290          */
291         getModulesFromDefinition: function(cfg) {
292             var modules = Editable.getRawModulesFromDefinition(cfg);
293             return Y.Object.keys(Y.Array.hash(modules));
294         },
295         /**
296          *  This method takes a js object and recuresively instantiate it based on
297          *  on their @class attribute. Target class are found in namespace
298          *  Y.Wegas.persistence.
299          *
300          *  @static
301          *  @param {Object} data the object to revive
302          *  @return Y.Wegas.Widget the resulting entity
303          */
304         revive: function(data) {
305             var walk = function(o, key) {
306                 var k, v, value = o[key];
307                 if (value && typeof value === "object") {
308                     for (k in value) {
309                         if (value.hasOwnProperty(k)) {
310                             v = walk(value, k);
311                             if (!Lang.isUndefined(v)) {
312                                 value[k] = v;
313                             }
314                         }
315                     }
316                     if (!Lang.isUndefined(value["@class"]) || !Lang.isUndefined(value.type)) {
317                         return Editable.reviver(value);
318                     }
319                 }
320                 return value;                                      // If no value was returned before, return raw original object
321             };
322 
323             //return typeof reviver === 'function' ? walk({'':data},'') : data;
324             return walk({
325                 '': data
326             }, '');
327         },
328         /**
329          *  Takes an js object and lookup the corresponding entity in the
330          *  Y.Wegas.persistence package, based on its @class or type attributes.
331          *
332          *  @function
333          *  @static
334          *  @param {Object} o the object to revive
335          *  @return {Y.Wegas.Widget} the resulting entity
336          */
337         reviver: function(o) {
338             var classDef = Y.Wegas.persistence.DefaultEntity;
339 
340             if (o["@class"]) {
341                 classDef = Y.Wegas.persistence[o["@class"]] || Y.Wegas.persistence.DefaultEntity;
342 
343             } else if (o.type) {
344                 classDef = Y.Wegas.persistence[o.type] || Y.Wegas.persistence.WidgetEntity;
345 
346             }
347             return new classDef(o);
348         },
349         /**
350          *
351          * Combine use and revive function to get a revived entity.
352          *
353          * @function
354          * @static
355          */
356         useAndRevive: function(cfg, cb) {
357             Editable.use(cfg, Y.bind(function(cb) {                            // Load target class dependencies
358                 cb(Editable.revive(this));
359             }, cfg, cb));
360         }
361     });
362     Y.namespace("Wegas").Editable = Editable;
363
