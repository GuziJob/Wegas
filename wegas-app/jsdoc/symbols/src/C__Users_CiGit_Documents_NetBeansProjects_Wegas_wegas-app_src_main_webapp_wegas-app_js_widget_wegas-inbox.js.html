 12 YUI.add('wegas-inbox', function(Y) {
 13     "use strict";
 14 
 15     var CONTENTBOX = 'contentBox', InboxDisplay;
 16 
 17     /**
 18      * @name Y.Wegas.InboxDisplay
 19      * @extends Y.Widget
 20      * @borrows Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable
 21      * @class  class to manage e-mail
 22      * @constructor
 23      * @description Display and allow to manage e-mail sent to the current player
 24      */
 25     InboxDisplay = Y.Base.create("wegas-inbox", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 26         /**
 27          * @lends Y.Wegas.InboxDisplay#
 28          */
 29         // ** Private fields ** //
 30         /**
 31          *
 32          */
 33         readRequestTid: null,
 34         // *** Lifecycle Methods *** //
 35         /**
 36          * @function
 37          * @private
 38          * @description Set variable with initials values.
 39          * Plug a toolbar widget (and add the delete button at this toolbar).
 40          */
 41         initializer: function() {
 42             this.dataSource = Y.Wegas.Facade.VariableDescriptor;
 43             this.handlers = {};
 44             this.jsTranslator = new Y.Wegas.JSTranslator();
 45         },
 46         /**
 47          * @function
 48          * @private
 49          * @description Render the TabView widget in the content box.
 50          */
 51         renderUI: function() {
 52             var cb = this.get(CONTENTBOX);
 53 
 54             this.tabView = new Y.TabView();                                     //TabView widget used to display message header and body
 55             this.tabView.render(cb);
 56             this.tabView.get("boundingBox").addClass("horizontal-tabview");
 57             cb.append("<div style='clear:both'></div>");
 58 
 59             //this.plug(Y.Plugin.WidgetToolbar);
 60             if (this.toolbar) {
 61                 var delBtn = new Y.Wegas.Button({
 62                     label: "<span class='wegas-icon wegas-icon-cancel'></span>" + this.jsTranslator.getRB().Delete
 63                 });
 64                 delBtn.on("click", function(e) {
 65                     var selection = this.tabView.get("selection");
 66                     if (!selection || !selection.msg) {
 67                         return;
 68                     }
 69                     if (confirm('The e-mail "' + selection.msg.get("subject") + '" will be deleted permanently. Continue ?')) {
 70                         this.deleteEmail(selection.msg);
 71                     }
 72                 }, this);
 73                 this.toolbar.add(delBtn);
 74             }
 75         },
 76         /**
 77          * @function
 78          * @private
 79          * @description bind function to events.
 80          * When selection change on tabView, call function "onTabSelected"
 81          * When dataSource is updated, do syncUI
 82          * When deleteButton is clicked, show an alert which allow to delete selected message
 83          */
 84         bindUI: function() {
 85             this.tabView.after("selectionChange", this.onTabSelected, this);
 86             this.handlers.dataUpdated = this.dataSource.after("update", function(e) {
 87                 if (e.tId !== this.readRequestTid) {
 88                     this.syncUI();
 89                 }
 90             }, this);
 91         },
 92         /**
 93          * @function
 94          * @private
 95          * @description Clear and re-fill th TabView with current player message.
 96          * display a message if there is no message.
 97          * Re-select the current selected msg;
 98          */
 99         syncUI: function() {
100             var i, msg, tab, from,
101                     inboxDescriptor = this.get('variable.evaluated'),
102                     inboxVariable, messages,
103                     selection = this.tabView.get("selection"),
104                     oldMsg = selection && selection.msg;
105             if (!inboxDescriptor) {
106                 this.tabView.add({
107                     label: '',
108                     content: '<center>Unable to find inbox variable</center>'
109                 });
110                 return;
111             }
112             inboxVariable = inboxDescriptor.getInstance();
113             messages = inboxVariable.get("messages");
114 
115             this.isSyncing = true;
116 
117             this.tabView.removeAll();
118             for (i = messages.length - 1; i >= 0; i -= 1) {
119                 msg = messages[i];
120                 from = msg.get("from") || "<i>No sender</i>";
121                 tab = new Y.Tab({
122                     label: '<div class="' + (msg.get("unread") ? "unread" : "read") + '"><div class="left">' + from + '</div>'
123                             + '<div class="right">' + msg.get("subject") + '</div></div>',
124                     content: '<div class="msg-header">'
125                             + '<div class="msg-subject">Subject: ' + msg.get("subject") + '</div>'
126                             + '<div class="msg-from">From: ' + from + '</div>'
127                             + '</div>'
128                             + '<div class="msg-body"><center><em><i>Loading</i></em></center></div>'
129                 });
130                 this.tabView.add(tab);
131                 tab.msg = msg;
132 
133                 if (oldMsg && oldMsg.get("id") === msg.get("id")) {
134                     tab.set("selected", 2);
135                 }
136             }
137 
138             if (messages.length === 0) {
139                 this.tabView.add({
140                     label: '',
141                     content: '<center><i>You have no messages</i></center>'
142                 });
143             }
144 
145             this.isSyncing = false;
146             if (!this.tabView.get("selection")) {
147                 this.tabView.selectChild(0);
148             }
149         },
150         /**
151          * @function
152          * @private
153          * @description Destroy TabView and all functions created by this widget
154          */
155         destructor: function() {
156             this.tabView.destroy();
157             this.handlers.dataUpdated.detach();
158         },
159         // *** Private Methods *** //
160         /**
161          * @function
162          * @private
163          * @description Send a REST request to delete selected message
164          */
165         deleteEmail: function(msg) {
166             this.dataSource.sendRequest({
167                 request: "/Inbox/Message/" + msg.get("id"),
168                 cfg: {
169                     method: "DELETE"
170                 }
171             });
172         },
173         /**
174          * @function
175          * @private
176          * @description retrieve selected message's body and set this as "read" if
177          * tab has been selected longer than 2000 miliseconde.
178          */
179         onTabSelected: function(e) {
180             var i, attachement, attachements = [];
181             if (this.isSyncing) {
182                 return;
183             }
184 
185             if (this.timer) {                                                   // If there is an active unread
186                 this.timer.cancel();                                            // message timer, we cancel it.
187             }
188 
189             if (e.newVal && e.newVal.msg) {
190                 this.dataSource.sendRequest({// Retrieve the message body from the server
191                     request: "/Inbox/Message/" + e.newVal.msg.get("id") + "?view=Extended",
192                     cfg: {
193                         updateCache: false
194                     },
195                     on: {
196                         success: Y.bind(function(e) {
197                             if (e.response.entity.get("attachements") && e.response.entity.get("attachements").length > 0) {
198                                 for (i = 0; i < e.response.entity.get("attachements").length; i += 1) {
199                                     attachement = e.response.entity.get("attachements")[i];
200                                     attachements.push("<a href='" + attachement + "' data-file='" + attachement + "'>"
201                                             + attachement + "</a>");
202                                 }
203                                 if (!this.get("panelNode").one(".msg-header .msg-attachement")) {
204                                     this.get("panelNode").one(".msg-header").append("<div class='msg-attachement'></div>");
205                                 }
206                                 this.get("panelNode").one(".msg-header .msg-attachement")
207                                         .setHTML("Attachements: " + attachements.join("; "));
208                             }
209                             this.get("panelNode").one(".msg-body")
210                                     .setHTML(e.response.entity.get("body") || "<center><em><i>Empty</i></center>");
211                         }, e.newVal)
212                     }
213                 });
214                 if (e.newVal.msg.get("unread")) {                               // If the message is currently unread,
215                     this.timer = Y.later(this.get("setToReadAfter") * 1000, this,
216                             function(msg, tab) {                                // Send a request to mark it as read
217                                 Y.log("Sending message read update", "info", "InboxDisplay");
218                                 msg.set("unread", false);
219                                 tab.get("contentBox").one(".unread").removeClass("unread").addClass("read");
220                                 this.readRequestTid = this.dataSource.sendRequest({
221                                     request: "/Inbox/Message/Read/" + msg.get("id"),
222                                     cfg: {
223                                         method: "PUT"
224                                     },
225                                     on: {
226                                         success: Y.bind(function(tab) {
227                                             //tab.get("contentBox").one(".unread").removeClass("unread").addClass("read");
228                                         }, this, tab)
229                                     }
230                                 });
231                             }, [e.newVal.msg, e.newVal]);
232                 }
233             }
234         },
235         getEditorLabel: function() {
236             var variable = this.get("variable.evaluated");
237             if (variable) {
238                 return variable.getEditorLabel();
239             }
240             return null;
241         }
242 
243     }, {
244         /**
245          * @lends Y.Wegas.InboxDisplay
246          */
247         EDITORNAME: "Inbox",
248         /**
249          * @field
250          * @static
251          * @description
252          * <p><strong>Attributes</strong></p>
253          * <ul>
254          *    <li>variable: The target variable, returned either based on the name
255          *     attribute, and if absent by evaluating the expr attribute.</li>
256          * </ul>
257          */
258         ATTRS: {
259             /**
260              * The target variable, returned either based on the name
261              * attribute, and if absent by evaluating the expr attribute.
262              */
263             variable: {
264                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
265                 _inputex: {
266                     _type: "variableselect",
267                     label: "variable",
268                     classFilter: ["InboxDescriptor"]
269                 }
270             },
271             setToReadAfter: {
272                 value: 0.05,
273                 type: "number",
274                 _inputex: {
275                     label: "Set to read after (s.)"
276                 }
277             }
278         }
279     });
280     Y.namespace('Wegas').InboxDisplay = InboxDisplay;
281 
282 });
