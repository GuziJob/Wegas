 12 YUI.add('wegas-pmg-plannificationactivitycolor', function(Y) {
 13     "use strict";
 14 
 15     /**
 16      *  @class color plannification in datatable
 17      *  @name Y.Plugin.Plannificationcolor
 18      *  @extends Y.Plugin.Base
 19      *  @constructor
 20      */
 21     var Wegas = Y.Wegas,
 22             PlannificationActivityColor = Y.Base.create("wegas-pmg-plannificationactivitycolor", Y.Plugin.Base, [Wegas.Plugin, Wegas.Editable], {
 23         /** @lends Y.Plugin.Plannificationcolor */
 24 
 25         /**
 26          * Lifecycle methods
 27          * @function
 28          * @private
 29          */
 30         initializer: function() {
 31             this.onceAfterHostEvent("render", function() {
 32                 this.findCell();
 33                 this.afterHostMethod("syncUI", this.findCell);
 34                 this.get("host").datatable.after("sort", this.findCell, this);
 35             }, this);
 36         },
 37         findCell: function() {
 38             Y.log("sync()", "log", "Wegas.PlannificationActivityColor");
 39             var i, ii, host = this.get("host"), dt = host.datatable,
 40                     taskActivities = this.taskActivitiesToAdd();
 41 
 42             for (i = 0; i < dt.data.size(); i++) {
 43                 for (ii = 0; ii < taskActivities.length; ii++) {
 44                     if (taskActivities[ii].get("taskDescriptorId") === dt.getRecord(i).get("id")) {
 45                        this.addColor(host.schedule.getCell(i, parseInt(taskActivities[ii].get("time"))));
 46                     }
 47                 }
 48             }
 49         },
 50         findTaskActivities: function() {
 51             var employees, resourDesc = Y.Wegas.Facade.VariableDescriptor.cache.find("name", "employees"),
 52                     i, ii, iii, taskIndex, work, activities, dt = this.get("host").datatable,
 53                     taskActivities = [];
 54             if (!resourDesc) {
 55                 return;
 56             } else {
 57                 employees = resourDesc.get("items");
 58             }
 59 
 60             for (i = 0; i < employees.length; i++) {
 61                 work = employees[i].get("items");
 62                 for (ii = 0; ii < work.length; ii++) {
 63                     activities = work[ii].getInstance().get("activities");
 64                     for (iii = 0; iii < activities.length; iii++) {
 65                         for (taskIndex = 0; taskIndex < dt.data.size(); taskIndex++) {
 66                             if (dt.data.item(taskIndex).get("id") === activities[iii].get("taskDescriptorId")) {
 67                                 taskActivities.push(activities[iii]);
 68                             }
 69                         }
 70                     }
 71                 }
 72 
 73             }
 74             return taskActivities;
 75         },
 76         taskActivitiesToAdd: function() {
 77             var taskActivities = this.findTaskActivities(), activitiesToAdd = [],
 78                     i, ii, exist;
 79             if (!taskActivities) {
 80                 this.get("host").showMessage("error", "No employees list found");
 81                 return;
 82             }
 83             for (i = 0; i < taskActivities.length; i++) {
 84                 exist = false;
 85                 if (activitiesToAdd.length === 0) {
 86                     activitiesToAdd.push(taskActivities[i]);
 87                 } else {
 88                     for (ii = 0; ii < activitiesToAdd.length; ii++) {
 89                         if (activitiesToAdd[ii].get("taskDescriptorId") === taskActivities[i].get("taskDescriptorId") &&
 90                                 activitiesToAdd[ii].get("time") === taskActivities[i].get("time")) {
 91                             exist = true;
 92                         }
 93                     }
 94                     if (!exist) {
 95                         activitiesToAdd.push(taskActivities[i]);
 96                     }
 97                 }
 98             }
 99             return activitiesToAdd;
100         },
101         addColor: function(cell) {
102             cell.append("<span class='editable activity'></span>");
103         }
104     }, {
105         NS: "plannificationactivitycolor",
106         NAME: "PlannificationActivityColor"
107     });
108     Y.namespace("Plugin").PlannificationActivityColor = PlannificationActivityColor;
109 });
