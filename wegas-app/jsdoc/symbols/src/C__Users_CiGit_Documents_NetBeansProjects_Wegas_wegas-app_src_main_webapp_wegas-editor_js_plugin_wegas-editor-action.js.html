130     /**
131      *  @name Y.Plugin.OpenGameAction
132      *  @extends Y.Plugin.OpenUrlAction
133      *  @class Open a game in the editor
134      *  @constructor
135      */
136     var OpenGameAction = function() {
137         OpenGameAction.superclass.constructor.apply(this, arguments);
138     };
139     Y.extend(OpenGameAction, Plugin.OpenUrlAction, {
140         /** @lends Y.Plugin.OpenGameAction# */
141 
142         /**
143          * @function
144          * @private
145          */
146         execute: function() {
147             var params,
148                     entity = this.get("entity"),
149                     testPlayer = function(game) {
150                 var teams = game.get("teams"), i, ret = false;
151                 for (i = 0; i < teams.length; i += 1) {
152                     ret = teams[i].get("players").length > 0;
153                     if (ret) {
154                         break;
155                     }
156                 }
157                 return ret;
158             };
159 
160             if (entity instanceof Wegas.persistence.GameModel) {
161                 params = "gameModelId=" + entity.get("id");
162             } else if (entity instanceof Wegas.persistence.Player) {
163                 params = "id=" + entity.get("id");
164             } else if (entity instanceof Wegas.persistence.Team) {
165                 if (entity.get("players").length < 1) {
166                     alert("Team " + entity.get("name") + " has no player");
167                     return;
168                 }
169                 params = "teamId=" + entity.get("id");
170             } else {
171                 if (entity.get("teams").length < 1) {
172                     alert("Game " + entity.get("name") + " has no Team");
173                     return;
174                 }
175                 else if (!testPlayer(entity)) {
176                     alert("Game " + entity.get("name") + " has no player");
177                     return;
178                 }
179                 params = "gameId=" + entity.get("id");
180             }
181             this.set("url", this.get("editorUrl") + params);
182             OpenGameAction.superclass.execute.call(this);
183         }
184     }, {
185         /** @lends Y.Wegas.OpenGameAction */
186 
187         NS: "wegas",
188         NAME: "OpenGameAction",
189 
190         /**
191          * <p><strong>Attributes</strong></p>
192          * <ul>
193          *    <li>editorUrl: url of the editor page<i>default: wegas-app/view/editor.html?</i></li>
194          *    <li>entity: the team, game, gamemodel or player entity that will be opened</li>
195          * </ul>
196          *
197          * @field
198          * @static
199          */
200         ATTRS: {
201             editorUrl: {
202                 value: 'wegas-app/view/editor.html?'
203             },
204             entity: {}
205         }
206     });
207     Plugin.OpenGameAction = OpenGameAction;

215     var LoadTreeviewNodeAction = function() {
216         LoadTreeviewNodeAction.superclass.constructor.apply(this, arguments);
217     };
218     Y.extend(LoadTreeviewNodeAction, Action, {
219         /** @lends Y.Plugin.LoadTreeviewNodeAction# */
220 
221         /**
222          * @function
223          * @private
224          */
225         execute: function() {
226             var entity = this.get("entity"),
227                     tabId = this.get("tabId") || this.get("host").get("label"),
228                     tabCfg = {
229                 label: entity.get("name") || "Unnamed"
230             },
231             tab = Wegas.TabView.createTab(tabId, this.get("tabSelector"), tabCfg);
232             tab.set("visible", true);
233             tab.set("selected", 2);
234             tab.witem(0).set("emptyMessage", "This game model has no games.");
235             tab.witem(0).toolbar.item(0).set("disabled", false);  // Allow game creation
236 
237             Wegas.GameFacade.set("source", // Change the source attribute on the datasource
238                     Wegas.app.get("base") + "rest/GameModel/" + entity.get("id") + "/Game");
239 
240             Wegas.GameFacade.sendRequest({
241                 request: "/"
242             });
243         }
244     }, {
245         /** @lends Y.Plugin.LoadTreeviewNodeAction */
246 
247         NS: "LoadTreeviewNodeAction",
248         NAME: "LoadTreeviewNodeAction",
249 
250         /**
251          * <p><strong>Attributes</strong></p>
252          * <ul>
253          *    <li>tabId: the id of the Y.Wegas.Tab widget that will bo opened.</li>
254          *    <li>tabSelector: the id of a Y.Wegas.TabView where a new tab will be created if none
255          *    was found <i>default: #centerTabView</i></li>
256          *    <li>entity: the entity, which the link will point to.</li>
257          * </ul>
258          *
259          * @field
260          * @static
261          */
262         ATTRS: {
263             tabId: {},
264             tabSelector: {
265                 value: '#centerTabView'
266             },
267             entity: {}
268         }
269     });

291     /**
292      * Class for display the player link in menu's
293      *
294      * @name Y.Wegas.Linkwidget
295      * @extends Y.Widget
296      * @class  Allows to display the player link in a menu.
297      * the link is in a textField. For this field inputEx is used.
298      * @constructor
299      * @param Object Will be used to fill attributes field
300      */
301     Linkwidget = Y.Base.create("wegas-playerlink-buttons", Y.Widget, [Y.WidgetChild, Wegas.Widget], {
302         /** @lends Y.Wegas.Linkwidget# */
303 
304         /**
305          * 1) Add a <div class="playerlink-label"><p>Player link</p><div> node fordisplay a label in the menu
306          * 2) Add the inputeExStringField
307          * 3) Stop the click event on this contentbox
308          * @function
309          * @private
310          */
311         renderUI: function() {
312             Linkwidget.superclass.renderUI.apply(this);
313             var cb = this.get(CONTENTBOX);
314 
315             cb.append('<div class="playerlink-label"><p>Share link</p><div>');
316 
317             this.textField = new Y.inputEx.StringField({
318                 parentEl: cb
319             });
320             cb.on("click", function(e) {
321                 e.halt(true);
322                 this.textField.el.select();
323             }, this);
324         },
325 
326         /**
327          * Add the new url
328          * @function
329          * @private
330          */
331         syncUI: function() {
332             var url = Wegas.app.get("base") + "game.html?token=" + this.get("entity").get("token");
333             this.textField.setValue(url);
334         }
335     }, {
336         /** @lends Y.Wegas.Linkwidget */
337 
338         /**
339          * <p><strong>Attributes</strong></p>
340          * <ul>
341          *    <li>entity: the gamemodel or team entity, which the link will point to.</li>
342          * </ul>
343          *
344          * @field
345          * @static
346          */
347         ATTRS: {
348             entity: {}
349         }
350     });
