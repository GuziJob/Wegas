 12 YUI.add('wegas-pmg-tablepopup', function(Y) {
 13     "use strict";
 14 
 15     /**
 16      *  @class add a popup on a table column
 17      *  @name Y.Plugin.Tablepopup
 18      *  @extends Y.Plugin.Base
 19      *  @constructor
 20      */
 21     var Wegas = Y.Wegas,
 22             Tablepopup = Y.Base.create("wegas-pmg-tablepopup", Y.Plugin.Base, [Wegas.Plugin, Wegas.Editable], {
 23         /** @lends Y.Plugin.Tablepopup */
 24         menuDetails: null,
 25         handlers: null,
 26         /**
 27          * Lifecycle methods
 28          * @function
 29          * @private
 30          */
 31         initializer: function() {
 32             this.handlers = [];
 33             this.menuDetails = new Y.Wegas.Menu({
 34                 width: this.get("width"),
 35                 points: ["tl", "tr"]
 36             });
 37             this.onceAfterHostEvent("render", this.bind);
 38             this.afterHostMethod("syncUI", this.bind);
 39             this.get("host").datatable.after("sort", this.bind, this);
 40         },
 41         bind: function() {
 42             Y.log("bind()", "log", "Wegas.Tablepopup");
 43             var i, record, dt = this.get("host").datatable, field, descriptor;
 44 
 45             for (i = 0; i < dt.get("data").size(); i++) {
 46                 dt.getCell([i, this.get("column")]).addClass("popup");
 47                 this.handlers.push(dt.getCell([i, this.get("column")]).on('click', function(e) {
 48                     dt = this.get("host").datatable;
 49                     field = this.get("field");
 50                     record = dt.getRecord(e.target);
 51                     descriptor = Y.Wegas.Facade.VariableDescriptor.cache.find("id", record.get("id"));
 52 
 53                     this.menuDetails.attachTo(e.target);
 54 
 55                     if (descriptor.get(field)) {
 56                         this.display(descriptor.get(field));
 57                     } else {
 58                         this.request(descriptor);
 59                     }
 60                 }, this));
 61 
 62                 this.handlers.push(dt.getCell([i, 1]).on('mouseout', function(e) {
 63                     this.menuDetails.hide();
 64                 }, this));
 65             }
 66         },
 67         request: function(descriptor) {
 68             Y.Wegas.Facade.VariableDescriptor.cache.getWithView(descriptor, "Extended", {// Retrieve the object from the server in Export view
 69                 on: Y.Wegas.superbind({
 70                     success: function(e) {
 71                         var field = this.get("field");
 72                         if (e.response.entity.get(field)) {
 73                             descriptor.set(field, e.response.entity.get(field));
 74                             this.display(e.response.entity.get(field));
 75                         } else {
 76                             this.get("host").showMessage("error", "This information does not exist");
 77                         }
 78 
 79                     },
 80                     failure: function(e) {
 81                         this.get("host").showMessage("error", "An error occurs");
 82                     }
 83                 }, this)
 84             });
 85         },
 86         display: function(value) {
 87             this.menuDetails.show();
 88             this.menuDetails.get("contentBox").setHTML('<div style="padding:5px 10px"><p>' + value + '</p></div>');
 89         },
 90         /**
 91          * Destructor methods.
 92          * @function
 93          * @private
 94          */
 95         destructor: function() {
 96             var i;
 97             for (i = 0; i < this.handlers.length; i++) {
 98                 this.handlers[i].detach();
 99             }
100         }
101     }, {
102         ATTRS: {
103             width: {
104                 type: "string",
105                 value: "250px"
106             },
107             field: {
108                 type: "string",
109                 value: "description"
110             },
111             column: {
112                 value: 1,
113                 _inputex: {
114                     _type: "integer",
115                     label: "popup on column"
116                 }
117             }
118         },
119         NS: "tablepopup",
120         NAME: "Tablepopup"
121     });
122     Y.namespace("Plugin").Tablepopup = Tablepopup;
123 });
