 13     "use strict";

 15     /**
 16      * @name Y.Wegas.JoinTeam
 17      * @extends Y.Widget
 18      * @augments Y.WidgetChild
 19      * @augments Y.Wegas.Widget
 20      * @class class for join a team
 21      * @constructor
 22      * @description Allows just to join a team
 23      */
 24     var CONTENTBOX = "contentBox", DATASOURCE = "dataSource", NAME = "name",
 25             RENDER = "render", HOST = "host", Wegas = Y.Wegas,
 26             GameDataTable;
 27 
 28     GameDataTable = Y.Base.create("wegas-lobby-datatable", Y.Widget, [Y.WidgetChild, Wegas.Widget, Wegas.Editable], {
 29         // *** Private fields *** //
 30 
 31         /**
 32          * @function
 33          * @private
 34          * @description All button and fields are created.
 35          * For creating the field inputEx libary is used
 36          */
 37         renderUI: function() {
 38             var cfg = this.get("dataTableCfg");
 39 
 40             cfg.columns = Y.Array.map(cfg.columns, function(c) {                // Add some default properties to columns
 41                 return Y.mix(c, {
 42                     allowHTML: true,
 43                     sortable: true
 44                 });
 45             });
 46             cfg = Y.mix(cfg, {//                                                // Add cfg default values
 47                 //data: data,
 48                 width: "100%"
 49             });
 50             this.dataTable = new Y.DataTable(cfg);                              // Render datatable
 51             this.dataTable.render(this.get(CONTENTBOX));
 52             this.dataTable.set('strings.emptyMessage', "<em><center><br /><br />" + this.get("emptyMessage") + "<br /><br /><br /></center></em>");
 53             this.plug(Y.Plugin.EditorDTMenu);
 54 
 55             this.get(CONTENTBOX).addClass("yui3-skin-wegas");
 56         },
 57         /**
 58          * @function
 59          * @private
 60          */
 61         bindUI: function() {
 62             var ds = this.get(DATASOURCE),
 63                     request = this.get("request");
 64             if (ds) {
 65                 this.updateHandler = ds.after("update", this.syncUI, this);                          // Listen updates on the target datasource
 66                 this.failureHandler = ds.after("failure", this.defaultFailureHandler, this);          // GLOBAL error message
 67 
 68                 if (request) {
 69                     ds.sendRequest(request);
 70                 }
 71             }
 72         },
 73         /**
 74          * @function
 75          * @private
 76          */
 77         syncUI: function() {
 78             Y.log("sync()", "info", "Wegas.LobbyDataTable");
 79 
 80             if (!this.get(DATASOURCE)) {
 81                 this.get(CONTENTBOX).append("Unable to find datasource");
 82                 return;
 83             }
 84 
 85             var ds = this.get(DATASOURCE),
 86                     data = this.genData(ds.cache.findAll());
 87 
 88             this.dataTable.set("data", data);
 89             this.hideOverlay();
 90         },
 91         destructor: function() {
 92             this.dataTable.destroy();
 93             this.updateHandler.detach();
 94             this.failureHandler.detach();
 95         },
 96         // *** Private Methods *** //
 97         /**
 98          * @function
 99          * @private
100          */
101         genData: function(elements) {
102             var ret = [], i, entity, data;
103 
104             for (i in elements) {
105                 if (elements.hasOwnProperty(i)) {
106                     entity = elements[i];
107                     data = this.genEntityData(entity);
108                     if (data) {
109                         Y.mix(data, {//                                         // Add some default properties
110                             entity: entity,
111                             iconCSS: 'wegas-icon-' + entity.get("@class").toLowerCase()
112                         });
113                         ret.push(data);
114                     }
115                 }
116             }
117             return ret;
118         },
119         genEntityData: function(entity) {
120             switch (entity.get("@class")) {
121 
122                 case 'DebugGame':
123                     break;
124 
125                 case 'Game':
126                     var createdBy = entity.get("createdBy"),
127                             gameModel = Wegas.Facade.GameModel.cache.findById(entity.get("gameModelId"))
128 
129                     if (!gameModel) {
130                         Y.log("Unable to find game model for game: " + entity.get(NAME) + "(" + entity.get("gameModelId") + ")", "error");
131                     }
132                     return {
133                         name: entity.get(NAME),
134                         createdTime: entity.get("createdTime"),
135                         token: gameModel && gameModel.get("properties.freeForAll") ? entity.get("token") : "",
136                         gameModelName: gameModel ? gameModel.get(NAME) : "",
137                         createdBy: createdBy ? createdBy.get(NAME) : "undefined",
138                         teamsCount: gameModel && gameModel.get("properties.freeForAll") ? -1 : entity.get("teams").length,
139                         playersCount: entity.get("playersCount")
140                     };
141                     break;
142 
143                 case 'Team':
144                     return {
145                         name: entity.get(NAME),
146                         token: entity.get("token")
147                     };
148 
149                 case 'Player':
150                     return {
151                         name: entity.get(NAME)
152                     };
153 
154                 case 'GameModel':
155                     if (entity.get("canEdit")) {
156                         return {
157                             name: entity.get(NAME)
158                         };
159                     }
160                     break;
161 
162                 case 'User':
163                     return {
164                         label: entity.get(NAME),
165                         entity: entity.getMainAccount()
166                     };
167 
168                 case 'Role':
169                     return {
170                         label: entity.get(NAME)
171                     };
172 
173                 default:
174                     Y.log("Unable to generate data for entity " + entity.get("@class") + '(' + entity.get(NAME) + ")");
175                     return null;
176             }
177         }
178     }, {
179         ATTRS: {
180             dataSource: {
181                 getter: function(val) {
182                     if (Y.Lang.isString(val)) {
183                         return Wegas.Facade[val];
184                     }
185                     return val;
186                 }
187             }, emptyMessage: {
188                 value: "You have not joined any game yet"
189             },
190             dataTableCfg: {
191                 value: {
192                     columns: [{
193                             key: "iconCSS",
194                             formatter: "icon",
195                             label: " ",
196                             sortable: false,
197                             width: "22px"
198                         }, {
199                             key: "name",
200                             label: "Name"
201                         }, {
202                             key: "createdBy",
203                             label: "Created by",
204                             width: "150px"
205                         }, {
206                             key: "createdTime",
207                             label: "Created",
208                             width: "140px",
209                             formatter: "date"
210                         }, {
211                             key: "playersCount",
212                             label: "Players",
213                             width: "80px",
214                             formatter: "count"
215                         }, {
216                             key: "teamsCount",
217                             label: "Teams",
218                             width: "80px",
219                             formatter: "count"
220                         }, {
221                             key: "gameModelName",
222                             label: "Game model",
223                             width: "150px"
224                         }
225 //                        , {
226 //                            key: "menu",
227 //                            label: " ",
228 //                            sortable: false,
229 //                            formatter: "menu",
230 //                            allowHTML: true,
231 //                            emptyCellValue: "<em>(not set)</em>"
232 //  formatter: function(o){
233 //                    // just retrieve the selected Master record and return the
234 //                    // "aname" column
235 //                    var parent_rec = dt_master.getRecord(
236 //                        dt_master.get('selectedRow') );
237 //
238 //                    return parent_rec.get('aname');
239 //                }
240 //                        }
241                     ]
242                 }
243             }
244         }
245     });
246     Y.namespace('Wegas').GameDataTable = GameDataTable;
247     /**
248      * @class Open a menu on click, containing the admin edition field
249      * @constructor
250      */
251     Y.Plugin.EditorDTMenu = Y.Base.create("admin-menu", Y.Plugin.Base, [], {
252         initializer: function() {
253 
254             this.buttons = [];
255 
256             this.afterHostEvent(RENDER, function() {
257                 var host = this.get(HOST);
258                 host.dataTable.addAttr("selectedRow", {value: null});
259 
260                 host.dataTable.delegate('click', function(e) {
261                     this.set('selectedRow', e.currentTarget);
262                 }, '.yui3-datatable-data tr[data-yui3-record]', host.dataTable);
263 
264                 host.dataTable.after('selectedRowChange', this.onClick, this);
265             });
266         },
267         onClick: function(e) {
268             var host = this.get(HOST),
269                     tr = e.newVal, // the Node for the TR clicked ...
270                     last_tr = e.prevVal, //  "   "   "   the last TR clicked ...
271                     rec = host.dataTable.getRecord(tr), // the current Record for the clicked TR
272                     menuItems = this.get("children"),
273                     entity = rec.get("entity"),
274                     data = {
275                 entity: entity,
276                 dataSource: host.get(DATASOURCE)
277             };
278 
279             Y.Plugin.EditorDTMenu.currentGameModel = entity;                  // @hack so game model creation will work
280 
281             if (last_tr) {
282                 last_tr.removeClass("wegas-datatable-selected");
283             }
284             tr.addClass("wegas-datatable-selected");
285             if (entity) {
286                 if (menuItems) {                                                // If there are menu items in the cfg
287                     Wegas.Editable.mixMenuCfg(menuItems, data);                 // use them
288                 } else {                                                        // Otherwise
289                     menuItems = entity.getMenuCfg(data);                        // use entity default menu
290                 }
291 
292                 Y.Array.each(menuItems, function(i) {                           // @hack add icons to some buttons
293                     switch (i.label) {
294                         case "Delete":
295                         case "Duplicate":
296                         case "Open in editor":
297                             i.label = '<span class="wegas-icon wegas-icon-' + i.label.replace(/ /g, "-").toLowerCase() + '"></span>' + i.label;
298                     }
299                 });
300                 if (menuItems.length === 0) {
301                     return;
302                 }
303                 Y.Array.each(this.buttons, function(b) {                        // Remove existing buttons,
304                     b.destroy();
305                 });
306                 //this.buttons = Y.Array.map(menuItems, function(i) {           // Add new buttons to the toolbar
307                 //    return host.toolbar.add(i);
308                 //});
309                 if (this.get("autoClick")) {                                    // If we are in autoclick,
310 
311                     Y.once("rightTabShown", function() {
312                         var target = Y.Widget.getByNode("#rightTabView").item(0).witem(0);
313                         if (target && target.toolbar) {
314                             this.buttons = Y.Array.map(menuItems, function(i) { // Add new buttons to the right tab's toolbar
315                                 return target.toolbar.add(i);
316                             });
317                             this.buttons[0].set("visible", false);
318                         }
319                         //.fire("click");                                       // launch first button actionF
320                     }, this);
321 
322                     var button = Wegas.Widget.create(menuItems[0]);
323                     button.render().fire("click");
324                     button.destroy();
325                 }
326             } else {
327                 Y.log("Menu item has no target entity", "info", "Y.Plugin.EditorTVAdminMenu");
328                 host.currentSelection = null;
329             }
330         }
331     }, {
332         NS: "EditorDTMenu",
333         NAME: "EditorDTMenu",
334         ATTRS: {
335             children: {},
336             autoClick: {
337                 value: true
338             }
339         }
340     });

342     Y.DataTable.BodyView.Formatters.icon = function(col) {
343         col.className = 'wegas-lobby-datatable-icon';
344         col.sortable = false;
345         return function(o) {
346             return '<span class="wegas-icon ' + o.value + '"></span>';
347         };
348     };
349     Y.DataTable.BodyView.Formatters.menu = function(col) {
350         col.className = 'wegas-datatable-menu';
351         col.label = " ";
352         col.sortable = false;
353         return function(o) {
354             return '<span class="wegas-icon-menu"></span>';
355         };
356     };
357     Y.DataTable.BodyView.Formatters.date = function() {
358         return function(o) {
359             return Wegas.Helper.smartDate(o.value);
360         };
361     };
362     Y.DataTable.BodyView.Formatters.count = function(col) {
363         col.className = "wegas-datatable-center";
364         return function(o) {
365             return (o.value === 0) ? o.value : (o.value === -1) ? "" : o.value;
366         };
367     };
