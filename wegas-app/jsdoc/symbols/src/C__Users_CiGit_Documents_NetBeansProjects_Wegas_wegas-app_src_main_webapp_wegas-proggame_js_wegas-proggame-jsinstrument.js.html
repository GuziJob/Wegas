 11 YUI.add('wegas-proggame-jsinstrument', function(Y) {
 12     "use strict";
 13 
 14     /**
 15      * Level display, should handle canvas, for now renders the level as a
 16      * table element.
 17      *
 18      */
 19     var JSInstrument = Y.Base.create("wegas-proggame-display", Y.Base, [], {
 20         instrument: function(code) {
 21             var tree, getDebugStatement = function(line) {
 22                 return {
 23                     type: "ExpressionStatement",
 24                     expression: {
 25                         type: "CallExpression",
 26                         callee: {
 27                             type: "Identifier",
 28                             name: "_____debug"
 29                         },
 30                         "arguments": [{
 31                                 type: "Literal",
 32                                 value: line - 1
 33                             }, {
 34                                 type: "ThisExpression"
 35                             }, {
 36                                 type: "Identifier",
 37                                 name: "___DEBUGBLOCK___"
 38                             }]
 39                     }
 40                 };
 41             }, getWatchStatement = function(s) {
 42                 return {
 43                     type: "ExpressionStatement",
 44                     expression: {
 45                         type: "CallExpression",
 46                         callee: {
 47                             type: "Identifier",
 48                             name: "_____watch"
 49                         },
 50                         "arguments": Y.Array.map(s.declarations, function(s) {
 51                             return {
 52                                 type: "Literal",
 53                                 value: s.id.name
 54                             };
 55                         })
 56                     }
 57                 };
 58             };
 59             try {
 60                 tree = window.esprima.parse(code, {//                           // Generate the syntaxic tree using esprima
 61                     //raw: true,
 62                     //range: true,
 63                     loc: true
 64                 });
 65 
 66                 //console.log("Wegas.JSInstrument: Parsed source tree:", tree);
 67                 this.traverse(tree, function(object, path) {
 68                     var add = [], inter;
 69                     switch (object['type']) {
 70                         case "BlockStatement":
 71                         case "Program":
 72                             //for (var i = 0; i < object.body.length; i += 1) {
 73                             Y.Array.each(object.body, function(s) {
 74                                 inter = [getDebugStatement(s.loc.start.line)];
 75                                 switch (s.type) {
 76                                     case "VariableDeclaration":
 77                                         //inter.push(getWatchStatement(s));     // Automatically add declared variables to the list of watches (may be useful if we remove them at the end of execution)
 78                                         break;
 79                                 }
 80                                 add.push(inter);
 81                             });
 82                             object.body = this.zip(object.body, add);
 83                             //object.body.push(getDebugStatement(object.loc.end.line));
 84                             break;
 85                     }
 86                 }, true);
 87 
 88                 var instrumentedCode = window.escodegen.generate(tree, {
 89                     indent: true
 90                 });
 91                 instrumentedCode = instrumentedCode.replace(/___DEBUGBLOCK___/g, "(function(){ var i, w = watches, ret = {};"
 92                         + "for(i=0;i<w.length;i++){"
 93                         + "try {"
 94                         + "ret[w[i]]=eval(w[i]);"
 95                         //ret[this.watches[i]] = this.doEval(this.watches[i]);
 96                         + "} catch(e){}"
 97                         + "}"
 98                         + "return ret;})()");                                   // This function returns any watched variable int he scope, so it can be used in the breakpoint event
 99 
100                 return instrumentedCode;
101             } catch (e) {
102                 Y.log("Error evaluating client code", "error");
103                 return;
104             }
105         },
106         /**
107          * Zip any item of lists in b and objects in a (very specific)
108          *
109          * @param {Array} a
110          * @param {Array} b
111          * @returns {Array}
112          */
113         zip: function(a, b) {
114             var results = [];
115             Y.Array.each(a, function(item, index) {
116                 results = results.concat(b[index]);
117                 results.push(item);
118             });
119             return results;
120         },
121         /**
122          *
123          * @param {Array} object
124          * @param {function} visitor
125          * @param {type} path
126          * @returns
127          */
128         traverse: function(object, visitor, path) {
129             var key, child;
130 
131             if (typeof path === 'undefined') {
132                 path = [];
133             }
134             for (key in object) {
135                 if (object.hasOwnProperty(key)) {
136                     child = object[key];
137                     if (typeof child === 'object' && child !== null) {
138                         this.traverse(child, visitor, [object].concat(path));
139                     }
140                 }
141             }
142             visitor.call(this, object, path);
143 
144         }
145     });
146     Y.namespace("Wegas").JSInstrument = JSInstrument;
147 
148 });
