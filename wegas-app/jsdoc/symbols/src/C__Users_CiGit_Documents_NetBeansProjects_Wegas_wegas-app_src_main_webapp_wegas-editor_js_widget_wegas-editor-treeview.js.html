 35         // ** Lifecycle methods ** //
 36         /**
 37          *
 38          */
 39         renderUI: function () {
 40             this.treeView = new Y.TreeView();
 41             this.treeView.render(this.get(CONTENTBOX));
 42 
 43             this.plug(Y.Plugin.EditorTVAdminMenu);
 44             this.plug(Y.Plugin.RememberExpandedTreeView);

 46 
 47         /**
 48          * @function
 49          * @private
 50          */
 51         bindUI: function () {
 52             var ds = this.get(DATASOURCE),
 53             request = this.get("request");
 54             if (ds) {
 55                 ds.after("response", this.syncUI, this);                        // Listen updates on the target datasource
 56                 ds.after("error", function (e) {                                // GLOBAL error message
 57                     this.showMessage("error", e.response.results.message);
 58                 }, this);
 59 
 60                 if (request) {
 61                     ds.rest.sendRequest(request);
 62                 }
 63             }
 64         },
 65 
 66         /**
 67          * @function
 68          * @private
 69          */
 70         syncUI: function () {
 71             if (!this.get(DATASOURCE)) {
 72                 this.get(CONTENTBOX).append("Unable to find datasource");
 73                 return;
 74             }
 75 
 76             var ds = this.get(DATASOURCE),
 77             selector = this.get("dataSelector"),
 78             entities = (selector) ? ds.rest.find(selector.key, selector.val) : ds.rest.getCache(),
 79             msg = this.get(CONTENTBOX).one(".wegas-smallmessage");
 80 
 81             if (msg) {
 82                 msg.remove(true);
 83             }
 84             this.treeView.removeAll();
 85             if (entities.length === 0) {
 86                 this.get(CONTENTBOX).append('<div class="wegas-smallmessage">' + this.get("emptyMessage") + '</div>');
 87                 return;
 88             }
 89             this.treeView.add(this.genTreeViewElements(entities));
 90             this.treeView.syncUI();
 91         },

329         /**
330          * @function
331          * @private
332          */
333         genScopeTreeViewElements: function (el) {
334             var children = [], i, label, team, player, instance,
335             instances = el.get("scope").get("variableInstances");
336 
337             for (i in instances) {
338                 if (instances.hasOwnProperty(i)) {
339                     instance = instances[i];
340                     label = '';
341                     switch (el.get("scope").get(CLASS)) {
342                         case 'PlayerScope':
343                             player = Wegas.GameFacade.rest.getPlayerById(i);
344 
345                             if (!player) continue;
346 
347                             label = (player) ? player.get(NAME) : "undefined";
348                             break;
349                         case 'TeamScope':
350                             team = Wegas.GameFacade.rest.getTeamById(i);
351 
352                             if (!team) continue;
353 
354                             label = (team) ? team.get(NAME) : "undefined";
355                             break;
356                         case 'GameScope':
357                         case 'GameModelScope':
358                             label = 'Global';
359                             break;
360                     }
361                     children.push(this.genVariableInstanceElements(label, instance));
362                 }
363             }
364             return children;
365         },

367         /**
368          * @function
369          * @private
370          */
371         genVariableInstanceElements: function (label, el) {
372             var l,
373             selected = (this.currentSelection == el.get(ID)) ? 2 : 0;
374 
375             switch (el.get(CLASS)) {
376                 case 'StringInstance':
377                 case 'NumberInstance':
378                 case 'ListInstance':
379                     return {
380                         label: label + ': ' + el.get("value"),
381                         selected: selected,
382                         data: {
383                             entity: el
384                         }
385                     };
386 
387                 case 'QuestionInstance':
388                     l = label + ((el.get("replies").length > 0) ? ': ' + el.get("replies").get(NAME) : ': unanswered');
389                     return {
390                         label: l,
391                         selected: selected,
392                         data: {
393                             entity: el
394                         }
395                     };
396 
397                 case 'InboxInstance':
398                     var k, children = [], collapsed = !this.isNodeExpanded(el);
399 
400                     label += "(" + el.get("messages").length + ")";
401 
402                     for (k = 0; k < el.get("messages").length; k += 1) {
403                         children.push({
404                             label: el.get("messages")[k].get("subject")
405                         });
406                     }
407                     return {
408                         type: 'TreeNode',
409                         label: label,
410                         selected: selected,
411                         collapsed: collapsed,
412                         data: {
413                             entity: el
414                         },
415                         children: children
416                     };
417 
418                 default:
419                     return {
420                         label: label,
421                         selected: selected,
422                         data: {
423                             entity: el
424                         }
425                     };
426             }
427         },

429         /**
430          * @function
431          * @private
432          */
433         genPageTreeViewElements: function (elts) {
434             var ret = [], j, text, el;
435             if (!Y.Lang.isArray(elts)) {
436                 elts = [elts];
437             }
438 
439             for (j = 0; j < elts.length; j += 1) {
440                 el = elts[j];
441                 text = el.type + ': ' + (el.label || el.name || el.id || 'unnamed');
442                 switch (el.type) {
443                     case 'List':
444                         ret.push({
445                             type: 'TreeNode',
446                             label: 'List: ' + (el.label || 'unnamed'),
447                             data: {
448                                 entity: el
449                             },
450                             children: this.genPageTreeViewElements(el.children)
451                         });
452                         break;
453                     case 'VariableDisplay':
454                         text = 'Variable displayer: ' + (el.variable);
455                         ret.push({
456                             label: text,
457                             data: {
458                                 entity: el
459                             }
460                         });
461                         break;
462                     case 'Text':
463                         ret.push({
464                             label: 'Text: ' + el.content.substring(0, 15) + "...",
465                             data: {
466                                 entity: el
467                             }
468                         });
469                         break;
470                     case 'Button':
471                         ret.push({
472                             label: text,
473                             data: {
474                                 entity: el
475                             }
476                         });
477                         break;
478                     default:
479                         ret.push({
480                             label: text,
481                             data: {
482                                 entity: el
483                             }
484                         });
485                         break;
486 
487                 }
488             }
489             return ret;
490         },

682     Y.Plugin.RememberExpandedTreeView = Y.Base.create("wegas-rememberexpandedtreeview", Y.Plugin.Base, [], {
683         expandedIds: null,
684         initializer: function () {
685             this.expandedIds = {};
686             this.afterHostEvent(RENDER, function () {

690                     this.expandedIds[e.node.get("data").entity.get(ID)] = true;

693                 host.treeView.before("*:nodeCollapsed", function (e) {
694                     delete this.expandedIds[e.node.get("data").entity.get(ID)];
695                 }, this);
696             });
697         }

759     Y.Plugin.EditorTVAdminMenu = Y.Base.create("admin-menu", Y.Plugin.Base, [], {
760         initializer: function () {
761             this.menu = new Wegas.Menu();
762 
763             this.afterHostEvent(RENDER, function () {
764                 this.get(HOST).treeView.on("*:click", this.onTreeViewClick, this);
765             });
766         },
767         onTreeViewClick: function (e) {
768             //Y.log(e.target.get("label") + " label was clicked", "info", "Wegas.EditorTreeView");
769 
770             var menuItems, data = e.node.get("data"),
771             domTarget = e.domEvent.target,
772             host = this.get(HOST);
773 
774             if (data) {
775                 host.currentSelection = data.entity.get(ID);
776                 data.dataSource = host.get(DATASOURCE);
777                 menuItems = data.entity.getMenuCfg(data);
778 
779                 if (menuItems.length === 0) {
780                     return;
781                 }
782 
783                 this.menu.removeAll();                                  // Populate the menu with the elements associated to the
784                 this.menu.add(menuItems);
785 
786                 if (domTarget.hasClass("wegas-treeview-editmenubutton")) {          // If user clicked on the edit button
787                     this.menu.attachTo(domTarget);                                  // Display the edit button next to it
788                 } else {                                                            // Otherwise the user clicked on the node
789                     this.menu.item(0).fire("click");             // Excute the actions associated to the first item of the menu
790                 }
791             } else {
792                 Y.log("Menu item has no target entity", "info", "Y.Plugin.EditorTVAdminMenu");
793                 host.currentSelection = null;
794             }
795 
796         }
797     }, {
798         NS: "EditorTVMenu",
799         NAME: "EditorTVAdminMenu"
