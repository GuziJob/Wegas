 13 YUI.add("wegas-inputex-colorpicker", function(Y) {
 14 
 15     var inputEx = Y.inputEx,
 16             lang = Y.Lang;
 17 
 18     Y.namespace("inputEx.Wegas").ColorField = function(options) {
 19         inputEx.Wegas.ColorField.superclass.constructor.call(this, options);
 20     };
 21     Y.extend(inputEx.Wegas.ColorField, inputEx.Field, {
 22         setOptions: function(options) {
 23             inputEx.Wegas.ColorField.superclass.setOptions.call(this, options);
 24 
 25             // Overwrite options
 26             this.options.className = options.className ? options.className : 'inputEx-Field inputEx-ColorField inputEx-PickerField';
 27             this.options.zIndex = options.zIndex || 4;
 28         },
 29         renderComponent: function() {
 30             this.el = inputEx.cn('input', {
 31                 name: this.options.name || '',
 32                 value: this.options.value || '#FFFFFF',
 33                 size: 9
 34             });
 35 
 36             // This element wraps the input node in a float: none div
 37             this.wrapEl = inputEx.cn('div', {className: 'inputEx-PickerField-wrapper'});
 38             this.wrapEl.appendChild(this.el);
 39 
 40             // Create button
 41             this.button = Y.Node.create("<button> </button>").addClass("inputEx-ColorField-button");
 42             this.button.appendTo(this.wrapEl);
 43 
 44             this.fieldContainer.appendChild(this.wrapEl);
 45             
 46             this.button.on('click', this._toggleOverlay, this, true);
 47 //            Y.one(this.el).on('click', this._toggleOverlay, this, true);
 48             
 49             Y.one(this.el).on("change",function(event) {
 50                 this.markSelectedColor(this.el.value);
 51             }, this);
 52 
 53             // plugin colorpicker
 54             this.grid = inputEx.cn('div', {className: 'inputEx-ColorField-Grid'});
 55             var thisBis = this;
 56             Y.use('gallery-colorpickercss', 'gallery-colorpicker', function(Y) {
 57                 var swatch = Y.one(thisBis.grid);
 58                 // create an instance of the widget
 59                 thisBis.colorpicker = new Y.ColorPicker();
 60                 // render the widget into the #picker node
 61                 thisBis.colorpicker.render(thisBis.grid);
 62                 
 63                 swatch.one(".yui3-colorpicker-swatch").setStyle("border", "1px solid white");
 64                 swatch.one(".yui3-colorpicker-swatch").appendChild("<p style='text-align: center; margin-top: 9px'>Update Color</p>");
 65                 swatch.one(".yui3-colorpicker-swatch").on('click', function(e) {
 66                     thisBis.el.value = "#" + thisBis.colorpicker.get('hex');
 67                     thisBis.oOverlay.hide();
 68                     thisBis.markSelectedColor();
 69                 }, this);
 70             });
 71 
 72         },
 73         _toggleOverlay: function(e) {
 74             // PreventDefault to prevent submit in a form
 75             e.preventDefault();
 76 
 77             // palette may not have been rendered yet
 78             this.renderPalette();
 79 
 80             this.oOverlay[this.oOverlay.get('visible') ? 'hide' : 'show']();
 81         },
 82         renderPalette: function() {
 83 
 84             if (!this.oOverlay) {
 85                 this.renderOverlay();
 86             }
 87 
 88             // render once !
 89             if (this.paletteRendered) {
 90                 return;
 91             }
 92             
 93             this.colorpicker.set("hex", this.el.value.substr(1));
 94             if (this.el.value === "#ffffff"){
 95                 this.colorpicker.set("hsl", {h: 0, s: 1, l: 1});
 96             }
 97             this.colorGrid = this.grid;
 98             this.oOverlay.set('bodyContent', this.colorGrid);
 99 
100             this.paletteRendered = true;
101         },
102         renderOverlay: function() {
103 
104             // Create overlay
105             this.oOverlay = new Y.Overlay({
106                 visible: false,
107                 constrain: true,
108                 zIndex: 1000
109             });
110             //his.oOverlay.render(this.fieldContainer);
111             
112             this.oOverlay.render();
113 
114             this.oOverlay.on('visibleChange', function(e) {
115 
116                 if (e.newVal) { // show
117                     // align
118                     this.oOverlay.set("align", {node: this.button, points: [Y.WidgetPositionAlign.CC, Y.WidgetPositionAlign.TC]});
119 
120                     // Activate outside event handler
121                     this.outsideHandler = this.oOverlay.get('boundingBox').on('mousedownoutside', function(e) {
122                         this.oOverlay.hide();
123                     }, this);
124                 }
125                 else { // hide
126                     this.outsideHandler.detach();
127                 }
128 
129             }, this);
130         },
131         markSelectedColor: function(value) {
132 
133             value = value || this.getValue();
134             
135             // set background color on colorEl
136             Y.one(this.el).setStyle('backgroundColor', value);
137 
138         },
139         setValue: function(value, sendUpdatedEvt) {
140 
141             this.el.value = value;
142 
143             this.markSelectedColor(value);
144 
145             // Call Field.setValue to set class and fire updated event
146             inputEx.Wegas.ColorField.superclass.setValue.call(this, value, sendUpdatedEvt);
147         },
148         /**
149          * Return the color value
150          * @method getValue
151          * @return {String} Color value
152          */
153         getValue: function() {
154             return this.el.value;
155         }
156     });
157     inputEx.registerType("colorpicker", inputEx.Wegas.ColorField);
158 });
