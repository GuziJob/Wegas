 12 YUI.add('wegas-helper', function(Y) {
 13     "use strict";
 14     /**
 15      * @name Y.Wegas.Helper
 16      * @class
 17      * @constructor
 18      */
 19 
 20     var Helper = {
 21         /**
 22          * Generate ID an unique id based on current time.
 23          * @function
 24          * @static
 25          * @return {Number} time
 26          * @description
 27          */
 28         genId: function() {
 29             var now = new Date();
 30             return now.getHours() + now.getMinutes() + now.getSeconds();
 31         },
 32         /**
 33          * Escape a html string by replacing <, > and " by their html entities.
 34          *
 35          * @function
 36          * @static
 37          * @param {String} str
 38          * @return {String} Escaped string
 39          */
 40         htmlEntities: function(str) {
 41             return String(str).replace(/&/g, '&')
 42                     .replace(/</g, '<')
 43                     .replace(/>/g, '>')
 44                     .replace(/"/g, '"');
 45         },
 46         /**
 47          * Replace any text line return
 48          * @function
 49          * @static
 50          * @param {String} str the string to escape
 51          * @param {String} replaceBy The value to replace with, default is \<br \/\>
 52          * @return {String} Escaped string
 53          */
 54         nl2br: function(str, replaceBy) {
 55             replaceBy = replaceBy || '<br />';
 56             return (str + '').replace(/([^>\r\n]?)(\r\n|\n\r|\r|\n)/g, '$1' + replaceBy + '$2');
 57         },
 58         escapeJSString: function(str) {
 59             return str.replace(/"/g, '\\"').replace(/(\r\n|\n\r|\r|\n)/g, "\\n");
 60             //return Helper.nl2br(str.replace(/"/g, '\\"'), "\\n");
 61         },
 62         unesacapeJSString: function(str) {
 63             return str.replace(/\\"/g, '"');
 64         },
 65         /**
 66          * Format a date, using provided format string.
 67          *
 68          * @function
 69          * @static
 70          * @argument {Number} timestamp
 71          * @argument {String} fmt the format to apply, ex. '%d.%M.%Y at %H:%i:%s' <br />
 72          * d	Day of the month, 2 digits with leading zeros <br />
 73          * m	Numeric representation of a month, with leading zeros <br />
 74          * M	A short textual representation of a month, three letters <br />
 75          * Y	A full numeric representation of a year, 4 digits <br />
 76          * H	24-hour format of an hour with leading zeros <br />
 77          * i	Minutes with leading zeros <br />
 78          * s	Seconds, with leading zeros <br />
 79          * @returns {String} formated date
 80          */
 81         formatDate: function(timestamp, fmt) {
 82             var date = new Date(timestamp),
 83                     months = ["Jan", "Feb", "Mar", "Apr", "Mai", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"];
 84             function pad(value) {
 85                 return (value.toString().length < 2) ? '0' + value : value;
 86             }
 87             return fmt.replace(/%([a-zA-Z])/g, function(_, fmtCode) {
 88                 switch (fmtCode) {
 89                     case 'Y':
 90                         return date.getFullYear();
 91                     case 'M':
 92                         return months[date.getMonth()];
 93                     case 'm':
 94                         return pad(date.getMonth() + 1);
 95                     case 'd':
 96                         return pad(date.getDate());
 97                     case 'H':
 98                         return pad(date.getHours());
 99                     case 'i':
100                         return pad(date.getMinutes());
101                     case 's':
102                         return pad(date.getSeconds());
103                     default:
104                         throw new Error('Unsupported format code: ' + fmtCode);
105                 }
106             });
107         },
108         /**
109          * Returns a time lapse between provided timestamp and now, e.g. "a month ago",
110          * "2 hours ago", "10 minutes ago"
111          * @function
112          * @static
113          * @argument {Number} timestamp
114          * @return {String} The formatted time
115          */
116         smartDate: function(timestamp) {
117             var date = new Date(timestamp),
118                     now = new Date(),
119                     diffN = now.getTime() - timestamp,
120                     oneMinute = 60 * 1000,
121                     oneHour = 60 * oneMinute,
122                     oneDay = 24 * oneHour;
123             // oneMonth =  30 * oneDay,
124             // oneYear =  365 * oneDay;
125 
126             if (!date.getTime()) {
127                 return "undefined";
128             }
129 
130             if (diffN < oneMinute) {                                           // last minute
131                 return Math.round(diffN / 1000) + " seconds ago";
132             } else if (diffN < oneHour) {                                      // last hour
133                 return  Math.round(diffN / oneMinute) + " minutes ago";
134             } else if (diffN < oneDay
135                     && now.getDay() === date.getDay()) {                            // Today
136                 return Helper.formatDate(timestamp, "%H:%i");
137             } else if (date.getYear() === now.getYear()) {                      // This year
138                 return Helper.formatDate(timestamp, "%d %M");
139             } else {                                                             // Older
140                 return Helper.formatDate(timestamp, "%d %M %Y");
141             }
142 
143         },
144         /**
145          * Java hashCode implementation
146          * @param {String} value to hash
147          * @returns {Number}
148          */
149         hashCode: function(value) {
150             return Y.Array.reduce(value.split(""), 0, function(prev, curr) {
151                 prev = ((prev << 5) - prev) + curr.charCodeAt(0);
152                 return prev |= 0;                                               //Force 32 bits
153             });
154         }
155     };
156     Y.namespace("Wegas").Helper = Helper;
157 });
