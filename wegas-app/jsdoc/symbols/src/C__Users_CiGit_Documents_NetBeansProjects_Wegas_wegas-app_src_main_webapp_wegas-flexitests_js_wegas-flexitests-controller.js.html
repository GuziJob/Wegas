 13 YUI.add("wegas-flexitests-controller", function(Y) {
 14     "use strict";
 15 
 16     Y.Wegas.FlexitestsController = Y.Base.create("wegas-flexitests-controller", Y.Wegas.AbsoluteLayout, [], {
 17         /**
 18          * Lifecycle method
 19          * @function
 20          * @private
 21          * @returns {undefined}
 22          */
 23         initializer: function() {
 24             this.leftElement = null;
 25             this.rightElement = null;
 26             this.centerElement = null;
 27             this.mcq = null;
 28             this.maxSize = 0;
 29             this.ongoing = false;
 30             this.currentQuestionId = -1;
 31             this.questionToDo = [];
 32             this.startTime = null;
 33             this.events = [];
 34         },
 35         /**
 36          * Lifecycle method
 37          * @function
 38          * @private
 39          * @returns {undefined}
 40          */
 41         bindUI: function() {
 42             this.set("tabIndex", -1);
 43             this.after("*:currentLoadingChange", function(e) {
 44                 var noready = false;
 45                 for (var i in e.newVal) {
 46                     noready = (noready || e.newVal[i]);
 47                     if (noready) {
 48                         break;
 49                     }
 50                 }
 51                 if (!noready) {
 52                     this.startStimuli();
 53                 }
 54             });
 55             this.on("*:clientResponse", function(e) {
 56                 if (this.ongoing) {
 57                     this.responseGiven(e.value);
 58                 }
 59             });
 60         },
 61         /**
 62          * Lifecycle method
 63          * @function
 64          * @private
 65          * @returns {undefined}
 66          */
 67         syncUI: function() {
 68             this.leftElement = this.getChildById("leftElement");
 69             this.rightElement = this.getChildById("rightElement");
 70             this.centerElement = this.getChildById("centerElement");
 71             this.mcq = this.getChildById("flexi-mcq");
 72             this.maxSize = Math.max(this.leftElement.size(), this.rightElement.size(), this.centerElement.size());
 73             for (var i = 0; i < this.maxSize; i += 1) {
 74                 this.questionToDo[i] = i;
 75             }
 76             this.mask();
 77             this.next();
 78         },
 79         responseGiven: function(response) {
 80             var responseTime = Y.Lang.now() - this.startTime, reponseElement;
 81             this.ongoing = false;
 82             if ((reponseElement = this.centerElement.getActiveElement().flexiresponse) instanceof Y.Plugin.FlexiResponse &&
 83                     reponseElement.get("value") === response) {
 84                 this.mcq.success(responseTime);
 85             } else {
 86                 this.mcq.error(responseTime);
 87             }
 88             if (this.questionToDo.length === 0) {
 89                 this.mask();
 90                 alert("Test finished");
 91             } else {
 92                 this.next();
 93             }
 94         },
 95         next: function() {
 96             this.mask();
 97             this.set("currentLoading", {"left": true, "center": true, "right": true});
 98             this.currentQuestionId = this.generateNextId();
 99             this.centerElement.set("element", +this.currentQuestionId);
100             this.leftElement.set("element", +this.currentQuestionId);
101             this.rightElement.set("element", +this.currentQuestionId);
102             Y.later(this.get("fixPoint"), this, this.createLoadingEvent);
103         },
104         createLoadingEvent: function() {
105             this.centerElement.getActiveElement().onceAfter("render", function(e) {
106                 this.set("currentLoading.center", false);
107             }, this);
108             this.leftElement.getActiveElement().onceAfter("render", function(e) {
109                 this.set("currentLoading.left", false);
110             }, this);
111             this.rightElement.getActiveElement().onceAfter("render", function(e) {
112                 this.set("currentLoading.right", false);
113             }, this);
114         },
115         generateNextId: function() {
116             return this.get("random") ?
117                     this.questionToDo.splice(Math.round(Math.random() * (this.questionToDo.length - 1)), 1)[0] :
118                     this.questionToDo.shift();
119         },
120         startStimuli: function() {
121             this.get("boundingBox").focus();
122             this.restartTimer(this.leftElement);
123             this.restartTimer(this.rightElement);
124             this.restartTimer(this.centerElement);
125             Y.later(3, this, this.unmask);
126             this.ongoing = true;
127         },
128         restartTimer: function(widget) {
129             if (widget) {
130                 widget.fire("visibility-timer:restart");
131             }
132         },
133         mask: function() {
134             this.showOverlay();
135         },
136         unmask: function() {
137             this.hideOverlay();
138             this.startTime = Y.Lang.now();
139         },
140         getChildById: function(id) {
141             var returnItem = null;
142             this.some(function(item) {
143                 if (item.get("id") === id) {
144                     returnItem = item;
145                     return true;
146                 }
147             });
148             return returnItem;
149         },
150         /**
151          * Lifecycle method
152          * @function
153          * @private
154          * @returns {undefined}
155          */
156         destructor: function() {
157             var i;
158             for (i = 0; i < this.events.length; i += 1) {
159                 this.events[i].detach();
160             }
161         }
162     }, {
163         EDITORNAME: "Flexitests Controller",
164         ATTRS: {
165             currentLoading: {
166                 value: {},
167                 "transient": true
168             },
169             fixPoint: {
170                 value: 2000,
171                 type: "number"
172             },
173             random: {
174                 value: true,
175                 type: "boolean"
176             }
177         }
178     });
179 
180     Y.Plugin.FlexiResponse = Y.Base.create("wegas-flexi-response", Y.Plugin.Base, [Y.Wegas.Plugin, Y.Wegas.Editable], {
181     }, {
182         NS: "flexiresponse",
183         NAME: "flexiresponse",
184         EDITORNAME: "Flexitest Response",
185         ATTRS: {
186             "value": {
187                 value: "",
188                 type: "string"
189             }
190         }
191     });
192 });
