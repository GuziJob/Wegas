 11  * @author Francois-Xavier Aeberhard <fx@red-agent.com>

 13 YUI.add('wegas-variabledisplay', function(Y) {
 14     "use strict";
 15 
 16     var CONTENTBOX = 'contentBox', VariableDisplay;
 17 
 18     /**
 19      * @name Y.Wegas.VariableDisplay
 20      * @extends Y.Widget
 21      * @borrows Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable
 22      * @class class to display Wegas variables instance (or/and descriptor).
 23      * @constructor
 24      * @deprecated replaced by Y.Wegas.Template
 25      * @description  Display  Wegas variables instance (or/and descriptor) under
 26      * specifique shape : text, title, box, fraction and valuebox.
 27      */
 28     VariableDisplay = Y.Base.create("wegas-variabledisplay", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 29         /** @lends Y.Wegas.VariableDisplay# */
 30 
 31         // *** Private fields *** //
 32         /**
 33          * Reference to each used functions
 34          */
 35         handlers: null,
 36         // ** Lifecycle Methods ** //
 37         /**
 38          * @function
 39          * @private
 40          * @description Set variable with initials values.
 41          */
 42         initializer: function() {
 43             Y.log("Y.Wegas.VariableDisplay is deprecated, use Y.Wegas.Template instead.", "info", "Y.Wegas.VariableDisplay");
 44             this.handlers = [];
 45         },
 46         /**
 47          * @function
 48          * @private
 49          * @description bind function to events.
 50          * When VariableDescriptorFacade is updated, do syncUI
 51          */
 52         bindUI: function() {
 53             this.handlers.push(
 54                     Y.Wegas.Facade.VariableDescriptor.after("update", this.syncUI, this));
 55         },
 56         /**
 57          * @function
 58          * @private
 59          * @description get given variable descriptor and call
 60          *  "genMarkup" function with this variable.
 61          */
 62         syncUI: function() {
 63             var variableDescriptor = this.get("variable.evaluated");
 64             if (!variableDescriptor) {
 65                 return;
 66             }
 67 
 68             if (variableDescriptor instanceof Y.Wegas.persistence.ListDescriptor) {   // If the widget is a list,
 69                 variableDescriptor = variableDescriptor.get("currentItem");     // display it with the current list and the current element
 70             }
 71 
 72             this.get(CONTENTBOX).setHTML(this.genMarkup(variableDescriptor));   // Display the variable
 73 
 74         },
 75         /**
 76          * @function
 77          * @private
 78          * @description Detach all functions created by this widget
 79          */
 80         destructor: function() {
 81             for (var i = 0; i < this.handlers.length; i += 1) {
 82                 this.handlers[i].detach();
 83             }
 84         },
 85         // *** Private Methods *** //
 86         /**
 87          * @function
 88          * @private
 89          * @param variableDescriptor
 90          * @description format variables informations, in case of :
 91          * text : Display "as is" the value with the label.
 92          * title : Display "as is" he public label with the label.
 93          * box : Display the label, so many div as shown by the number
 94          *  from variable descriptor and the number.
 95          * fraction : Display the label, the min val of the descriptor, a
 96          *  separator, the value, a other separator and the max value.
 97          * valuebox : Display the label and so many div as shown from the min
 98          *  value to the max value. On each created div indicate if the div is
 99          *  under value, on value or over the value.
100          */
101         genMarkup: function(variableDescriptor) {
102             var acc, i, maxVal = variableDescriptor.get("maxValue"),
103                     minVal = variableDescriptor.get("minValue"),
104                     value = variableDescriptor.getInstance().get("value"),
105                     label = this.get("label") || variableDescriptor.getLabel();
106 
107             switch (this.get('view')) {
108                 case 'text':
109                     return '<span class="wegas-label wegas-variabledisplay-text-label">' + label + '</span>'
110                             + ' <span class="wegas-value wegas-variabledisplay-text-value">' + value + '</span>';
111                     break;
112 
113                 case 'title':
114                     return '<span class="wegas-label wegas-variabledisplay-text-label">' + label + '</span>'
115                             + ' <span class="wegas-value wegas-variabledisplay-text-value">' + variableDescriptor.getLabel() + '</span>';
116                     break;
117 
118                 case 'box':
119                     acc = [];
120                     for (i = 0; i < value; i += 1) {
121                         acc.push('<div class="wegas-variabledisplay-box-unit"></div>');
122                     }
123                     return '<div class="wegas-variabledisplay-box-label">' + label + '</div>'
124                             + '<span class="wegas-variabledisplay-box-units">' + acc.join('') + '</span>'
125                             + '<span class="wegas-variabledisplay-box-value">(' + value + '<span class="wegas-variabledisplay-box-valueMax">/' + maxVal + '</span>)</span>';
126                     break;
127 
128                 case 'fraction':
129                     return '<span class="wegas-variabledisplay-fraction-label">' + label + '</span>'
130                             + '<span class="wegas-variabledisplay-fraction-minValue">' + minVal + '</span>'
131                             + '<span class="wegas-variabledisplay-fraction-minSeparator"> / </span>'
132                             + '<span class="wegas-variabledisplay-fraction-value">' + value + '</span>'
133                             + '<span class="wegas-variabledisplay-fraction-maxSeparator"> / </span>'
134                             + '<span class="wegas-variabledisplay-fraction-maxValue">' + maxVal + '</span>';
135                     break;
136 
137                 case 'valuebox':
138                     acc = [];
139                     if (variableDescriptor) {
140                         for (i = variableDescriptor.get("minValue"); i <= variableDescriptor.get("maxValue"); i += 1) {
141                             acc.push('<div class="wegas-valuebox-unit '
142                                     + ((i === value) ? "wegas-valuebox-selected" : "")
143                                     + ((i < value) ? "wegas-valuebox-previous" : "")
144                                     + ((i > value) ? "wegas-valuebox-next" : "")
145                                     + '">' + i + '</div>');
146                         }
147                     }
148                     return '<span class="wegas-label wegas-variabledisplay-valuebox-label">' + label + '</span>'
149                             + '<div class="wegas-value wegas-variabledisplay-valuebox-units">' + acc.join('') + "</div>";
150                     break;
151             }
152 
153         }
154     }, {
155         /** @lends Y.Wegas.VariableDisplay# */
156 
157         /**
158          * @field
159          * @static
160          * @description
161          * <p><strong>Attributes</strong></p>
162          * <ul>
163          *    <li>variable: The target variable, returned either based on the variableName attribute,
164          *    and if absent by evaluating the expr attribute.</li>
165          *    <li>label: A label if you want display other label than public label
166          * of the variable descriptor</li>
167          *    <li>view: Choose the view to display variable descriptor</li>
168          * </ul>
169          */
170         ATTRS: {
171             /**
172              * The target variable, returned either based on the variableName attribute,
173              * and if absent by evaluating the expr attribute.
174              */
175             variable: {
176                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
177                 _inputex: {
178                     _type: "variableselect",
179                     label: "variable"
180                 }
181             },
182             /**
183              * A label if you want display other label than public label
184              * of the variable descriptor
185              */
186             label: {
187                 type: "string",
188                 optional: true,
189                 validator: Y.Lang.isString
190             },
191             /**
192              * Choose the view to display variable descriptor
193              */
194             view: {
195                 type: "string",
196                 value: "text",
197                 choices: [{
198                         value: "text"
199                     }, {
200                         value: "title"
201                     }, {
202                         value: "box"
203                     }, {
204                         value: "valuebox"
205                     }, {
206                         value: "fraction"
207                     }]
208             }
209         }
210     });
211     Y.namespace('Wegas').VariableDisplay = VariableDisplay;
212 
213 });
