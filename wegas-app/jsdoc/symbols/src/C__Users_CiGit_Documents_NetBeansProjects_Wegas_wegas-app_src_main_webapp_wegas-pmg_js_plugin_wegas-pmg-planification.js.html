 12 YUI.add('wegas-pmg-planification', function(Y) {
 13     "use strict";
 14 
 15     /**
 16      *  @class save or delete reservation
 17      *  @name Y.Plugin.Planification
 18      *  @extends Y.Plugin.Base
 19      *  @constructor
 20      */
 21     var Wegas = Y.Wegas,
 22             Planification = Y.Base.create("wegas-pmg-planification", Y.Plugin.Base, [Wegas.Plugin, Wegas.Editable], {
 23         /** @lends Y.Plugin.Planification */
 24 
 25         /**
 26          * Lifecycle methods
 27          * @function
 28          * @private
 29          */
 30         initializer: function() {
 31             this.get("host").datatable.delegate("click", function(e, a) {
 32                 var dt = this.get("host").datatable,
 33                         id = dt.getRecord(e.target).get("id"),
 34                         columnsCfg = dt.get('columns')[dt.getCell(e.target).get("cellIndex")];
 35                 this.checkCache(id, columnsCfg.time);
 36             }, "tbody .present, tbody .futur", this);
 37         },
 38         checkCache: function(descriptorId, periode) {
 39             var vd = Y.Wegas.Facade.VariableDescriptor.cache.find("id", descriptorId),
 40                     i, planPeriode;
 41 
 42             for (i = 0; i < vd.getInstance().get("plannification").length; i++) {
 43                 planPeriode = vd.getInstance().get("plannification")[i];
 44                 if (planPeriode === periode) {
 45                     // remove plannif
 46                     this.request(vd.getInstance().get("id"), periode, "DELETE");
 47                     return;
 48                 }
 49             }
 50             // add plannif
 51             this.request(vd.getInstance().get("id"), periode, "POST");
 52         },
 53         request: function(taskInstanceId, periode, method) {
 54             Wegas.Facade.VariableDescriptor.sendRequest({
 55                 request: "/ResourceDescriptor/Plannification/" + taskInstanceId + "/" + periode,
 56                 cfg: {
 57                     method: method
 58                 }
 59             });
 60         }
 61     }, {
 62         NS: "planification",
 63         NAME: "Planification"
 64     });
 65     Y.namespace("Plugin").Planification = Planification;
 66 });
