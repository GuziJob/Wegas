 13 YUI.add('wegas-gaugedisplay', function (Y) {
 14     "use strict";
 15 
 16     var CONTENTBOX = 'contentBox', GaugeDisplay;
 17 
 18     /**
 19      * @name Y.Wegas.GaugeDisplay
 20      * @extends Y.Widget
 21      * @borrows Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable
 22      * @class Manage a canevas gauge
 23      * @constructor
 24      * @description Manage a canevas gauge based on a instance's value
 25      */
 26     GaugeDisplay = Y.Base.create("wegas-gauge", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 27         /**
 28          * @lends Y.Wegas.GaugeDisplay#
 29          */
 30         // *** Private fields *** //
 31         /**
 32          * Content box of this widget, static
 33          */
 34         CONTENT_TEMPLATE: '<div style="text-align: center;line-height:3px"><canvas height="50px" width="100px"></canvas><center class="label"></center><center class="percent"></center></div>',
 35         /**
 36          * Maximum value displayed by the gauge, static
 37          */
 38         MAXVAL: 200,
 39         /**
 40          * Reference to each used functions
 41          */
 42         handlers: null,
 43         /**
 44          * reference to the gauge object
 45          */
 46         gauge: null,
 47         // ** Lifecycle Methods ** //
 48         /**
 49          * @function
 50          * @private
 51          * @description Set variables with initials values.
 52          */
 53         initializer: function () {
 54             this.handlers = [];
 55         },
 56         /**
 57          * @function
 58          * @private
 59          * @description create  and render a gauge with a defined configuration
 60          *  or given cfg (in ATTRS)
 61          */
 62         renderUI: function () {
 63             var opts = {
 64                 lines: this.get('cfg').lines || 12, // The number of lines to draw
 65                 angle: this.get('cfg').angle || 0.15, // The length of each line
 66                 lineWidth: this.get('cfg').lineWidth || 0.44, // The line thickness
 67                 pointer: this.get('cfg').pointer || {
 68                     length: 0.5, // The radius of the inner circle
 69                     strokeWidth: 0.035, // The rotation offset
 70                     color: '#000000'                                            // Fill color
 71                 },
 72                 colorStart: this.get('cfg').colorStart || '#0981A9', // Colors
 73                 colorStop: this.get('cfg').colorStop || '#000000',
 74                 //strokeColor: '#E0E0E0',
 75                 strokeColor: this.get('cfg').strokeColor || '#FFFFFF',
 76                 generateGradient: this.get('cfg').generateGradient || true
 77             };
 78             this.gauge = new Gauge(this.get("contentBox").one("canvas").getDOMNode());// create the  gauge!conso
 79             this.gauge.setOptions(opts);
 80             this.gauge.maxValue = this.MAXVAL;                                  // set max gauge value
 81             this.gauge.animationSpeed = 32;                                     // set animation speed (32 is default value)
 82         },
 83         /**
 84          * @function
 85          * @private
 86          * @description bind function to events.
 87          * When VariableDescriptorFacade is updated, do sync.
 88          */
 89         bindUI: function () {
 90             this.handlers.push(Y.Wegas.VariableDescriptorFacade.after("update", this.syncUI, this));
 91         },
 92         /**
 93          * @function
 94          * @private
 95          * @description Set the label, the min value, the max value, the
 96          *  value and display the percentage based on these value. This values
 97          *   are based on the descriptor/instance variable given in ATTRS.
 98          */
 99         syncUI: function () {
100             var maxVal, minVal, value, label,
101                     variableDescriptor = this.get("variable.evaluated");
102             if (!variableDescriptor) {
103                 return;
104             }
105 
106             label = this.get("label") || variableDescriptor.getPublicLabel();
107             minVal = variableDescriptor.get("minValue");
108             maxVal = variableDescriptor.get("maxValue") - minVal;
109             value = (variableDescriptor.getInstance().get("value") - minVal) / maxVal * this.MAXVAL;
110             if (!value) {
111                 value = 0.1;                                                    // @hack @fixme unkown bug, value seams to be treated by gauge as false...
112             }
113 
114             this.gauge.set(value);                                              // set actual value
115             this.get(CONTENTBOX).one(".label").setContent(label);
116             this.get(CONTENTBOX).one(".percent").setContent(Math.round(value / this.MAXVAL * 100) + "%");
117         },
118         /**
119          * @function
120          * @private
121          * @description Detach all functions created by this widget.
122          */
123         destructor: function () {
124             for (var i = 0; i < this.handlers.length; i += 1) {
125                 this.handlers[i].detach();
126             }
127         }
128     }, {
129         /**
130          * @lends Y.Wegas.GaugeDisplay#
131          */
132         /**
133          * @field
134          * @static
135          * @description
136          * <p><strong>Method</strong></p>
137          * <ul>
138          *    <li>variable: The target variable, returned either based on the name attribute,
139          * and if absent by evaluating the expr attribute.</li>
140          *    <li>label: A label for the gauge, if no one is given, take the
141          *     public label of the variable</li>
142          *    <li>cfg: the configuration of the gauge (object)</li>
143          * </ul>
144          */
145         ATTRS: {
146             /**
147              * The target variable, returned either based on the name attribute,
148              * and if absent by evaluating the expr attribute.
149              */
150             variable: {
151                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
152                 _inputex: {
153                     _type: "variableselect",
154                     label: "variable"
155                 }
156             },
157             /**
158              * A label for the gauge, if no one is given, take the public
159              *  label of the variable
160              */
161             label: {
162                 type: "string",
163                 optional: true,
164                 validator: Y.Lang.isString,
165                 _inputex: {
166                     label: "Label"
167                 }
168             },
169             /**
170              * The configuration of the gauge (object)
171              */
172             cfg: {
173                 value: {},
174                 _inputex: {
175                     _type: "object"
176                 }
177             }
178         }
179     });
180 
181     Y.namespace('Wegas').GaugeDisplay = GaugeDisplay;
182 });
