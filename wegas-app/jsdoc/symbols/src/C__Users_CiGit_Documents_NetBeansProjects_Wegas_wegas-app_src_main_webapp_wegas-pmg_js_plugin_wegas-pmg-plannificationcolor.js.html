 12 YUI.add('wegas-pmg-plannificationcolor', function(Y) {
 13     "use strict";
 14 
 15     /**
 16      *  @class color plannification in datatable
 17      *  @name Y.Plugin.Plannificationcolor
 18      *  @extends Y.Plugin.Base
 19      *  @constructor
 20      */
 21     var Wegas = Y.Wegas,
 22             Plannificationcolor = Y.Base.create("wegas-pmg-plannificationcolor", Y.Plugin.Base, [Wegas.Plugin, Wegas.Editable], {
 23         /** @lends Y.Plugin.Plannificationcolor */
 24 
 25         /**
 26          * Lifecycle methods
 27          * @function
 28          * @private
 29          */
 30         initializer: function() {
 31             this.onceAfterHostEvent("render", function() {
 32                 this.findCell();
 33 
 34                 this.afterHostMethod("syncUI", this.findCell);
 35 
 36                 this.get("host").datatable.after("sort", this.findCell, this);
 37             });
 38         },
 39         findCell: function() {
 40             Y.log("findCell()", "log", "Wegas.Plannificationcolor");
 41             var i, ii, host = this.get("host"),
 42                     dt = host.datatable,
 43                     plannification;
 44 
 45             for (i = 0; i < dt.data.size(); i++) {
 46                 plannification = dt.data.item(i).get("descriptor").getInstance().get("plannification");
 47                 for (ii = 0; ii < plannification.length; ii++) {
 48                     this.addColor(host.schedule.getCell(i, plannification [ii]));
 49                 }
 50             }
 51         },
 52         addColor: function(cell) {
 53             cell.append("<span class='editable plannification'></span>");
 54         }
 55     }, {
 56         NS: "plannificationcolor",
 57         NAME: "Plannificationcolor"
 58     });
 59     Y.namespace("Plugin").Plannificationcolor = Plannificationcolor;
 60 });
