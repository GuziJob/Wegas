 12 YUI.add('wegas-join-token', function(Y) {
 13     "use strict";
 14 
 15     var CONTENTBOX = 'contentBox', TokenJoin;
 16 
 17     /**
 18      * @name Y.Wegas.TokenJoin
 19      * @extends Y.Widget
 20      * @class  class for join a game and a team
 21      * @borrows Y.WidgetChild, Y.Wegas.Widget
 22      * @constructor
 23      * @description Allows to join a game by token or a public game. Then you can
 24      * join or create a new team
 25      */
 26     TokenJoin = Y.Base.create("wegas-join-token", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
 27         /**
 28          * @lends Y.Wegas.TokenJoin#
 29          */
 30         // *** Private fields *** //
 31 
 32         /**
 33          * @function
 34          * @private
 35          * @description All button and fields are created.
 36          * For creating the field inputEx libary is used
 37          */
 38         renderUI: function() {
 39             var cb = this.get(CONTENTBOX),
 40                     tokenParameter = Y.Wegas.Helper.getURLParameter("token");
 41 
 42             this.tokenField = new Y.inputEx.StringField({// Render
 43                 required: true,
 44                 parentEl: cb,
 45                 label: "Enrolment key",
 46                 typeInvite: "type here"
 47             });
 48 
 49             if (this.get("displayPublicGames")) {
 50                 this.renderPublicGames();
 51             }
 52 
 53             this.button = new Y.Button({
 54                 label: "Submit",
 55                 render: cb
 56             });
 57 
 58             if (tokenParameter) {
 59                 this.sendTokenJoin(tokenParameter);
 60             }
 61         },
 62         /**
 63          * @function
 64          * @private
 65          * @description All events are added to the buttons
 66          * Create team button call rest url : rest/GameModel/{gameModelID}/Game/{gameID}/CreateTeam/{teamName}
 67          */
 68         bindUI: function() {
 69 
 70             if (this.get("displayPublicGames")) {
 71                 this.tokenField.on("updated", function(e) {
 72                     if (this.tokenField.getValue() !== "") {
 73                         this.selectPublicGame.setValue("");
 74                     }
 75                 }, this);
 76 
 77                 this.selectPublicGame.on("updated", function(e) {
 78                     if (this.selectPublicGame.getValue() !== "") {
 79                         this.tokenField.setValue("");
 80                     }
 81                 }, this);
 82             }
 83 
 84             this.button.on("click", function() {                                // join a game based on a token
 85                 if (this.tokenField.validate()) {
 86                     this.sendTokenJoin(this.tokenField.getValue());
 87                     //} else if (this.selectPublicGame.getValue() !== "") {
 88                     // @Todo
 89                     // this.joinTeam = new Y.Wegas.SelectTeam({
 90                     //    entity: this.selectPublicGame.getValue()
 91                     //}
 92                 } else {
 93                     this.showMessage("error", "Enter a valid token");
 94                     return;
 95                 }
 96             }, this);
 97         },
 98         destructor: function() {
 99             if (this.teamWidget) {
100                 this.teamWidget.destroy();
101             }
102             this.button.destroy();
103             this.tokenField.destroy();
104         },
105         /**
106          * @function
107          * @private
108          * @description Method for join a game by token
109          * Call rest request for join the game : rest/GameModel/1/Game/{gameModelID}/JoinGame/{token}
110          */
111         sendTokenJoin: function(token) {
112             this.showOverlay();
113             Y.log("sendTokenJoin()", "info", "Wegas.TokenJoin");
114 
115             Y.Wegas.Facade.Game.sendRequest({
116                 request: "/JoinGame/" + token + "?view=Extended",
117                 cfg: {
118                     updateCache: false
119                 },
120                 on: {
121                     success: Y.bind(function(e) {
122                         var cb = this.get(CONTENTBOX),
123                                 entity = e.response.entity,
124                                 gm;
125                         this.hideOverlay();
126 
127                         if (entity === "Team token required") {                 // Team token is required, game token was provided
128                             // @fixme should show a message when this happens from the lobby
129 
130 //                        } else if (entity instanceof Y.Wegas.persistence.Game) {
131 //                            gm = entity.get("gameModel");
132 //                                    &&
133 //                                !(gm.get("properties.allowCreateTeam") || gm.get("properties.allowJoinTeam"))) {
134                         } else if (e.response.entities[0] instanceof Y.Wegas.persistence.Team
135                                 && !(e.response.entities[1].get("gameModel").get("properties.freeForAll")
136                                 || e.response.entities[0].get("players").length === 0)) {// If the token is already in use
137 
138                             this.showMessageBis("error",
139                                     "This team has already been created. You can contact it's members so they can join you in.");
140                         } else {
141                             Y.log("sendTokenJoin(): Rendering team widget", "info", "Wegas.TokenJoin");
142                             this.destructor();
143                             this.teamWidget = new Y.Wegas.JoinTeam({//          // Player can choose or create its team
144                                 entity: e.response.entities,
145                                 render: cb
146                             });
147                             this.teamWidget.addTarget(this);                    // So overlay and message events will be forwarded
148                         }
149                     }, this),
150                     failure: Y.bind(function(e) {
151                         this.hideOverlay();
152                         this.showMessage("error", e.response.results.message || "Invalid token");
153                     }, this)
154                 }
155             });
156         },
157         /**
158          * @function
159          * @private
160          * @description Add all public game in a listbox
161          * Use rest request for get public games: rest/PublicGames/Games/{userID}
162          */
163         renderPublicGames: function() {
164             var cb = this.get("contentBox");
165             cb.append('<div class="lobbyOr"><p>Or</p><div>');
166 
167             this.selectPublicGame = new Y.inputEx.SelectField({//               // Render public games
168                 required: false,
169                 parentEl: cb,
170                 label: "Select a public game",
171                 choices: [{
172                         label: "Select game",
173                         value: null
174                     }]
175             });
176 
177             Y.Wegas.Facade.PublicGames.sendRequest({//                          // Retrieve the list of public games from the server
178                 request: "/" + Y.Wegas.app.get('currentUser').id,
179                 on: {
180                     success: Y.bind(function(e) {
181                         var data = e.response.results.entities;
182                         Y.Array.each(data, function(game) {
183                             this.selectPublicGame.addChoice({
184                                 label: game.get("name"),
185                                 value: game
186                             });
187                         }, this);
188                     }, this),
189                     failure: Y.bind(this.defaultFailureHandler, this)
190                 }
191             });
192         }
193     }, {
194         ATTRS: {
195             displayPublicGames: {
196                 value: false
197             }
198         }
199     });
200 
201     Y.namespace('Wegas').TokenJoin = TokenJoin;
202 });
