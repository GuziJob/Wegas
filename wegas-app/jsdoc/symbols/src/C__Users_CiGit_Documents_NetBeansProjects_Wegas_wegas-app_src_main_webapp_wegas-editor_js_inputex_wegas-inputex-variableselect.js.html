 12 YUI.add("wegas-inputex-variableselect", function(Y) {
 13     "use strict";
 14     var inputEx = Y.inputEx;
 15     /**
 16      * Edit an object referencing a variable with format:
 17      *    {
 18      *      name: "name",
 19      *      expr: "expr",
 20      *      dataSource: "ds"
 21      *    }
 22      *
 23      * @class inputEx.Variableselect
 24      * @extends inputEx.StringGroup
 25      **/
 26     Y.namespace("inputEx.Wegas").Variableselect = function(options) {
 27         inputEx.Wegas.Variableselect.superclass.constructor.call(this, options);
 28     };
 29     Y.extend(inputEx.Wegas.Variableselect, inputEx.Group, {
 30         setOptions: function(options) {
 31             inputEx.Wegas.Variableselect.superclass.setOptions.call(this, options);
 32             this.options.fields = [{
 33                     label: "Variable to display",
 34                     type: 'variabledescriptorgetter',
 35                     classFilter: options.classFilter,
 36                     name: 'name'
 37                 }, {
 38                     label: 'Or expression',
 39                     wrapperClassName: 'inputEx-fieldWrapper wegas-advanced-feature',
 40                     name: 'expr'
 41                 }];
 42         },
 43         validate: function() {
 44             return this.inputs[1].validate() || this.inputs[0].validate();
 45         },
 46         getValue: function() {
 47             if (this.inputs[1].getValue() && !this.inputs[1].getValue().length > 0) {
 48                 return {
 49                     expr: this.inputs[1].getValue()
 50                 };
 51             } else {
 52                 return {//save name for portability instead of id
 53                     name: this.inputs[0].currentEntityField.getValue()
 54                 };
 55             }
 56         },
 57         setValue: function(val, fireUpdatedEvent) {
 58             if (val.id) {
 59                 var findVal = Y.Wegas.Facade.VariableDescriptor.cache.findById(val.id);
 60                 val.name = findVal.get("name");                                 // @fixme
 61             }
 62             inputEx.Wegas.Variableselect.superclass.setValue.call(this, val, fireUpdatedEvent);
 63         }
 64 
 65     });
 66     inputEx.registerType("variableselect", inputEx.Wegas.Variableselect); // Register this class as "wegasurl" type
 67 
 68 });
