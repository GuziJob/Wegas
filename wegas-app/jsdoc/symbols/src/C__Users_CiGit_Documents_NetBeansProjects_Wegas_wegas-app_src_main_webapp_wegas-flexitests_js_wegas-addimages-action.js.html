 13 YUI.add("wegas-addimages-action", function(Y) {
 14     "use strict";
 15     /**
 16      * Add images all images from specified file directory
 17      * @constructor
 18      * @name Y.Plugin.AddImagesWidgetAction
 19      * @extends Y.Plugin.Action
 20      */
 21     var AddImagesWidget = function() {
 22         AddImagesWidget.superclass.constructor.apply(this, arguments);
 23     };
 24     Y.extend(AddImagesWidget, Y.Plugin.Action, {
 25         /** @lends Y.Plugin.AddImagesWidgetAction# */
 26         execute: function() {
 27 //            var dir = prompt("Directory to add", "/");
 28             this.fs = new Y.Wegas.FileSelect({
 29                 title: "Choose a directory",
 30                 filter: function() {
 31                     return /application\/wfs-directory/.test(this.get("data.mimeType"));
 32                 }
 33             });
 34             this.fs.on("directorySelected", function(e, path) {
 35                 this.get("fileDataSource").sendRequest({
 36                     request: "list" + path,
 37                     cfg: {
 38                         headers: {
 39                             'Content-Type': 'application/json; charset=ISO-8859-1',
 40                             'Managed-Mode': false
 41                         }
 42                     },
 43                     on: {
 44                         success: Y.bind(function(e) {
 45                             this._processNodes(e.response.results);
 46                         }, this),
 47                         failure: null
 48                     }
 49                 });
 50                 this.fs.destroy();
 51             }, this);
 52         },
 53         _processNodes: function(list) {
 54             var path, i;
 55             for (i = 0; i < list.length; i += 1) {
 56                 if (list[i].mimeType.match("image/.*")) {
 57                     path = list[i].path + (list[i].path.match(".*/$") ? "" : "/") + list[i].name;
 58                     this.get("widget").add({type: "Image", url: path});
 59                 }
 60             }
 61             this.get("dataSource").cache.patch(this.get("widget").get("root").toObject());
 62         }
 63     }, {
 64         /** @lends Y.Plugin.AddImagesWidgetAction */
 65         NS: "AddImagesWidgetAction",
 66         NAME: "AddImagesWidgetAction",
 67         ATTRS: {
 68             widget: {},
 69             dataSource: {
 70                 getter: function(val) {
 71                     if (!val) {
 72                         return Y.Wegas.Facade.Page;
 73                     }
 74                     return val;
 75                 }
 76             },
 77             fileDataSource: {
 78                 getter: function(val) {
 79                     if (!val) {
 80                         return Y.Wegas.Facade.File;
 81                     }
 82                     return val;
 83                 }
 84             }
 85         }
 86     });
 87 
 88     Y.Plugin.AddImagesWidgetAction = AddImagesWidget;
 89 });
