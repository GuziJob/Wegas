 12 /**
 13  * @module inputex-select
 14  */
 15 YUI.add("wegas-inputex-multipleoptions", function(Y) {
 16 
 17     var inputEx = Y.inputEx;
 18 
 19     /**
 20      * Create a select field
 21      * @class inputEx.MultipleOptions
 22      * @extends inputEx.Group
 23      * @constructor
 24      */
 25     inputEx.MultipleOptions = function(options) {
 26         options.className = options.className || "inputEx-Group wegas-inputex-multioptions";
 27         inputEx.MultipleOptions.superclass.constructor.call(this, options);
 28     };
 29 
 30     Y.extend(inputEx.MultipleOptions, inputEx.Group, {
 31         renderField: function(fieldOptions) {
 32             var field = inputEx.MultipleOptions.superclass.renderField.call(this, fieldOptions);
 33             field.on("focus", function(e) {
 34                 this.select(e.target);
 35             }, this);
 36             if (this.inputs.length > 1) {
 37                 this.toggle(field, false);
 38             } else {
 39                 this.select(field);
 40             }
 41             (new Y.Node(field.divEl)).on("click", field.el.focus, field.el);    // When a user click
 42             return field;
 43         },
 44         select: function(field) {
 45             this.selectedField = field;
 46             this.toggleAll(false);
 47             this.toggle(field, true);
 48         },
 49         getSelected: function() {
 50             return this.selectedField;
 51         },
 52         addField: function(fieldOptions) {
 53             inputEx.MultipleOptions.superclass.addField.call(this, fieldOptions);
 54             if (this.inputs.length === 1) {
 55                 //(new Y.Node(this.fieldset)).append("<div class=\"wegas-inputex-nulti-separator\"><br /><center>OR</center><br /></div>");
 56             }
 57         },
 58         toggleAll: function(value) {
 59             var i;
 60             for (i = 0; i < this.inputs.length; i += 1) {
 61                 this.toggle(this.inputs[i], value);
 62             }
 63         },
 64         toggle: function(field, value) {
 65             (new Y.Node(field.divEl)).toggleClass("wegas-inputex-multi-selected", value);
 66         }
 67     });
 68 
 69     Y.inputEx.Field.prototype.onFocus = function(e) {
 70         var el = Y.one(this.getEl());
 71         el.removeClass('inputEx-empty');
 72         el.addClass('inputEx-focused');
 73         this.publish("focus", {
 74             emitFacade: true
 75         });
 76         this.fire("focus");
 77     };
 78 
 79     // Register this class as "select" type
 80     inputEx.registerType("multipleoptions", inputEx.MultipleOptions);
 81 
 82 }, '3.1.0', {
 83     requires: ['inputex-field', 'inputex-group']
 84 });
