 12 YUI.add('wegas-cssstyles', function(Y) {
 13     "use strict";
 14 
 15     /**
 16      *  @class Add styles CSS styles
 17      *  @name Y.Plugin.CSSStyles
 18      *  @extends Y.Plugin.Base
 19      *  @constructor
 20      */
 21     var Wegas = Y.Wegas,
 22         styleList = [],
 23         node,
 24     CSSStyles = Y.Base.create("wegas-cssstyles", Y.Plugin.Base, [Wegas.Plugin, Wegas.Editable], {
 25         /** @lends Y.Plugin.CSSStyles */
 26 
 27         /**
 28          * @function
 29          * @private
 30          */
 31         initializer: function() {
 32             this.setValue(this.get("styles"));
 33             this.after("stylesChange", function(e){
 34                 this.removeStyle(e);
 35                 this.setValue(e.newVal);
 36             });
 37         },
 38 
 39         removeStyle: function(e) {
 40             var styleToRemove;
 41             for (styleToRemove in e.prevVal){
 42                 if (!e.newVal.hasOwnProperty(styleToRemove)){
 43                     node.setStyle(styleToRemove, "");
 44                     styleList.splice(styleList.indexOf(styleToRemove));
 45                 }
 46             }
 47         },
 48                 
 49         setStyle: function(newStylesList, style){
 50             if (styleList.indexOf(style) === -1){
 51                 styleList.push(style);
 52             }
 53             node.setStyle(style, newStylesList[style]);
 54         },
 55         
 56         destructor: function(){
 57             var styleToRemove;
 58             for (styleToRemove in this.get("styles")){
 59                 node.setStyle(styleToRemove, "");
 60                 styleList.splice(styleList.indexOf(styleToRemove));
 61             }
 62         },
 63         
 64         setValue: function(styles) {
 65             if (this.get("host") instanceof Y.Widget) {
 66                 node = this.get("host").get(this.get("targetNode"));
 67             } else if (this.get("host") instanceof Y.Node){
 68                 node = this.get("host");
 69             } else {
 70                 Y.log("Host's type mistmach", "warn", "Y.Plugin.CSSStyles");
 71                 return;
 72             }
 73 
 74             if (styles){
 75                 for (var style in styles){
 76                     var value = styles[style];                    
 77                     if (value){
 78                         if (style === "fontSize" || style === "top" || style === "right" || style === "bottom" || style === "left" || style === "minWidth" || style === "width" || style === "height"){
 79                             if (value.substr(-2) !== "px" && value.substr(-2) !== "pt" && value.substr(-2) !== "em" && value.substr(-1) !== "%"){
 80                                   styles[style] = value + "pt";
 81                             }
 82                         }
 83                     }
 84                     this.setStyle(styles, style);
 85                 }
 86             }
 87         }
 88     }, {
 89         ATTRS: {
 90             styles: {         
 91                 value: {},
 92                 _inputex: {
 93                     _type: "wegasobject",
 94                     elementType: {
 95                         type:"wegaskeyvalue",
 96                         availableFields: [{
 97                             name: "backgroundColor",
 98                             type: "colorpicker",
 99                             palette:3
100                         }, {
101                             name: "color",
102                             type: "colorpicker",
103                             palette:3
104                         }, {
105                             type: "string",
106                             name: "fontSize"
107                         }, {
108                             type: "select",
109                             name: "fontStyle",
110                             choices: ["", "normal", "italic", "oblique", "inherit"]
111                         }, {
112                             type: "select",
113                             name: "textAlign",
114                             choices: ["", "left", "right", "center", "justify", "inherit"]
115                         }, {
116                             type: "string",
117                             name: "minWidth"
118                         }]
119                     }
120                 }
121             },
122             targetNode: {
123                 value: "boundingBox",
124                 type: "string",
125                 "transient": true
126             }
127         },
128         NS: "CSSStyles",
129         NAME: "CSSStyles"
130     });
131     Y.namespace("Plugin").CSSStyles = CSSStyles;
132 
133 });
