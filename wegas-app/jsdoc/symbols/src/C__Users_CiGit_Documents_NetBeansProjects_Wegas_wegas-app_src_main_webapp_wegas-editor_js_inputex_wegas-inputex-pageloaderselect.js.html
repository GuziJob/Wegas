 13 YUI.add("wegas-inputex-pageloaderselect", function(Y) {
 14     "use strict";
 15 
 16     Y.namespace("inputEx.Wegas").PageloaderSelect = function(options) {
 17         Y.inputEx.Wegas.PageloaderSelect.superclass.constructor.call(this, options);
 18     };
 19     Y.extend(Y.inputEx.Wegas.PageloaderSelect, Y.inputEx.SelectField, {
 20         setOptions: function(options) {
 21             var list, root = Y.Wegas.PageLoader.pageLoaderInstances.previewPageLoader;
 22             Y.inputEx.Wegas.PageloaderSelect.superclass.setOptions.call(this, options);
 23             list = root ?
 24                     root.get("contentBox").all(".wegas-pageloader") :
 25                     new Y.ArrayList();
 26             list.each(function(item) {
 27                 var w = Y.Widget.getByNode(item);
 28                 if (root.get("widget") === w.get("root")) { //only pageloader on currently edited page. Not those on subpages.
 29                     this.options.choices.push({value: w.get("pageLoaderId")});
 30                 }
 31 
 32             }, this);
 33 
 34         }
 35     });
 36 
 37     Y.inputEx.registerType("pageloaderselect", Y.inputEx.Wegas.PageloaderSelect);
 38 });
