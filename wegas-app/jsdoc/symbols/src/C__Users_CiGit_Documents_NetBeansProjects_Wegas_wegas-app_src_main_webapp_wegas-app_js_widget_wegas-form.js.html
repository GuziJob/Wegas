 15 YUI.add('wegas-form', function (Y) {
 16     "use strict";
 17 
 18     /**
 19      * @name Y.Wegas.Form
 20      * @extends Y.Widget
 21      * @borrows Y.WidgetChild, Y.Wegas.Widget
 22      * @class  class to submit a form 
 23      * @constructor
 24      * @description Add a toolbar with buttons "submit" and "cancel" to manger forms.
 25      */
 26     var Form = Y.Base.create("wegas-form", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
 27         /**
 28          * @lends Y.Wegas.Form#
 29          */
 30 
 31         // ** Lifecycle Methods ** //
 32         /**
 33          * @function
 34          * @private
 35          * @description plug a toolbar and publich "submit" event.
 36          */
 37         initializer: function () {
 38             this.plug(Y.Plugin.WidgetToolbar);
 39             this.publish("submit", {
 40                 emitFacade: true
 41             });
 42         },
 43         /**
 44          * @function
 45          * @private
 46          * @description call function "renderToolbar".
 47          */
 48         renderUI: function () {
 49             this.renderToolbar();
 50         },
 51         // ** Private Methods ** //
 52         /**
 53          * @function
 54          * @private
 55          * @description Add a submit button (with "afterValidation" and
 56          *  "submit" events) and a cancel button (with "cancel" event).
 57          * Render the Toolbar.
 58          */
 59         renderToolbar: function () {
 60             var toolbarNode = this.toolbar.get('header');
 61 
 62             this.saveButton = new Y.Button({
 63                 label: "<span class=\"wegas-icon wegas-icon-save\" ></span>Save",
 64                 on: {
 65                     click: Y.bind(function () {
 66                         var form = this.get("form"),
 67                                 val = form.getValue();
 68 
 69                         if (!form.validate()) {
 70                             return;
 71                         }
 72                         form.fire("afterValidation");
 73                         if (val.valueselector) {
 74                             val = val.valueselector;
 75                         }
 76                         this.fire("submit", {
 77                             value: val
 78                         });
 79                     }, this)
 80                 }
 81             }).render(toolbarNode);
 82 
 83             this.cancelButton = new Y.Button({
 84                 label: "<span class=\"wegas-icon wegas-icon-cancel\" ></span>Cancel",
 85                 on: {
 86                     click: Y.bind(function () {
 87                         this.fire("cancel");
 88                     }, this)
 89                 }
 90             }).render(toolbarNode);
 91         },
 92         /**
 93          * @function
 94          * @private
 95          * @description set the given form to null
 96          */
 97         destroyForm: function () {
 98             this.set("form", null);
 99         }
100         /**
101          * @lends Y.Wegas.Form
102          */
103         /**
104          * @field
105          * @static
106          * @description
107          * <p><strong>Method</strong></p>
108          * <ul>
109          *    <li>values: values of fields of the form</li>
110          *    <li>form: the form to manage (see YUI Form)</li>
111          *    <li>cfg: configuation of the form (see YUI Form)</li>
112          * </ul>
113          */
114     }, {
115         ATTRS: {
116             /**
117              * Values of fields of the form
118              */
119             values: {
120                 value: {
121                     setter: function (val) {
122                         if (this.get("form")) {
123                             this.form.setValue(val);
124                         }
125                     }
126                 }
127             },
128             /**
129              * The form to manage
130              */
131             form: {
132                 setter: function (val) {
133                     if (this.get("form")) {                                     // If there is alread a form instantiated, destroy it
134                         this.get("form").destroy();
135                     }
136                     return val;
137                 }
138             },
139             /**
140              * Configuation of the form
141              */
142             cfg: {
143                 setter: function (val) {
144                     val.parentEl = this.get("contentBox");                        // Set up the form parentEl attribute, so it knows where to render
145                     val.className = "wegas-form-ix";
146                     Y.inputEx.use(val, Y.bind(function (cfg) {                  // Load form dependencies
147                         var form = Y.inputEx(cfg);                              // Initialize and render form
148                         form.setValue(this.get("values"));                      // Sync form with "values" ATTR
149                         this.set("form", form);
150                     }, this, val));
151                     return val;
152                 }
153             }
154         }
155     });
156 
157     Y.namespace("Wegas").Form = Form;
158 
159     var inputEx = Y.inputEx,
160             lang = Y.Lang;
161 
162     /**
163      *  @hack So we can easily change classs on inputex fields
164      */
165     Y.inputEx.Field.prototype.addClassName = function (className) {
166         Y.one(this.divEl).addClass(className);
167     };
168     Y.inputEx.Field.prototype.removeClassName = function (className) {
169         Y.one(this.divEl).removeClass(className);
170     };
171     /*
172      * @hack Automatically add the "optional" message when necessary
173      */
174     Y.inputEx.StringField.prototype.setOptions = function (options) {
175         Y.inputEx.StringField.superclass.setOptions.call(this, options);
176 
177         this.options.regexp = options.regexp;
178         this.options.size = options.size;
179         this.options.maxLength = options.maxLength;
180         this.options.minLength = options.minLength;
181         this.options.typeInvite = options.typeInvite;
182         if (!this.options.required && this.options.typeInvite === undefined) {  // !!!MODIFIED!!!
183             this.options.typeInvite = "optional";
184         }
185         this.options.readonly = options.readonly;
186         this.options.autocomplete = lang.isUndefined(options.autocomplete) ?
187                 inputEx.browserAutocomplete :
188                 (options.autocomplete === false || options.autocomplete === "off") ? false : true;
189         this.options.trim = (options.trim === true) ? true : false;
190     };
191     /**
192      * @hack Let inputex also get requirement from selectfields, lists
193      */
194     Y.inputEx.getRawModulesFromDefinition = function (inputexDef) {
195 
196         var type = inputexDef.type || 'string',
197                 module = YUI_config.groups.inputex.modulesByType[type],
198                 modules = [module || type],
199                 //set fields if they exist
200                 fields = inputexDef.fields
201                 //else see if we have elementType for lists - if neither then we end up with null
202                 || inputexDef.availableFields || [];
203 
204         if (inputexDef.elementType) {
205             fields.push(inputexDef.elementType);
206         }
207 
208         // recursive for group,forms,list,combine, etc...
209         Y.Array.each(fields, function (field) {
210             modules = modules.concat(this.getModulesFromDefinition(field));
211         }, this);
212 
213         // TODO: inplaceedit  editorField
214 
215         return modules;
216     };
217 });
