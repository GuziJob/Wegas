 12 YUI.add('wegas-widgettoolbar', function (Y) {
 13     "use strict";
 14 
 15     /**
 16      *  @name Y.Wegas.WidgetToolbar
 17      *  @class Adds a toolbar to target Y.Widget
 18      *  @extends Y.Plugin.Base
 19      *  @constructor
 20      */
 21     var  BOUNDINGBOX = "boundingBox", HOST ="host",
 22     WidgetToolbar = function () {
 23         WidgetToolbar.superclass.constructor.apply(this, arguments);
 24     };
 25 
 26     Y.extend(WidgetToolbar, Y.Plugin.Base, {
 27         /** @lends Y.Wegas.WidgetToolbar */
 28 
 29         // *** Lifecycle methods *** //
 30         /**
 31          * @function
 32          * @private
 33          */
 34         initializer: function () {
 35             this.children = [];
 36             this.render();
 37         //this.afterHostEvent("render", this.render, this);
 38         },
 39 
 40         /**
 41          * @function
 42          * @private
 43          */
 44         destructor: function () {
 45             var i;
 46             for (i = 0; i < this.children.length; i = i + 1) {
 47                 this.children[i].destroy();
 48             }
 49         },
 50 
 51         // *** Private methods *** //
 52         /**
 53          * @function
 54          * @private
 55          */
 56         render: function () {
 57             var i, host = this.get(HOST),
 58             children = this.get("children");
 59             host.get(BOUNDINGBOX).addClass("wegas-hastoolbar")
 60             .prepend('<div class="wegas-toolbar"><div class="wegas-toolbar-header"></div><div class="wegas-toolbar-panel"></div></div>');
 61             host.get('contentBox').addClass("wegas-toolbar-sibling");
 62 
 63             for (i = 0; i < children.length; i = i + 1) {
 64                 this.add(children[i]);
 65             }
 66         },
 67 
 68         /**
 69          *
 70          * Adds a widget to the toolbar
 71          *
 72          * @function
 73          * @param {Y.Widget|Object} widget A widget instance or an widget cfg object to be instantiated
 74          * @return {Y.Widget} the newly created widget
 75          */
 76         add: function (widget) {
 77             if (!(widget instanceof Y.Widget)) {
 78                 widget = Y.Wegas.Widget.create(widget);
 79             }
 80             widget.render(this.get("header"));
 81             widget.addTarget(this.get(HOST));
 82             this.children.push(widget);
 83             return widget;
 84         },
 85 
 86         /**
 87          * Returns a toolbar widget based on its index
 88          * @function
 89          * @param {Number} index
 90          * @returns {Y.Widget}
 91          */
 92         item: function (index) {
 93             return this.children[index];
 94         },
 95 
 96         size: function () {
 97             return this.children.length;
 98         }
 99 
100     }, {
101         /** @lends Y.Plugin.WidgetToolbar */
102         NS: "toolbar",
103         NAME: "toolbar",
104 
105         /**
106          * <p><strong>Config attributes</strong></p>
107          * <ul>
108          *    <li>children {Y.Widget[]} A list of all widgets contained in the toolbar</li>
109          *    <li>header {Y.Node} The main toolbar node</li>
110          *    <li>panel {Y.Node} The panel node, visible on mouse over</li>
111          * </ul>
112          *
113          * @field
114          * @static
115          */
116         ATTRS: {
117             children: {
118                 value: []
119             },
120             header: {
121                 lazyAdd: false,
122                 value: false,
123                 getter : function () {
124                     return this.get(HOST).get(BOUNDINGBOX).one(".wegas-toolbar-header");
125                 }
126             },
127             panel: {
128                 lazyAdd: false,
129                 value: false,
130                 getter : function () {
131                     return this.get(HOST).get(BOUNDINGBOX).one(".wegas-toolbar-panel");
132                 }
133             }
134         }
135     });
136     Y.namespace('Plugin').WidgetToolbar = WidgetToolbar;
137 
138 });
