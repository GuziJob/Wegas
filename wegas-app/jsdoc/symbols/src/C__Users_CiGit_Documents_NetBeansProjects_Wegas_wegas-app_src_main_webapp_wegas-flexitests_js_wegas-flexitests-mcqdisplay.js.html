 13 YUI.add("wegas-flexitests-mcqdisplay", function(Y) {
 14     "use strict";
 15     var INITIALVALUE = "flexi_initial_value";
 16 
 17     Y.namespace('Wegas').FlexitestsMCQ = Y.Base.create("wegas-flexitests-mcqdisplay", Y.Widget,
 18             [Y.WidgetChild, Y.Wegas.Widget, Y.Wegas.Editable], {
 19         /**
 20          * Lifecycle method
 21          * @function
 22          * @private
 23          * @returns {undefined}
 24          */
 25         initializer: function() {
 26             this.events = [];
 27             this.choices = null;
 28         },
 29         /**
 30          * Lifecycle method
 31          * @function
 32          * @private
 33          * @returns {undefined}
 34          */
 35         renderUI: function() {
 36             var cb = this.get("contentBox");
 37             cb.append("<div class='" + this.getClassName("feedback") + "'></div>");
 38             cb.append("<div class='" + this.getClassName("input") + "'></div>");
 39         },
 40         /**
 41          * Lifecycle method
 42          * @function
 43          * @private
 44          * @returns {undefined}
 45          */
 46         syncUI: function() {
 47             this.generators[this.get("responseType")].apply(this);
 48         },
 49         /**
 50          * Lifecycle method
 51          * @function
 52          * @private
 53          * @returns {undefined}
 54          */
 55         bindUI: function() {
 56             this.get("contentBox").delegate("click", function(e) {
 57                 e.halt(true);
 58                 this.fire("clientResponse", {value: e.currentTarget.getData("reference")});
 59             }, "." + this.getClassName("input") + " > span", this);
 60             this.get("contentBox").one("." + this.getClassName("input")).delegate("change", function(e) {
 61                 e.halt(true);
 62                 this.fire("clientResponse", {value: e.target.getDOMNode().value});
 63                 e.target.getDOMNode().value = INITIALVALUE;
 64             }, "select", this);
 65             this.events.push(this.get("root").get("boundingBox").after("keypress", function(e) {
 66                 this.keyPressed(String.fromCharCode(e.keyCode));
 67             }, this));
 68         },
 69         keyPressed: function(charCode) {
 70             var choice;
 71             if (this.get("responseType") !== "key") {
 72                 return;
 73             }
 74             switch (charCode) {
 75                 case "f":
 76                     choice = 0;
 77                     break;
 78                 case "h":
 79                     choice = 1;
 80                     break;
 81                 default:
 82                     return;
 83             }
 84             this.fire("clientResponse", {value: this.choices[choice].get("label")});
 85         },
 86         success: function(time) {
 87             if (+this.get("feedback") > 0) {
 88                 this.get("contentBox").one("." + this.getClassName("feedback")).setContent("<i style='color:lightgreen'>GOOD: " + time + " ms</i>");
 89                 Y.later(+this.get("feedback"), this, this.clearFeedBack);
 90             }
 91         },
 92         error: function(time) {
 93             if (+this.get("feedback") > 0) {
 94                 this.get("contentBox").one("." + this.getClassName("feedback")).setContent("<i style='color:red'>BAD: " + time + "ms</i>");
 95                 Y.later(+this.get("feedback"), this, this.clearFeedBack);
 96             }
 97         },
 98         clearFeedBack: function() {
 99             this.get("contentBox").one("." + this.getClassName("feedback")).empty();
100         },
101         generators: {
102             link: function() {
103                 var cb = this.get("contentBox"),
104                         inputDiv = cb.one("." + this.getClassName("input")),
105                         question = this.get("variable.evaluated");
106                 inputDiv.empty();
107                 for (var i in question.get("items")) {
108                     inputDiv.append("<span data-reference='" + question.get("items")[i].get("label") + "'>" + question.get("items")[i].get("description") + "</span><br/>");
109                 }
110             },
111             selector: function() {
112                 var cb = this.get("contentBox"),
113                         inputDiv = cb.one("." + this.getClassName("input")),
114                         question = this.get("variable.evaluated");
115                 inputDiv.empty();
116                 inputDiv.append("<select><option value='" + INITIALVALUE + "'>Choose:</option></select>");
117                 inputDiv = inputDiv.get("firstChild");
118                 for (var i in question.get("items")) {
119                     inputDiv.append("<option value='" + question.get("items")[i].get("label") + "'>" + question.get("items")[i].get("description") + "</option>");
120                 }
121             },
122             key: function() {
123                 this.choices = this.get("variable.evaluated").get("items");
124             }
125         },
126         /**
127          * Lifecycle method
128          * @function
129          * @private
130          * @returns {undefined}
131          */
132         destructor: function() {
133             for (var i in this.events) {
134                 this.events[i].detach();
135             }
136         }
137     }, {
138         EDITORNAME: "Flexitests MCQ Display",
139         ATTRS: {
140             feedback: {
141                 value: 1000,
142                 type: "number",
143                 _inputex: {
144                     label: "Feedback time (ms).<br/> 0 or less to disable"
145                 }
146             },
147             responseType: {
148                 value: "link",
149                 type: "string",
150                 choices: [
151                     {
152                         value: "link",
153                         label: "Link"
154                     },
155                     {
156                         value: "selector",
157                         label: "Selector"
158                     },
159                     {
160                         value: "key",
161                         label: "Keyboard F/H"
162                     }
163                 ]
164             },
165             variable: {
166                 /**
167                  * The target variable, returned either based on the name attribute,
168                  * and if absent by evaluating the expr attribute.
169                  */
170                 getter: Y.Wegas.Widget.VARIABLEDESCRIPTORGETTER,
171                 _inputex: {
172                     _type: "variableselect"
173                 }
174             }
175         }
176     });
177 });
