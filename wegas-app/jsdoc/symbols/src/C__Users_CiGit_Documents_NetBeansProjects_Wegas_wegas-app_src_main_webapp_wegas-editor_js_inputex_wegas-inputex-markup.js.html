 12 YUI.add("wegas-inputex-markup", function(Y) {
 13 
 14     var lang = Y.Lang,
 15             inputEx = Y.inputEx,
 16             MarkupField;
 17 
 18     /**
 19      * Basic string field (equivalent to the input type "text")
 20      * @class MarkupField
 21      * @extends inputEx.Field
 22      * @constructor
 23      * @param {Object} options Added options:
 24      * <ul>
 25      *	  <li>regexp: regular expression used to validate (otherwise it always validate)</li>
 26      *   <li>size: size attribute of the input</li>
 27      *   <li>maxLength: maximum size of the string field (no message display, uses the maxlength html attribute)</li>
 28      *   <li>minLength: minimum size of the string field (will display an error message if shorter)</li>
 29      *   <li>typeInvite: string displayed when the field is empty</li>
 30      *   <li>readonly: set the field as readonly</li>
 31      * </ul>
 32      */
 33     MarkupField = function(options) {
 34         MarkupField.superclass.constructor.call(this, options);
 35 
 36         if (this.options.typeInvite) {
 37             this.updateTypeInvite();
 38         }
 39     };
 40 
 41     Y.extend(MarkupField, inputEx.Field, {
 42         /**
 43          * Render an 'INPUT' DOM node
 44          * @method renderComponent
 45          */
 46         renderComponent: function() {
 47             // This element wraps the input node in a float: none div
 48             this.wrapEl = inputEx.cn('div', {className: 'MarkupField-wrapper'});
 49 
 50             // Attributes of the input field
 51             var attributes = {};
 52             attributes.id = this.divEl.id ? this.divEl.id + '-field' : Y.guid();
 53             if (this.options.size) {
 54                 attributes.size = this.options.size;
 55             }
 56             if (this.options.name) {
 57                 attributes.name = this.options.name;
 58             }
 59 
 60             // Create the node
 61             this.el = inputEx.cn('div');
 62 
 63             // Append it to the main element
 64             this.wrapEl.appendChild(this.el);
 65             this.fieldContainer.appendChild(this.wrapEl);
 66         },
 67         /**
 68          * Function to set the value
 69          * @method setValue
 70          * @param {String} value The new value
 71          * @param {boolean} [sendUpdatedEvt] (optional) Wether this setValue should fire the 'updated' event or not (default is true, pass false to NOT send the event)
 72          */
 73         setValue: function(value, sendUpdatedEvt) {
 74             this.el.innerHTML = value;
 75             
 76             // call parent class method to set style and fire "updated" event
 77             MarkupField.superclass.setValue.call(this, value, sendUpdatedEvt);
 78         }
 79     });
 80 
 81     inputEx.registerType("markup", MarkupField);                                // Register this class as "markup" type
 82 
 83 });
