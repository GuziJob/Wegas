 16     /**
 17      * @name Y.Wegas.TabView
 18      * @extends Y.TabView

 21      * @class Manage a tabview specific to Wegas

 23      */
 24     TabView = Y.Base.create("tabview", Y.TabView, [Y.WidgetChild, Y.Wegas.Widget], {
 25         /** @lends Y.Wegas.TabView# */
 26 
 27         // *** Private fields *** //
 28         /**
 29          * Reference to each used functions
 30          */
 31         handlers: null,
 32 
 33         /**
 34          * @function
 35          * @private
 36          * @description Set variables with initials values.
 37          * init widget parent.
 38          */
 39         initializer: function () {
 40             this.handlers = {};
 41             TabView.superclass.initializer.apply(this, arguments);
 42             //this.plug(Removeable);
 43         },
 44 
 45         /**
 46          * @function
 47          * @private
 48          * @description bind function to events.
 49          * When selection change on a tab, fire event "layout:resize"
 50          */
 51         bindUI: function () {
 52             TabView.superclass.bindUI.apply(this, arguments);

 54             // @fixme we notify the editor for any change, so widget can be updated

 62          * Detach all functions created by this widget.
 63          * @function

 67             for (var k in this.handlers) {
 68                 this.handlers[k].detach();

 73         /** @lends Y.Wegas.TabView# */
 74

 76          * Prefix css. Static
 77          */
 78         CSS_PREFIX: "yui3-tabview",

 80         /**
 81          * References to tab

 85         /**
 86          * Return A tab from tabview selected by id.
 87          * @function
 88          * @private
 89          * @param id
 90          * @return A tab from tabview
 91          */
 92         getTab: function (id) {
 93             return TabView.tabs[id];
 94         },
 95 
 96         /**
 97          * @function
 98          * @private
 99          * @param id
100          * @param tabViewSelector
101          * @param tabCfg
102          * @return the new tab
103          * @description create and return a tab based on a given id, the
104          *  tabview reference and the configuration of the new tab.
105          */
106         createTab: function (id, tabViewSelector, tabCfg) {
107             if (!TabView.tabs[id]) {                                            // If the tab does not exist,
108                 var tabs, tabView = Y.Widget.getByNode(tabViewSelector);        // Look for the parent
109                 tabCfg = tabCfg || {};
110                 Y.mix(tabCfg, {
111                     type: "Tab",
112                     label: id,
113                     id: id
114                 });
115                 tabs = tabView.add(tabCfg);                                     // Instantiate a new tab
116                 return tabs.item(0);
117             } else {                                                            // Otherwise,
118                 //TabView.tabs[id].setAttrs(tabCfg);                              // update the tab config
119             }
120             return TabView.tabs[id];
121         },
122 
123         /**
124          * @function
125          * @private
126          * @param id
127          * @param tabViewSelector
128          * @param tabCfg
129          * @param widgetCfg
130          * @param fn
131          * @description Load a tab corresponding with the given parameters.
132          */
133         findTabAndLoadWidget: function (id, tabViewSelector, tabCfg, widgetCfg, fn) {
134             var nTab = TabView.createTab(id, tabViewSelector, tabCfg);          // create a new one
135 
136             nTab.removeAll().each(function (i) {
137                 i.destroy();                                                    // Empty it
138             });
139             nTab.get("panelNode").empty();                                      // @fixme since the above method is not enough
140             nTab.load(widgetCfg, fn);                                           // Load target widget
141             nTab.set("selected", 2);
142             nTab.plug(Removeable);
143         }
144 
145     });

148     /**
149      * Extension enabling a Tab to be a parent of another Widget.
150      *
151      * @modified from original WidgetParent module
152      *
153      * @module widget-parent
154      */
155     function Parent (config) {
156         //Y.WidgetParent.call(this, config);
157         this.publish("addChild", {
158             defaultTargetOnly: true,
159             defaultFn: this._defAddChildFn
160         });
161         this.publish("removeChild", {
162             defaultTargetOnly: true,
163             defaultFn: this._defRemoveChildFn
164         });
165 
166         this._items = [];
167 
168         var children,
169                 handle;
170 
171         if (config && config.children) {
172 
173             children = config.children;
174 
175             handle = this.after("initializedChange", function (e) {
176                 this._add(children);
177                 handle.detach();
178             });
179 
180         }
181 
182         //  Widget method overlap
183         Y.after(this._renderChildren, this, "renderUI");
184         Y.after(this._bindUIParent, this, "bindUI");
185 
186         //        this.after("selectionChange", this._afterSelectionChange);
187         //        this.after("selectedChange", this._afterParentSelectedChange);
188         //        this.after("activeDescendantChange", this._afterActiveDescendantChange);
189 
190         this._hDestroyChild = this.after("*:destroy", this._afterDestroyChild);
191         this.after("*:focusedChange", this._updateActiveDescendant);
192         ;
193     }
194 
195     //Y.extend(Parent, Y.WidgetParent);
196     Y.mix(Parent, Y.WidgetParent);
197     Y.augment(Parent, Y.WidgetParent);
198     Parent.ATTRS = {};
199     Y.mix(Parent.ATTRS, Y.WidgetParent.ATTRS);
200     delete Parent.ATTRS.selected;

202     /**
203      * @name Y.Wegas.Tab
204      * @extends Y.Tab
205      * @borrows Y.Wegas.Widget, Parent, Y.WidgetChild
206      * @class class to manage a tabspecific to Wegas

210     Tab = Y.Base.create("tab", Y.Tab, [Y.Wegas.Widget, Parent, Y.WidgetChild], {
211         /** @lends Y.Wegas.Tab# */
212 
213         // *** Private Fields *** //
214         /**
215          * Array of widget items.
216          */
217         _witems: null,
218 
219         // *** Lifecycle Methods *** //
220         /**
221          * @function
222          * @private
223          * @description Set variables with initials values.
224          * call explicitly initializer method of widget parent.
225          * assign this tab in owner TabView.
226          */
227         initializer: function (cfg) {
228             Tab.superclass.initializer.apply(this, arguments);
229             TabView.tabs[cfg.id || cfg.label] = this;
230             this._witems = [];
231 
232             //this.plug(Closable);
233         },
234 
235         /**
236          * @function
237          * @private
238          * @description call render method of widget parent.
239          */
240         renderUI: function () {
241             Tab.superclass.renderUI.apply(this, arguments);
242         },
243 
244         /**
245          * @function
246          * @private
247          * @description call sync method of widget parent.
248          */
249         syncUI: function () {
250             Tab.superclass.syncUI.apply(this, arguments);
251         },
252 
253         /**
254          * @function
255          * @private
256          * @description delete Tab in owner Tabview.
257          *  call explicitly destructor method of widget parent.
258          */
259         destructor: function () {
260             delete TabView.tabs[this.get("id")];
261             Tab.superclass.destructor.apply(this, arguments);
262         },
263 
264         // *** Private Methods *** //
265         /**
266          * @function
267          * @private
268          * @param cfg
269          * @param callback
270          * @description Retrieves the given widget configuration and add it
271          * to the tab
272          */
273         load: function (cfg, callback) {
274             Y.Wegas.Widget.use(cfg, Y.bind(function (cfg, callback) {          // Load the subpage dependencies
275                 var widgets = this.add(cfg);                                    // Render the subpage
276                 if (callback) {
277                     callback(widgets.item(0));                                  // Trigger the callback
278                 }
279             }, this, cfg, callback));
280         },
281 
282         /**
283          * @function
284          * @private
285          * @description Override Y.WidgetParent to render children in the panel node;
286          * to the tab
287          */
288         _renderChildren: function () {
289             var renderTo = this._childrenContainer || this.get("panelNode");    // @modified
290 
291             this._childrenContainer = renderTo;
292 
293             this.each(function (child) {
294                 this._witems.push(child);                                       // @modified
295                 child.render(renderTo);
296             }, this);
297         },
298 
299         /**
300          * @function
301          * @private
302          * @param index
303          * @description return the widget item at given index
304          */
305         witem: function (index) {
306             return this._witems[index];
307         }
308 
309     }, {

311 
312         /** @lends Y.Wegas.Tab */

319          * <ul>
320          *    <li>Content: Overrides the panelNode management</li>
321          * </ul>
322          */
323         ATTRS: {
324             /**
325              * Overrides the panelNode management
326              */
327             content: {
328                 setter: function () {
329                 }
330             }
331         }
332     });

409     /**
410      * Plugin to toggle visibility of a tab
411      *
412      * @name Y.Plugin.LayoutToggleTab
413      * @extends Y.Plugin.Base
414      * @class plugin to toggle visibility of a tab
415      * @constructor
416      */
417     var LayoutToggleTab = function () {
418         LayoutToggleTab.superclass.constructor.apply(this, arguments);
419     };
420     Y.extend(LayoutToggleTab, Y.Plugin.Base, {
421         /** @lends Y.Wegas.LayoutToggleTab# */
422 
423         // *** Private fields *** //
424         /**
425          * Html template added in host's contentbox
426          */
427         REMOVE_TEMPLATE: '<a class="yui3-tab-remove" title="remove tab">x</a>',
428 
429         /**
430          * @function
431          * @private
432          * @description if "removeChild" is fired by host, hide tab.
433          *  If "addChild" is fired by host, show tab.
434          */
435         initializer: function () {
436             this.onHostEvent("removeChild", function () {
437                 if (this.get("host").size() === 1) {
438                     Y.Wegas.app.widget.hidePosition("right");
439                 }
440             });
441             this.onHostEvent("addChild", function () {
442                 if (this.get("host").isEmpty()) {
443                     Y.Wegas.app.widget.showPosition("right");
444                 }
445             });
446         }
447     }, {
448         NS: "LayoutToggleTab",
449         NAME: "LayoutToggleTab"
450     });
