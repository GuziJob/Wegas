 12 YUI.add('wegas-logger', function (Y) {
 13     "use strict";
 14 
 15     /**
 16      * @name Y.Wegas.Logger
 17      * @class Displays a Y.console in a widget
 18      * @constructor
 19      * @extends Y.Widget
 20      * @augments Y.WidgetChild
 21      * @augments Y.Wegas.Widget
 22      */
 23     var Logger = Y.Base.create("wegas-logger", Y.Widget, [Y.WidgetChild, Y.Wegas.Widget], {
 24         /** @lends Y.Wegas.Logger# */
 25 
 26         /**
 27          * @field Y.Console The displayed console instance
 28          * @private
 29          */
 30         console: null,
 31 
 32         /**
 33          * @function
 34          * @private
 35          */
 36         renderUI: function () {
 37             Y.log('renderUI()', 'log', "Wegas.Logger");
 38 
 39             var node = Y.Node.create('<div style="height:50px"></div>');
 40             this.get('contentBox').appendChild(node);
 41 
 42             this.console = new Y.Console({
 43                 logSource: Y.Global,
 44                 plugins: [ Y.Plugin.ConsoleFilters ],
 45                 width: '100%',
 46                 style: 'block'                                                  // 'inline'
 47             //height: '300px',
 48             //height: '98%',
 49             //newestOnTop: false,
 50             //logLevel :'log'
 51             }).render(node);
 52         }
 53     });
 54     Y.namespace('Wegas').Logger = Logger;
 55 
 56 });
