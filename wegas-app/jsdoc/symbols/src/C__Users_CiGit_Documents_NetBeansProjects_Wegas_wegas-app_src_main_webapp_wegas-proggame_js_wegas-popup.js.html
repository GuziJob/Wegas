 13 YUI.add('wegas-popup', function(Y) {
 14 
 15     /**
 16      *  @class Show a message when the host widget is rendered, useful for welcome messages
 17      *  messages
 18      *  @name Y.Plugin.Popup
 19      *  @extends Y.Plugin.Base
 20      *  @constructor
 21      *
 22      *  @deprecated Use Y.Wegas.Panel instead
 23      */
 24     var Popup = Y.Base.create("wegas-popup", Y.Plugin.Base, [Y.Wegas.Plugin, Y.Wegas.Editable], {
 25         initializer: function() {
 26             this.afterHostEvent("render", function() {
 27                 this.get("host").showMessage("info", this.get("content"));
 28             });
 29         }
 30     }, {
 31         NS: "Popup",
 32         NAME: "Popup",
 33         ATTRS: {
 34             content: {
 35                 type: "string",
 36                 format: "html",
 37                 _inputex: {
 38                     label: "Welcome message"
 39                 }
 40             }
 41         }
 42     });
 43     Y.Plugin.Popup = Popup;
 44 });
